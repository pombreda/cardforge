42          File file = new File("res/quest/themes/");
43  
44          if (!file.exists())
45              throw new RuntimeException("GenerateThemeDeck : getThemeNames error -- file not found -- filename is "
46                      + file.getAbsolutePath());
47  
48          if (!file.isDirectory())
49              throw new RuntimeException("GenerateThemeDeck : getThemeNames error -- not a direcotry -- "
50                      + file.getAbsolutePath());
51  
52          String[] fileList = file.list();
53          for (int i = 0; i < fileList.length; i++) {
54              if (fileList[i].endsWith(".thm"))
55                  ltNames.add(fileList[i].substring(0, fileList[i].indexOf(".thm")));

61      /**
62       * <p>getThemeDeck.</p>
63       *
64       * @param ThemeName a {@link java.lang.String} object.
65       * @param Size a int.
66       * @return a {@link forge.CardList} object.
67       */
68      public CardList getThemeDeck(String ThemeName, int Size) {
69          CardList tDeck = new CardList();
70  
71          ArrayList<Grp> Groups = new ArrayList<Grp>();
72  
73          Map<String, Integer> CardCounts = new HashMap<String, Integer>();
74  
75          String s = "";
76          int BLandPercentage = 0;
77          boolean Testing = false;
78  
79          // read theme file
80          String tFileName = "res/quest/themes/" + ThemeName + ".thm";
81          File tFile = new File(tFileName);
82          if (!tFile.exists())
83              throw new RuntimeException("GenerateThemeDeck : getThemeDeck -- file not found -- filename is " + tFile.getAbsolutePath());
84  
85          try {
86              in = new BufferedReader(new FileReader(tFile));
87          } catch (Exception ex) {
88              ErrorViewer.showError(ex, "File \"%s\" exception", tFile.getAbsolutePath());
89              throw new RuntimeException("GenerateThemeDeck : getThemeDeck -- file exception -- filename is " + tFile.getPath());
90          }
91  
92          s = readLine();
93          while (!s.equals("End")) {
94              if (s.startsWith("[Group")) {
95                  Grp G = new Grp();
96  
97                  String ss[] = s.replaceAll("[\\[\\]]", "").split(" ");
98                  for (int i = 0; i < ss.length; i++) {
99                      if (ss[i].startsWith("Percentage")) {
100                         String p = ss[i].substring("Percentage".length() + 1);
101                         G.Percentage = Integer.parseInt(p);
102                     }
103                     if (ss[i].startsWith("MaxCnt")) {
104                         String m = ss[i].substring("MaxCnt".length() + 1);
105                         G.MaxCnt = Integer.parseInt(m);
106                     }
107                 }
108 
109                 s = readLine();
110                 while (!s.equals("[/Group]")) {
111                     G.Cardnames.add(s);
112                     CardCounts.put(s, 0);
113 
114                     s = readLine();
115                 }
116 
117                 Groups.add(G);
118             }
119 
120             if (s.startsWith("BasicLandPercentage"))
121                 BLandPercentage = Integer.parseInt(s.substring("BasicLandPercentage".length() + 1));
122 
123             if (s.equals("Testing"))
124                 Testing = true;
125 
126             s = readLine();
127         }
128 
129         try {
130             in.close();
131         } catch (IOException ex) {
132             ErrorViewer.showError(ex, "File \"%s\" exception", tFile.getAbsolutePath());
133             throw new RuntimeException("GenerateThemeDeck : getThemeDeck -- file exception -- filename is " + tFile.getPath());
134         }
135 
136         String tmpDeck = "";
137 
138         // begin assigning cards to the deck
139         Random r = MyRandom.random;
140 
141         for (int i = 0; i < Groups.size(); i++) {
142             Grp G = Groups.get(i);
143             float p = (float) ((float) G.Percentage * .01);
144             int GrpCnt = (int) (p * (float) Size);
145             int cnSize = G.Cardnames.size();
146             tmpDeck += "Group" + i + ":" + GrpCnt + "\n";
147 
148             for (int j = 0; j < GrpCnt; j++) {
149                 s = G.Cardnames.get(r.nextInt(cnSize));
150 
151                 int lc = 0;
152                 while (CardCounts.get(s) >= G.MaxCnt || lc > Size) // don't keep looping forever
153                 {
154                     s = G.Cardnames.get(r.nextInt(cnSize));
155                     lc++;
156                 }
157                 if (lc > Size)
158                     throw new RuntimeException("GenerateThemeDeck : getThemeDeck -- looped too much -- filename is " + tFile.getAbsolutePath());
159 
160                 int n = CardCounts.get(s);
161                 tDeck.add(AllZone.getCardFactory().getCard(s, AllZone.getComputerPlayer()));
162                 CardCounts.put(s, n + 1);
163                 tmpDeck += s + "\n";
164 
165             }
166         }
167 
168         int numBLands = 0;
169         if (BLandPercentage > 0)    // if theme explicitly defines this
170         {
171             float p = (float) ((float) BLandPercentage * .01);
172             numBLands = (int) (p * (float) Size);
173         } else     // otherwise, just fill in the rest of the deck with basic lands
174             numBLands = Size - tDeck.size();
175 
176         tmpDeck += "numBLands:" + numBLands + "\n";
177 
178         if (numBLands > 0)    // attempt to optimize basic land counts according to color representation
179         {
180             CCnt ClrCnts[] = {new CCnt("Plains", 0),
181                     new CCnt("Island", 0),
182                     new CCnt("Swamp", 0),
183                     new CCnt("Mountain", 0),
184                     new CCnt("Forest", 0)};
185 
186             // count each instance of a color in mana costs
187             // TODO: count hybrid mana differently?
188             for (int i = 0; i < tDeck.size(); i++) {
189                 String mc = tDeck.get(i).getManaCost();
190 
191                 for (int j = 0; j < mc.length(); j++) {
192                     char c = mc.charAt(j);
193 
194                     if (c == 'W')
195                         ClrCnts[0].Count++;
196                     else if (c == 'U')
197                         ClrCnts[1].Count++;
198                     else if (c == 'B')
199                         ClrCnts[2].Count++;
200                     else if (c == 'R')
201                         ClrCnts[3].Count++;
202                     else if (c == 'G')
203                         ClrCnts[4].Count++;
204                 }
205             }
206 
207             int totalColor = 0;
208             for (int i = 0; i < 5; i++) {
209                 totalColor += ClrCnts[i].Count;
210                 tmpDeck += ClrCnts[i].Color + ":" + ClrCnts[i].Count + "\n";
211             }
212 
213             tmpDeck += "totalColor:" + totalColor + "\n";
214 
215             for (int i = 0; i < 5; i++) {
216                 if (ClrCnts[i].Count > 0) {    // calculate number of lands for each color
217                     float p = (float) ClrCnts[i].Count / (float) totalColor;
218                     int nLand = (int) ((float) numBLands * p);
219                     tmpDeck += "numLand-" + ClrCnts[i].Color + ":" + nLand + "\n";
220 
221                     CardCounts.put(ClrCnts[i].Color, 2);
222                     for (int j = 0; j < nLand; j++)
223                         tDeck.add(AllZone.getCardFactory().getCard(ClrCnts[i].Color, AllZone.getComputerPlayer()));
224                 }
225             }
226         }
227         tmpDeck += "DeckSize:" + tDeck.size() + "\n";
228 
229         if (tDeck.size() < Size) {
230             int diff = Size - tDeck.size();
231 
232             for (int i = 0; i < diff; i++) {
233                 s = tDeck.get(r.nextInt(tDeck.size())).getName();
234 
235                 while (CardCounts.get(s) >= 4)
236                     s = tDeck.get(r.nextInt(tDeck.size())).getName();
237 
238                 int n = CardCounts.get(s);
239                 tDeck.add(AllZone.getCardFactory().getCard(s, AllZone.getComputerPlayer()));
240                 CardCounts.put(s, n + 1);
241                 tmpDeck += "Added:" + s + "\n";
242             }
243         } else if (tDeck.size() > Size) {
244             int diff = tDeck.size() - Size;
245 
246             for (int i = 0; i < diff; i++) {
247                 Card c = tDeck.get(r.nextInt(tDeck.size()));
248 
249                 while (c.isBasicLand())
250                     c = tDeck.get(r.nextInt(tDeck.size()));
251 
252                 tDeck.remove(c);
253                 tmpDeck += "Removed:" + s + "\n";
254             }
255         }
256 
257         tmpDeck += "DeckSize:" + tDeck.size() + "\n";
258         if (Testing)
259             ErrorViewer.showError(tmpDeck);
260 
261         return tDeck;
262     }
