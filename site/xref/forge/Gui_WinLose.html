117     private void setup() {
118         AllZone.getGameInfo().clearColorChanges();
119         QuestMatchState winLose = Constant.Runtime.matchState;
120         Phase.setGameBegins(0);
121         //3 is the match length, 3 is the number of games
122         //disable buttons if match is up, or human player won 2 or lost 2 games already
123         if ((winLose.countWinLose() == 3) || (winLose.getWin() == 2) || (winLose.getLose() == 2)) {
124 //      editDeckButton.setEnabled(false);
125             continueButton.setEnabled(false);
126             quitButton.grabFocus();
127         }
128 
129         if (winLose.getWin() == 2)
130             restartButton.setEnabled(false);
131 
132         //show Wins and Loses
133         statsLabel.setText(ForgeProps.getLocalized(WINLOSE_TEXT.WON) + winLose.getWin() + ForgeProps.getLocalized(WINLOSE_TEXT.LOST) + winLose.getLose());
134 
135         //show "You Won" or "You Lost"
136         if (winLose.didWinRecently()) {
137             titleLabel.setText(ForgeProps.getLocalized(WINLOSE_TEXT.WIN));
138 
139             int game = 0;
140             if (winLose.getWinTurns()[0] != 0)
141                 game = 1;
142             int turn = AllZone.getPhase().getTurn();
143             if (AllZone.getGameInfo().isComputerStartedThisGame())
144                 turn--;
145 
146             if (turn < 1)
147                 turn = 1;
148 
149             winLose.setWinTurn(game, turn);
150             winLose.setMulliganedToZero(game, AllZone.getGameInfo().getHumanMulliganedToZero());
151 
152             //winLose.setWinTurn(winLose.countWinLose()-1, AllZone.getPhase().getTurn());
153 
154             //System.out.println("CountwinLose:" + winLose.countWinLose());
155             //System.out.println("You won by turn: " + AllZone.getPhase().getTurn());
156         } else {
157             titleLabel.setText(ForgeProps.getLocalized(WINLOSE_TEXT.LOSE));
158             //System.out.println("You lost by turn: " + AllZone.getPhase().getTurn());
159         }

302     /**
303      * <p>getWinText.</p>
304      *
305      * @param creds a long.
306      * @param winLose a {@link forge.quest.data.QuestMatchState} object.
307      * @param q a {@link forge.quest.data.QuestData} object.
308      * @return a {@link java.lang.String} object.
309      */
310     private String getWinText(long creds, QuestMatchState winLose, forge.quest.data.QuestData q) {
311         // TODO use q.qdPrefs to write bonus credits in prefs file
312         StringBuilder sb = new StringBuilder();
313         String[] wins = winLose.getWinMethods();
314 
315         sb.append("<html>");
316 
317 
318         for (String s : wins) {
319             if (s != null) {
320                 sb.append("Alternate win condition: ");
321                 sb.append("<u>");
322                 sb.append(s);
323                 sb.append("</u>");
324                 sb.append("! Bonus: <b>+");
325 
326                 if (s.equals("Poison Counters"))
327                     sb.append(QuestPreferences.getMatchRewardPoisonWinBonus());
328                 else if (s.equals("Milled"))
329                     sb.append(QuestPreferences.getMatchRewardMilledWinBonus());
330                 else if (s.equals("Battle of Wits") ||
331                         s.equals("Felidar Sovereign") || s.equals("Helix Pinnacle") || s.equals("Epic Struggle") ||
332                         s.equals("Door to Nothingness") || s.equals("Barren Glory") || s.equals("Near-Death Experience") ||
333                         s.equals("Mortal Combat") || s.equals("Test of Endurance")) {
334 
335                     sb.append(QuestPreferences.getMatchRewardAltWinBonus());
336                 }
337 
338                 sb.append(" credits</b>.<br>");
339             }
340         }
341 
342         int[] winTurns = winLose.getWinTurns();
343 
344         for (int i : winTurns) {
345             System.out.println("Quest, won by turn:" + i);
346 
347             if (i != 0) {
348                 int turnCredits = 0;
349 
350                 if (i == 1) {
351                     sb.append("Won in one turn!");
352                     turnCredits += QuestPreferences.getMatchRewardWinFirst();
353                 } else if (i <= 5) {
354                     sb.append("Won by turn 5!");
355                     turnCredits += QuestPreferences.getMatchRewardWinByFifth();
356                 } else if (i <= 10) {
357                     sb.append("Won by turn 10!");
358                     turnCredits += QuestPreferences.getMatchRewardWinByTen();
359                 } else if (i <= 15) {
360                     sb.append("Won by turn 15!");
361                     turnCredits += QuestPreferences.getMatchRewardWinByFifteen();
362                 }
363                 if (turnCredits > 0)
364                     sb.append(" Bonus: <b>+").append(turnCredits).append(" credits</b>.<br>");
365             }
366         }
367 
368         boolean[] mulliganedToZero = winLose.getMulliganedToZero();
369 
370         for (boolean b : mulliganedToZero) {
371             if (b) {
372                 sb.append("Mulliganed to zero and still won! Bonus: <b>+");
373                 sb.append(QuestPreferences.getMatchMullToZero()).append(" credits</b>.<br>");
374             }
375         }
376 
377         if (winLose.getLose() == 0)
378             sb.append("You have not lost once! Bonus: <b>+");
379         sb.append(QuestPreferences.getMatchRewardNoLosses()).append(" credits</b>.<br>");
380 
381         if (q.getInventory().getItemLevel("Estates") == 1)
382             sb.append("Estates bonus: <b>10%</b>.<br>");
383         else if (q.getInventory().getItemLevel("Estates") == 2)
384             sb.append("Estates bonus: <b>15%</b>.<br>");
385         else if (q.getInventory().getItemLevel("Estates") == 3)
386             sb.append("Estates bonus: <b>20%</b>.<br>");
387 
388         sb.append("You have earned <b>" + creds + " credits</b> in total.");
389 
390         sb.append("</html>");
391         return sb.toString();
392     }
