64      private int getAverageTimePerCard() {
65          int aTime = 0;
66          int nz = 10;
67  
68          if (tptr > 9)
69              tptr = 0;
70  
71          times[tptr] = System.currentTimeMillis() - lTime;
72          lTime = System.currentTimeMillis();
73  
74          int tTime = 0;
75          for (int i = 0; i < 10; i++) {
76              tTime += times[i];
77              if (times[i] == 0)
78                  nz--;
79          }
80          aTime = tTime / nz;
81  
82          tptr++;
83  
84          return aTime;
85      }

264         BufferedInputStream in;
265         BufferedOutputStream out;

267         File base = ForgeProps.getFile(IMAGE_BASE);

269         Random r = MyRandom.random;

283             byte[] buf = new byte[1024];
284             int len;
285             System.out.println("basedir: " + base);
286             for (update(0); card < cards.length && !cancel; update(card + 1)) {
287                 try {
288                     String url = cards[card].url;
289                     String cName;
290                     if (cards[card].name.substring(0, 3).equals("[T]")) {
291                         base = ForgeProps.getFile(IMAGE_TOKEN);
292                         cName = cards[card].name.substring(3, cards[card].name.length());
293                     } else {

302                     //test for folder existence
303                     File test = new File(base, cards[card].folder);
304                     if (!test.exists()) {
305                         // create folder
306                         if (!test.mkdir())
307                             System.out.println("Can't create folder" + cards[card].folder);
308                     }
309 
310                     try {
311                         in = new BufferedInputStream(new URL(url).openConnection(p).getInputStream());
312                         out = new BufferedOutputStream(new FileOutputStream(f));
313 
314                         while ((len = in.read(buf)) != -1) {
315                             //user cancelled
316                             if (cancel) {
317                                 in.close();
318                                 out.flush();
319                                 out.close();
320 
321                                 //delete what was written so far
322                                 f.delete();
323 
324                                 return;
325                             }//if - cancel
326 
327                             out.write(buf, 0, len);
328                         }//while - read and write file
329 
330                         in.close();
331                         out.flush();
332                         out.close();
333                     } catch (MalformedURLException mURLe) {
334                         System.out.println("Error - possibly missing URL for: " + cards[card].name);
335                         //Log.error("LQ Pictures", "Malformed URL for: "+cards[card].name, mURLe);
336                     }
337                 } catch (FileNotFoundException fnfe) {
338                     System.out.println("Error - the LQ picture for " + cards[card].name + " could not be found on the server. [" + cards[card].url + "] - " + fnfe.getMessage());
339                 } catch (Exception ex) {
340                     Log.error("LQ Pictures", "Error downloading pictures", ex);
341                 }
342 
343                 // pause
344 
345                 try {
346                     Thread.sleep(r.nextInt(750) + 420);
347                 } catch (InterruptedException e) {
348                     Log.error("LQ Set Pictures", "Sleep Error", e);
349                 }
