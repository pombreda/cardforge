102     /** {@inheritDoc} */
103     public void setCard(Card card) {
104         cdLabel1.setText("");
105         cdLabel2.setText("");
106         cdLabel3.setText("");
107         cdLabel4.setText("");
108         cdLabel5.setText("");
109         cdLabel6.setText("");
110         cdLabel7.setText("");
111         cdLabel7.setOpaque(false);
112         cdLabel7.setBorder(null);
113         cdArea.setText("");
114         setBorder(GuiDisplayUtil.getBorder(card));
115 
116         this.card = card;
117         if (card == null) return;
118 
119         boolean faceDown = card.isFaceDown() && card.getController() != AllZone.getHumanPlayer();
120         if (!faceDown) {
121             if (card.isLand()) cdLabel1.setText(card.getName());
122             else cdLabel1.setText(card.getName() + "  - " + card.getManaCost());
123         } else cdLabel1.setText("Morph");
124 
125         if (!faceDown) cdLabel2.setText(GuiDisplayUtil.formatCardType(card));
126         else cdLabel2.setText("Creature");
127 
128         if (card.isCreature()) {
129             cdLabel3.setText(card.getNetAttack() + " / " + card.getNetDefense());
130             cdLabel4.setText("Damage: " + card.getDamage() + " Assigned Damage: " + card.getTotalAssignedDamage());
131         }
132         if (card.isPlaneswalker()) cdLabel4.setText("Assigned Damage: " + card.getTotalAssignedDamage());
133 
134         cdLabel5.setText("Card ID  " + card.getUniqueNumber());
135 
136         //rarity and set of a face down card should not be visible to the opponent
137         if (!card.isFaceDown() || card.getController().isHuman()) cdLabel7.setText(card.getCurSetCode());
138 
139         if (!cdLabel7.getText().equals("")) {
140             cdLabel7.setOpaque(true);
141             String csr = card.getCurSetRarity();
142             if (csr.equals("Common") || csr.equals("Land")) {
143                 cdLabel7.setBackground(Color.BLACK);
144                 cdLabel7.setForeground(Color.WHITE);
145                 cdLabel7.setBorder(BorderFactory.createLineBorder(Color.WHITE));
146             } else if (csr.equals("Uncommon")) {
147                 cdLabel7.setBackground(Color.LIGHT_GRAY);
148                 cdLabel7.setForeground(Color.BLACK);
149                 cdLabel7.setBorder(BorderFactory.createLineBorder(Color.BLACK));
150             } else if (csr.equals("Rare")) {
151                 cdLabel7.setBackground(Color.YELLOW);
152                 cdLabel7.setForeground(Color.BLACK);
153                 cdLabel7.setBorder(BorderFactory.createLineBorder(Color.BLACK));
154             } else if (csr.equals("Mythic")) {
155                 cdLabel7.setBackground(Color.RED);
156                 cdLabel7.setForeground(Color.BLACK);
157                 cdLabel7.setBorder(BorderFactory.createLineBorder(Color.BLACK));
158             }
159             //cdLabel7.setText(card.getCurSetCode());
160         }
161 
162         //fill the card text
163 
164         StringBuilder area = new StringBuilder();
165 
166         //Token
167         if (card.isToken()) area.append("Token");
168 
169         if (!faceDown) {
170             //card text
171             if (area.length() != 0) area.append("\n");
172             String text = card.getText();
173             //LEVEL [0-9]+-[0-9]+
174             //LEVEL [0-9]+\+
175 
176             String regex = "LEVEL [0-9]+-[0-9]+ ";
177             text = text.replaceAll(regex, "$0\r\n");
178 
179             regex = "LEVEL [0-9]+\\+ ";
180             text = text.replaceAll(regex, "\r\n$0\r\n");
181 
182             //displays keywords that have dots in them a little better:
183             regex = "\\., ";
184             text = text.replaceAll(regex, ".\r\n");
185 
186             area.append(text);
187         }
188 
189         //counter text
190         Counters[] counters = Counters.values();
191         for (Counters counter : counters) {
192             if (card.getCounters(counter) != 0) {
193                 if (area.length() != 0) area.append("\n");
194                 area.append(counter.getName() + " counters: ");
195                 area.append(card.getCounters(counter));
196             }
197         }
198 
199         // Regeneration Shields
200         int regenShields = card.getShield();
201         if (regenShields > 0) {
202             if (area.length() != 0) area.append("\n");
203             area.append("Regeneration Shield(s): ").append(regenShields);
204         }
205 
206         // Damage Prevention
207         int preventNextDamage = card.getPreventNextDamage();
208         if (preventNextDamage > 0) {
209             area.append("\n");
210             area.append("Prevent the next ").append(preventNextDamage).append(" damage that would be dealt to ");
211             area.append(card.getName()).append(" it this turn.");
212         }
213 
214         //top revealed
215         if (card.hasKeyword("Play with the top card of your library revealed.")
216                 && !card.getTopCardName().equals("")) {
217             area.append("\r\nTop card: ");
218             area.append(card.getTopCardName());
219         }
220 
221         //chosen type
222         if (card.getChosenType() != "") {
223             if (area.length() != 0) area.append("\n");
224             area.append("(chosen type: ");
225             area.append(card.getChosenType());
226             area.append(")");
227         }
228 
229         //chosen color
230         if (card.getChosenColor() != "") {
231             if (area.length() != 0) area.append("\n");
232             area.append("(chosen color: ");
233             area.append(card.getChosenColor());
234             area.append(")");
235         }
236 
237         //named card
238         if (card.getNamedCard() != "") {
239             if (area.length() != 0) area.append("\n");
240             area.append("(named card: ");
241             area.append(card.getNamedCard());
242             area.append(")");
243         }
244 
245         //equipping
246         if (card.getEquipping().size() > 0) {
247             if (area.length() != 0) area.append("\n");
248             area.append("=Equipping ");
249             area.append(card.getEquipping().get(0));
250             area.append("=");
251         }
252 
253         //equipped by
254         if (card.getEquippedBy().size() > 0) {
255             if (area.length() != 0) area.append("\n");
256             area.append("=Equipped by ");
257             for (Iterator<Card> it = card.getEquippedBy().iterator(); it.hasNext(); ) {
258                 area.append(it.next());
259                 if (it.hasNext()) area.append(", ");
260             }
261             area.append("=");
262         }
263 
264         //enchanting
265         if (card.getEnchanting().size() > 0) {
266             if (area.length() != 0) area.append("\n");
267             area.append("*Enchanting ");
268             if (card.getEnchanting().get(0).isFaceDown()
269                     && card.getEnchanting().get(0).getController().isComputer()) {
270                 area.append("Morph (");
271                 area.append(card.getEnchanting().get(0).getUniqueNumber());
272                 area.append(")");
273             } else {
274                 area.append(card.getEnchanting().get(0));
275             }
276             area.append("*");
277         }
278 
279         //enchanted by
280         if (card.getEnchantedBy().size() > 0) {
281             if (area.length() != 0) area.append("\n");
282             area.append("*Enchanted by ");
283             for (Iterator<Card> it = card.getEnchantedBy().iterator(); it.hasNext(); ) {
284                 area.append(it.next());
285                 if (it.hasNext()) area.append(", ");
286             }
287             area.append("*");
288         }
289 
290         //controlling
291         if (card.getGainControlTargets().size() > 0) {
292             if (area.length() != 0) area.append("\n");
293             area.append("+Controlling: ");
294             for (Iterator<Card> it = card.getGainControlTargets().iterator(); it.hasNext(); ) {
295                 area.append(it.next());
296                 if (it.hasNext()) area.append(", ");
297             }
298             area.append("+");
299         }
300 
301         //cloned via
302         if (card.getCloneOrigin() != null) {
303             if (area.length() != 0) area.append("\n");
304             area.append("^Cloned via: ");
305             area.append(card.getCloneOrigin().getName());
306             area.append("^");
307         }
308 
309         //Imprint
310         if (!card.getImprinted().isEmpty()) {
311             if (area.length() != 0) area.append("\n");
312             area.append("^Imprinting: ");
313             for (Iterator<Card> it = card.getImprinted().iterator(); it.hasNext(); ) {
314                 area.append(it.next());
315                 if (it.hasNext()) area.append(", ");
316             }
317             area.append("^");
318         }
319 
320         //uncastable
321         if (card.isUnCastable()) {
322             if (area.length() != 0) area.append("\n");
323             area.append("This card can't be cast.");
324         }
325 
326         if (card.hasAttachedCards()) {
327             if (area.length() != 0) area.append("\n");
328             Card[] cards = card.getAttachedCards();
329             area.append("=Attached: ");
330             for (Card c : cards) {
331                 area.append(c.getName());
332                 area.append(" ");
333             }
334             area.append("=");
335         }
336 
337         cdArea.setText(area.toString());
338     }
