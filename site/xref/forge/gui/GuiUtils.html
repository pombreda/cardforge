27       * This method takes a collection of components and sets the width of each component
28       * to the maximum of the collection
29       *
30       * @param components a {@link java.util.Collection} object.
31       */
32      public static void setWidthToMax(Collection<Component> components) {
33          int maxWidth = 0;
34  
35          for (Component c : components) {
36              if (c.getPreferredSize().getWidth() > maxWidth) {
37                  maxWidth = (int) c.getPreferredSize().getWidth();
38              }
39          }
40  
41          for (Component c : components) {
42              c.setMinimumSize(new Dimension(maxWidth, (int) c.getPreferredSize().getHeight()));
43              c.setMaximumSize(new Dimension(maxWidth, (int) c.getPreferredSize().getHeight()));
44              c.setPreferredSize(new Dimension(maxWidth, (int) c.getPreferredSize().getHeight()));
45          }
46  
47      }

193     /**
194      * <p>getChoices.</p>
195      *
196      * @param message a {@link java.lang.String} object.
197      * @param min a int.
198      * @param max a int.
199      * @param choices a T object.
200      * @param <T> a T object.
201      * @return a {@link java.util.List} object.
202      */
203     public static <T> List<T> getChoices(String message, int min, int max, T... choices) {
204         ListChooser<T> c = new ListChooser<T>(message, min, max, choices);
205         final JList list = c.getJList();
206         list.addListSelectionListener(new ListSelectionListener() {
207             public void valueChanged(ListSelectionEvent ev) {
208                 if (list.getSelectedValue() instanceof Card && AllZone.getDisplay() != null) {
209                     AllZone.getDisplay().setCard((Card) list.getSelectedValue());
210                 }
211             }
212         });
213         c.show();
214         return c.getSelectedValues();
215     }//getChoice()

217     /**
218      * Centers a frame on the screen based on its current size
219      *
220      * @param frame a fully laid-out frame
221      */
222     public static void centerFrame(Frame frame) {
223         Dimension screen = frame.getToolkit().getScreenSize();
224         Rectangle bounds = frame.getBounds();
225         bounds.width = frame.getWidth();
226         bounds.height = frame.getHeight();
227         bounds.x = (screen.width - bounds.width) / 2;
228         bounds.y = (screen.height - bounds.height) / 2;
229         frame.setBounds(bounds);
230     }
