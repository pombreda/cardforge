40  /**
41   * Label UI delegate that supports multiple lines and line wrapping. Hard line
42   * breaks (<code>\n</code>) are preserved. If the dimensions of the label is too
43   * small to fit all content, the string will be clipped and "..." appended to
44   * the end of the visible text (similar to the default behavior of
45   * <code>JLabel</code>). If used in conjunction with a {@link MultiLineLabel},
46   * text alignment (horizontal and vertical) is supported. The UI delegate can be
47   * used on a regular <code>JLabel</code> if text alignment isn't required. The
48   * default alignment, left and vertically centered, will then be used.
49   * <p/>
50   * Example of usage:
51   * <p/>
52   * <pre>
53   * JLabel myLabel = new JLabel();
54   * myLabel.setUI(MultiLineLabelUI.labelUI);
55   * myLabel.setText(&quot;A long label that will wrap automatically.&quot;);
56   * </pre>
57   * <p/>
58   * <p/>
59   * The line and wrapping support is implemented without using a
60   * <code>View</code> to make it easy for subclasses to add custom text effects
61   * by overriding {@link #paintEnabledText(JLabel, Graphics, String, int, int)}
62   * and {@link #paintDisabledText(JLabel, Graphics, String, int, int)}. This
63   * class is designed to be easily extended by subclasses.
64   *
65   * @author Samuel Sjoberg, http://samuelsjoberg.com
66   * @version 1.3.0
67   */
68  public class MultiLineLabelUI extends BasicLabelUI implements ComponentListener {

146     /**
147      * Calculate the paint rectangles for the icon and text for the passed
148      * label.
149      *
150      * @param l      a label
151      * @param fm     the font metrics to use, or <code>null</code> to get the font
152      *               metrics from the label
153      * @param width  label width
154      * @param height label height
155      */
156     protected void updateLayout(JLabel l, FontMetrics fm, int width, int height) {
157         if (fm == null) {
158             fm = l.getFontMetrics(l.getFont());
159         }
160         metrics = fm;
161 
162         String text = l.getText();
163         Icon icon = l.getIcon();
164         Insets insets = l.getInsets(paintViewInsets);
165 
166         paintViewR.x = insets.left;
167         paintViewR.y = insets.top;
168         paintViewR.width = width - (insets.left + insets.right);
169         paintViewR.height = height - (insets.top + insets.bottom);
170 
171         paintIconR.x = paintIconR.y = paintIconR.width = paintIconR.height = 0;
172         paintTextR.x = paintTextR.y = paintTextR.width = paintTextR.height = 0;
173 
174         layoutCL(l, fm, text, icon, paintViewR, paintIconR, paintTextR);
175     }

185     /** {@inheritDoc} */
186     public void paint(Graphics g, JComponent c) {
187 
188         // parent's update method fills the background
189         prepareGraphics(g);
190 
191         JLabel label = (JLabel) c;
192         String text = label.getText();
193         Icon icon = (label.isEnabled()) ? label.getIcon() : label
194                 .getDisabledIcon();
195 
196         if ((icon == null) && (text == null)) {
197             return;
198         }
199 
200         FontMetrics fm = g.getFontMetrics();
201 
202         updateLayout(label, fm, c.getWidth(), c.getHeight());
203 
204         if (icon != null) {
205             icon.paintIcon(c, g, paintIconR.x, paintIconR.y);
206         }
207 
208         if (text != null) {
209             View v = (View) c.getClientProperty("html");
210             if (v != null) {
211                 // HTML view disables multi-line painting.
212                 v.paint(g, paintTextR);
213             } else {
214                 // Paint the multi line text
215                 paintTextLines(g, label, fm);
216             }
217         }
218     }

220     /**
221      * Paint the wrapped text lines.
222      *
223      * @param g     graphics component to paint on
224      * @param label the label being painted
225      * @param fm    font metrics for current font
226      */
227     protected void paintTextLines(Graphics g, JLabel label, FontMetrics fm) {
228         List<String> lines = getTextLines(label);
229 
230         // Available component height to paint on.
231         int height = getAvailableHeight(label);
232 
233         int textHeight = lines.size() * fm.getHeight();
234         while (textHeight > height) {
235             // Remove one line until no. of visible lines is found.
236             textHeight -= fm.getHeight();
237         }
238         paintTextR.height = Math.min(textHeight, height);
239         paintTextR.y = alignmentY(label, fm, paintTextR);
240 
241         int textX = paintTextR.x;
242         int textY = paintTextR.y;
243 
244         for (Iterator<String> it = lines.iterator(); it.hasNext()
245                 && paintTextR.contains(textX, textY + getAscent(fm)); textY += fm
246                 .getHeight()) {
247 
248             String text = it.next().trim();
249 
250             if (it.hasNext()
251                     && !paintTextR.contains(textX, textY + fm.getHeight()
252                     + getAscent(fm))) {
253                 // The last visible row, add a clip indication.
254                 text = clip(text, fm, paintTextR);
255             }
256 
257             int x = alignmentX(label, fm, text, paintTextR);
258 
259             if (label.isEnabled()) {
260                 paintEnabledText(label, g, text, x, textY);
261             } else {
262                 paintDisabledText(label, g, text, x, textY);
263             }
264         }
265     }

267     /**
268      * Returns the available height to paint text on. This is the height of the
269      * passed component with insets subtracted.
270      *
271      * @param l a component
272      * @return the available height
273      */
274     protected int getAvailableHeight(JLabel l) {
275         l.getInsets(paintViewInsets);
276         return l.getHeight() - paintViewInsets.top - paintViewInsets.bottom;
277     }

279     /**
280      * Add a clip indication to the string. It is important that the string
281      * length does not exceed the length or the original string.
282      *
283      * @param text   the to be painted
284      * @param fm     font metrics
285      * @param bounds the text bounds
286      * @return the clipped string
287      */
288     protected String clip(String text, FontMetrics fm, Rectangle bounds) {
289         // Fast and lazy way to insert a clip indication is to simply replace
290         // the last characters in the string with the clip indication.
291         // A better way would be to use metrics and calculate how many (if any)
292         // characters that need to be replaced.
293         if (text.length() < 3) {
294             return "...";
295         }
296         return text.substring(0, text.length() - 3) + "...";
297     }

299     /**
300      * Establish the vertical text alignment. The default alignment is to center
301      * the text in the label.
302      *
303      * @param label  the label to paint
304      * @param fm     font metrics
305      * @param bounds the text bounds rectangle
306      * @return the vertical text alignment, defaults to CENTER.
307      */
308     protected int alignmentY(JLabel label, FontMetrics fm, Rectangle bounds) {
309         final int height = getAvailableHeight(label);
310         int textHeight = bounds.height;
311 
312         if (label instanceof MultiLineLabel) {
313             int align = ((MultiLineLabel) label).getVerticalTextAlignment();
314             switch (align) {
315                 case JLabel.TOP:
316                     return getAscent(fm) + paintViewInsets.top;
317                 case JLabel.BOTTOM:
318                     return getAscent(fm) + height - paintViewInsets.top
319                             + paintViewInsets.bottom - textHeight;
320                 default:
321             }
322         }
323 
324         // Center alignment
325         int textY = paintViewInsets.top + (height - textHeight) / 2
326                 + getAscent(fm);
327         return Math.max(textY, getAscent(fm) + paintViewInsets.top);
328     }

330     /**
331      * <p>getAscent.</p>
332      *
333      * @param fm a {@link java.awt.FontMetrics} object.
334      * @return a int.
335      */
336     private static int getAscent(FontMetrics fm) {
337         return fm.getAscent() + fm.getLeading();
338     }

340     /**
341      * Establish the horizontal text alignment. The default alignment is left
342      * aligned text.
343      *
344      * @param label  the label to paint
345      * @param fm     font metrics
346      * @param s      the string to paint
347      * @param bounds the text bounds rectangle
348      * @return the x-coordinate to use when painting for proper alignment
349      */
350     protected int alignmentX(JLabel label, FontMetrics fm, String s,
351                              Rectangle bounds) {
352         if (label instanceof MultiLineLabel) {
353             int align = ((MultiLineLabel) label).getHorizontalTextAlignment();
354             switch (align) {
355                 case JLabel.RIGHT:
356                     return bounds.x + paintViewR.width - fm.stringWidth(s);
357                 case JLabel.CENTER:
358                     return bounds.x + paintViewR.width / 2 - fm.stringWidth(s) / 2;
359                 default:
360                     return bounds.x;
361             }
362         }
363         return bounds.x;
364     }

366     /**
367      * Check the given string to see if it should be rendered as HTML. Code
368      * based on implementation found in
369      * <code>BasicHTML.isHTMLString(String)</code> in future JDKs.
370      *
371      * @param s the string
372      * @return <code>true</code> if string is HTML, otherwise <code>false</code>
373      */
374     private static boolean isHTMLString(String s) {
375         if (s != null) {
376             if ((s.length() >= 6) && (s.charAt(0) == '<')
377                     && (s.charAt(5) == '>')) {
378                 String tag = s.substring(1, 5);
379                 return tag.equalsIgnoreCase("html");
380             }
381         }
382         return false;
383     }

385     /** {@inheritDoc} */
386     public Dimension getPreferredSize(JComponent c) {
387         Dimension d = super.getPreferredSize(c);
388         JLabel label = (JLabel) c;
389 
390         if (isHTMLString(label.getText())) {
391             return d; // HTML overrides everything and we don't need to process
392         }
393 
394         // Width calculated by super is OK. The preferred width is the width of
395         // the unwrapped content as long as it does not exceed the width of the
396         // parent container.
397 
398         if (c.getParent() != null) {
399             // Ensure that preferred width never exceeds the available width
400             // (including its border insets) of the parent container.
401             Insets insets = c.getParent().getInsets();
402             Dimension size = c.getParent().getSize();
403             if (size.width > 0) {
404                 // If width isn't set component shouldn't adjust.
405                 d.width = size.width - insets.left - insets.right;
406             }
407         }
408 
409         updateLayout(label, null, d.width, d.height);
410 
411         // The preferred height is either the preferred height of the text
412         // lines, or the height of the icon.
413         d.height = Math.max(d.height, getPreferredHeight(label));
414 
415         return d;
416     }

418     /**
419      * The preferred height of the label is the height of the lines with added
420      * top and bottom insets.
421      *
422      * @param label the label
423      * @return the preferred height of the wrapped lines.
424      */
425     protected int getPreferredHeight(JLabel label) {
426         int numOfLines = getTextLines(label).size();
427         Insets insets = label.getInsets(paintViewInsets);
428         return numOfLines * metrics.getHeight() + insets.top + insets.bottom;
429     }

468     /**
469      * Prepare the text lines for rendering. The lines are wrapped to fit in the
470      * current available space for text. Explicit line breaks are preserved.
471      *
472      * @param l the label to render
473      * @return a list of text lines to render
474      */
475     protected List<String> prepareLines(JLabel l) {
476         List<String> lines = new ArrayList<String>(defaultSize);
477         String text = l.getText();
478         if (text == null) {
479             return null; // Null guard
480         }
481         PlainDocument doc = new PlainDocument();
482         try {
483             doc.insertString(0, text, null);
484         } catch (BadLocationException e) {
485             return null;
486         }
487         Element root = doc.getDefaultRootElement();
488         for (int i = 0, j = root.getElementCount(); i < j; i++) {
489             wrap(lines, root.getElement(i));
490         }
491         return lines;
492     }

494     /**
495      * If necessary, wrap the text into multiple lines.
496      *
497      * @param lines line array in which to store the wrapped lines
498      * @param elem  the document element containing the text content
499      */
500     protected void wrap(List<String> lines, Element elem) {
501         int p1 = elem.getEndOffset();
502         Document doc = elem.getDocument();
503         for (int p0 = elem.getStartOffset(); p0 < p1; ) {
504             int p = calculateBreakPosition(doc, p0, p1);
505             try {
506                 lines.add(doc.getText(p0, p - p0));
507             } catch (BadLocationException e) {
508                 throw new Error("Can't get line text. p0=" + p0 + " p=" + p);
509             }
510             p0 = (p == p0) ? p1 : p;
511         }
512     }

514     /**
515      * Calculate the position on which to break (wrap) the line.
516      *
517      * @param doc the document
518      * @param p0  start position
519      * @param p1  end position
520      * @return the actual end position, will be <code>p1</code> if content does
521      *         not need to wrap, otherwise it will be less than <code>p1</code>.
522      */
523     protected int calculateBreakPosition(Document doc, int p0, int p1) {
524         Segment segment = SegmentCache.getSegment();
525         try {
526             doc.getText(p0, p1 - p0, segment);
527         } catch (BadLocationException e) {
528             throw new Error("Can't get line text");
529         }
530 
531         int width = paintTextR.width;
532         int p = p0
533                 + Utilities.getBreakLocation(segment, metrics, 0, width, null,
534                 p0);
535         SegmentCache.releaseSegment(segment);
536         return p;
537     }

539     /**
540      * Static singleton {@link Segment} cache.
541      *
542      * @author Samuel Sjoberg
543      * @see javax.swing.text.SegmentCache
544      */
545     protected static final class SegmentCache {
546 
547         /**
548          * Reused segments.
549          */
550         private ArrayList<Segment> segments = new ArrayList<Segment>(2);
551 
552         /**
553          * Singleton instance.
554          */
555         private static SegmentCache cache = new SegmentCache();
556 
557         /**
558          * Private constructor.
559          */
560         private SegmentCache() {
561         }
562 
563         /**
564          * Returns a <code>Segment</code>. When done, the <code>Segment</code>
565          * should be recycled by invoking {@link #releaseSegment(Segment)}.
566          *
567          * @return a <code>Segment</code>.
568          */
569         public static Segment getSegment() {
570             int size = cache.segments.size();
571             if (size > 0) {
572                 return cache.segments.remove(size - 1);
573             }
574             return new Segment();
575         }
576 
577         /**
578          * Releases a <code>Segment</code>. A segment should not be used after
579          * it is released, and a segment should never be released more than
580          * once.
581          */
582         public static void releaseSegment(Segment segment) {
583             segment.array = null;
584             segment.count = 0;
585             cache.segments.add(segment);
586         }
587     }
