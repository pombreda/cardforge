90      private void addMenu() {
91          JMenuItem humanGraveyard = new JMenuItem("View Graveyard");
92          humanGraveyard.addActionListener(new ActionListener() {
93              public void actionPerformed(ActionEvent actionEvent) {
94                  Card c[] = AllZoneUtil.getPlayerGraveyard(AllZone.getHumanPlayer()).toArray();
95  
96                  if (AllZone.getNameChanger().shouldChangeCardName()) c = AllZone.getNameChanger().changeCard(c);
97  
98                  if (c.length == 0) GuiUtils.getChoiceOptional("Player's Grave", new String[]{"no cards"});
99                  else GuiUtils.getChoiceOptional("Player's Grave", c);
100             }
101         });
102 
103         JMenuItem computerGraveyard = new JMenuItem("Computer - View Graveyard");
104         computerGraveyard.addActionListener(new ActionListener() {
105             public void actionPerformed(ActionEvent actionEvent) {
106                 Card c[] = AllZoneUtil.getPlayerGraveyard(AllZone.getComputerPlayer()).toArray();
107 
108                 if (AllZone.getNameChanger().shouldChangeCardName()) c = AllZone.getNameChanger().changeCard(c);
109 
110                 if (c.length == 0) GuiUtils.getChoiceOptional("Computer's Grave", new String[]{"no cards"});
111                 else GuiUtils.getChoiceOptional("Computer's Grave", c);
112             }
113         });
114 
115 
116         JMenuItem concedeGame = new JMenuItem("Concede Game");
117         concedeGame.addActionListener(new ActionListener() {
118             public void actionPerformed(ActionEvent actionEvent) {
119                 dispose();
120                 Constant.Runtime.matchState.addLose();
121                 new Gui_WinLose();
122             }
123         });
124 
125 
126         JMenuItem gameMenu = new JMenu("Menu");
127         gameMenu.add(humanGraveyard);
128         gameMenu.add(computerGraveyard);
129         // gameMenu.add(this.eotCheckboxForMenu);  // The static field GuiDisplay2.eotCheckboxForMenu should be accessed in a static way
130         gameMenu.add(GuiDisplay2.eotCheckboxForMenu);
131         gameMenu.add(new JSeparator());
132         gameMenu.add(concedeGame);
133 
134         JMenuBar menuBar = new JMenuBar();
135         menuBar.add(gameMenu);
136         menuBar.add(new MenuItem_HowToPlay());
137         this.setJMenuBar(menuBar);
138     }//addMenu()

377         //stack
378         AllZone.getStack().addObserver(new Observer() {
379             public void update(Observable a, Object b) {
380                 stackPanel.removeAll();
381                 MagicStack stack = AllZone.getStack();
382                 int count = 1;
383                 JLabel label;
384 
385                 for (int i = stack.size() - 1; 0 <= i; i--) {
386                     String text = stack.peekInstance(i).getStackDescription();
387 
388                     //change card name
389                     if (AllZone.getNameChanger().shouldChangeCardName()) {
390                         Card c = stack.peekInstance(i).getSourceCard();
391                         text = AllZone.getNameChanger().changeString(c, text);
392                     }
393 
394                     label = new JLabel("" + (count++) + ". " + text);
395 
396                     //update card detail
397                     final CardPanel cardPanel = new CardPanel(stack.peekInstance(i).getSourceCard());
398                     cardPanel.setLayout(new BorderLayout());
399                     cardPanel.add(label);
400                     cardPanel.addMouseMotionListener(new MouseMotionAdapter() {
401                         @Override
402                         public void mouseMoved(MouseEvent me) {
403                             GuiDisplay2.this.setCard(cardPanel.getCard());
404                         }//mouseMoved
405                     });
406 
407                     stackPanel.add(cardPanel);
408                 }
409 
410                 stackPanel.revalidate();
411                 stackPanel.repaint();
412             }
413         });
414         AllZone.getStack().updateObservers();
415         //END, stack
416 
417 
418         //self hand
419         AllZone.getHumanHand().addObserver(new Observer() {
420             public void update(Observable a, Object b) {
421                 PlayerZone pZone = (PlayerZone) a;
422                 JPanel p = playerHandPanel;
423                 p.removeAll();
424 
425                 Card c[] = AllZoneUtil.getCardsInZone(pZone).toArray();
426 
427                 //change card names
428                 if (AllZone.getNameChanger().shouldChangeCardName()) {
429                     AllZone.getHumanHand().setUpdate(false);
430                     c = AllZone.getNameChanger().changeCard(c);
431                     AllZone.getHumanHand().setUpdate(true);
432                 }
433 
434                 JPanel panel;
435                 for (int i = 0; i < c.length; i++) {
436                     panel = new CardPicturePanel(c[i]);
437                     p.add(panel);
438                 }
439                 p.revalidate();
440                 p.repaint();
441             }
442         });
443         AllZone.getHumanHand().updateObservers();
444         //END, self hand
445 
446         //self play (land)
447         AllZone.getHumanBattlefield().addObserver(new Observer() {
448             public void update(Observable a, Object b) {

460                 }

469         //END - self play (only land)
