163             public void run() {
164                 try {
165                     if (preferences.laf.equals(""))
166                         UIManager.setLookAndFeel(UIManager.getSystemLookAndFeelClassName());

331     /**
332      * <p>setupSealed.</p>
333      */
334     private void setupSealed() {
335         Deck deck = new Deck(Constant.GameType.Sealed);
336 
337         //ReadBoosterPack booster = new ReadBoosterPack();
338         //CardList pack = booster.getBoosterPack5();
339 
340         ArrayList<String> sealedTypes = new ArrayList<String>();
341         sealedTypes.add("Full Cardpool");
342         sealedTypes.add("Block / Set");
343         sealedTypes.add("Custom");
344 
345         String prompt = "Choose Sealed Deck Format:";
346         Object o = GuiUtils.getChoice(prompt, sealedTypes.toArray());
347 
348         SealedDeck sd = null;
349 
350         if (o.toString().equals(sealedTypes.get(0)))
351             sd = new SealedDeck("Full");
352 
353         else if (o.toString().equals(sealedTypes.get(1)))
354             sd = new SealedDeck("Block");
355 
356         else if (o.toString().equals(sealedTypes.get(2)))
357             sd = new SealedDeck("Custom");
358 
359         CardList sDeck = sd.getCardpool();
360 
361         if (sDeck.size() > 1) {
362 
363             for (int i = 0; i < sDeck.size(); i++)
364                 deck.addSideboard(sDeck.get(i).getName() + "|" + sDeck.get(i).getCurSetCode());
365 
366             for (int i = 0; i < Constant.Color.BasicLands.length; i++) {
367                 for (int j = 0; j < 18; j++)
368                     deck.addSideboard(Constant.Color.BasicLands[i] + "|" + sd.LandSetCode[0]);
369             }
370 
371             String sDeckName = JOptionPane.showInputDialog(null, ForgeProps.getLocalized(NEW_GAME_TEXT.SAVE_SEALED_MSG),
372                     ForgeProps.getLocalized(NEW_GAME_TEXT.SAVE_SEALED_TTL), JOptionPane.QUESTION_MESSAGE);
373             deck.setName(sDeckName);
374             deck.addMetaData("PlayerType", "Human");
375 
376             Constant.Runtime.HumanDeck[0] = deck;
377             Constant.Runtime.GameType[0] = Constant.GameType.Sealed;
378 
379             Deck aiDeck = sd.buildAIDeck(sd.getCardpool());
380             aiDeck.setName("AI_" + sDeckName);
381             aiDeck.addMetaData("PlayerType", "AI");
382             deckManager.addDeck(aiDeck);
383             deckManager.close();
384 
385             deckEditorButton_actionPerformed(null);
386             editor.customMenu.setCurrentGameType(Constant.GameType.Sealed);
387             editor.customMenu.showSealedDeck(deck);
388 
389 
390             Constant.Runtime.ComputerDeck[0] = aiDeck;
391         } else
392             new Gui_NewGame();
393     }

800      * <p>genDecks.</p>
801      *
802      * @param p a {@link java.lang.String} object.
803      */
804     private void genDecks(String p) {
805         Deck d = null;
806 
807         ArrayList<String> decks = new ArrayList<String>();
808         decks.add("2-Color Deck (original)");
809         decks.add("3-Color Deck (original)");
810         decks.add("5-Color Deck (original)");
811         decks.add("Semi-Random Theme Deck");
812         decks.add("2-Color Deck (new)");
813         decks.add("3-Color Deck (new)");
814 
815         String prompt = "Generate ";
816         if (p.equals("H"))
817             prompt += "Human ";
818         else
819             prompt += "Computer ";
820         prompt += "Deck";
821 
822         Object o = GuiUtils.getChoice(prompt, decks.toArray());
823 
824         if (o.toString().equals(decks.get(0)))
825             d = generateConstructedDeck();
826         else if (o.toString().equals(decks.get(1)))
827             d = generateConstructed3ColorDeck();
828         else if (o.toString().equals(decks.get(2)))
829             d = generateConstructed5ColorDeck();
830         else if (o.toString().equals(decks.get(3)))
831             d = generateConstructedThemeDeck();
832         else if (o.toString().equals(decks.get(4)))
833             d = generate2ColorDeck(p);
834         else if (o.toString().equals(decks.get(5)))
835             d = generate3ColorDeck(p);
836 
837         if (p.equals("H"))
838             Constant.Runtime.HumanDeck[0] = d;
839         else if (p.equals("C"))
840             Constant.Runtime.ComputerDeck[0] = d;
841     }
