226      *
227      * @param e a {@link java.awt.event.ActionEvent} object.
228      */
229     void newQuestButton_actionPerformed(ActionEvent e) {
230         int difficulty = 0;
231 
232         String mode = fantasyRadio.isSelected() ? forge.quest.data.QuestData.FANTASY : forge.quest.data.QuestData.REALISTIC;
233 
234         if (easyRadio.isSelected()) difficulty = 0;
235 
236         else if (mediumRadio.isSelected()) difficulty = 1;
237 
238         else if (hardRadio.isSelected()) difficulty = 2;
239 
240         else if (veryHardRadio.isSelected()) difficulty = 3;
241 
242         else //user didn't select a difficulty{
243             return;
244 
245         if (questData.hasSaveFile()) {
246             // this will overwrite your save file!
247             Object[] possibleValues = {"Yes", "No"};
248             Object choice = JOptionPane.showOptionDialog(null, "Starting a new quest will overwrite your current quest. Continue?",
249                     "Start New Quest?", JOptionPane.YES_NO_OPTION, JOptionPane.QUESTION_MESSAGE,
250                     null, possibleValues, possibleValues[1]);
251 
252             if (!choice.equals(0))
253                 return;
254         }
255 
256         //give the user a few cards to build a deck
257         questData.newGame(difficulty, mode, cbStandardStart.isSelected());
258 
259         questData.saveData();
260 
261 
262         //set global variable
263         AllZone.setQuestData(questData);
264 
265         dispose();
266         new QuestFrame();
267     }
