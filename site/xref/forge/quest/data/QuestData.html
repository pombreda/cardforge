307     /**
308      * <p>getDeck.</p>
309      *
310      * @param deckName a {@link java.lang.String} object.
311      * @return a {@link forge.deck.Deck} object.
312      */
313     public Deck getDeck(String deckName) {
314         //have to always set the card pool aka the Deck sideboard
315         //because new cards may have been added to the card pool by addCards()
316 
317         if (!myDecks.containsKey(deckName)) {
318             ErrorViewer.showError(new Exception(),
319                     "QuestData : getDeckFromMap(String deckName) error, deck name not found - %s", deckName);
320         }
321 
322         Deck d = myDecks.get(deckName);
323 
324         //below is probably not needed
325 
326         //remove old cards from card pool
327         for (int i = 0; i < d.countSideboard(); i++) {
328             d.removeSideboard(i);
329         }
330 
331         //add all cards to card pool
332         for (int i = 0; i < cardPool.size(); i++) {
333             d.addSideboard(cardPool.get(i));
334         }
335 
336         return d;
337     }

444             String str = cardPool.get(i);

448             }

481     /**
482      * <p>getCreditsToAdd.</p>
483      *
484      * @param matchState a {@link forge.quest.data.QuestMatchState} object.
485      * @return a long.
486      */
487     public long getCreditsToAdd(QuestMatchState matchState) {
488         long creds = (long) (QuestPreferences.getMatchRewardBase() + (QuestPreferences.getMatchRewardTotalWins() * win));
489         String[] wins = matchState.getWinMethods();
490         int[] winTurns = matchState.getWinTurns();
491         boolean[] mulliganedToZero = matchState.getMulliganedToZero();
492 
493         if (matchState.getLose() == 0) {
494             creds += QuestPreferences.getMatchRewardNoLosses();
495         }
496 
497         for (String s : wins) {
498             if (s != null) {
499                 if (s.equals("Poison Counters")) {
500                     creds += QuestPreferences.getMatchRewardPoisonWinBonus();
501                 } else if (s.equals("Milled")) {
502                     creds += QuestPreferences.getMatchRewardMilledWinBonus();
503                 } else if (s.equals("Battle of Wits") || s.equals("Felidar Sovereign") || s.equals("Helix Pinnacle") ||
504                         s.equals("Epic Struggle") || s.equals("Door to Nothingness") || s.equals("Barren Glory") ||
505                         s.equals("Near-Death Experience") || s.equals("Mortal Combat") || s.equals("Test of Endurance")) {
506                     creds += QuestPreferences.getMatchRewardAltWinBonus();
507                 }
508             }
509         }
510         for (int i : winTurns) {
511             if (i == 1) {
512                 creds += QuestPreferences.getMatchRewardWinFirst();
513             } else if (i <= 5) {
514                 creds += QuestPreferences.getMatchRewardWinByFifth();
515             } else if (i <= 10) {
516                 creds += QuestPreferences.getMatchRewardWinByTen();
517             } else if (i <= 15) {
518                 creds += QuestPreferences.getMatchRewardWinByFifteen();
519             }
520         }
521 
522 
523         for (boolean b : mulliganedToZero) {
524             if (b == true) {
525                 creds += QuestPreferences.getMatchMullToZero();
526             }
527         }
528 
529         if (inventory.getItemLevel("Estates") == 1) {
530             creds *= 1.1;
531         } else if (inventory.getItemLevel("Estates") == 2) {
532             creds *= 1.15;
533         } else if (inventory.getItemLevel("Estates") == 3) {
534             creds *= 1.2;
535         }
536 
537         this.addCredits(creds);
538 
539         return creds;
540     }

793      *
794      * @param didWin a boolean.
795      * @return a boolean.
796      */
797     public boolean shouldAddAdditionalCards(boolean didWin) {
798         float chance = 0.5f;
799         if (inventory.getItemLevel("Lucky Coin") == 1) {
800             chance = 0.65f;
801         }
802 
803         float r = MyRandom.random.nextFloat();
804 
805         if (didWin) {
806             return r <= chance;
807         } else {
808             return false;
809         }
