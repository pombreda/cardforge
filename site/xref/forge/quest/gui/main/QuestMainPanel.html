427     /**
428      * <p>refresh.</p>
429      */
430     void refresh() {
431         AllZone.getQuestData().saveData();
432 
433         devModeCheckBox.setSelected(Constant.Runtime.DevMode[0]);
434         smoothLandCheckBox.setSelected(Constant.Runtime.Smooth[0]);
435         //newGUICheckbox.setSelected(Gui_NewGame.preferences.newGui);
436 
437         creditsLabel.setText(" " + questData.getCredits());
438         statsLabel.setText(questData.getWin() + " wins / " + questData.getLost() + " losses");
439         titleLabel.setText(questData.getRank());
440 
441         //copy lastUsedDeck as removal triggers selection change. 
442         String lastUsedDeck = QuestMainPanel.lastUsedDeck;
443         deckComboBox.removeAllItems();
444 
445         if (questData.getDeckNames().size() > 0) {
446             deckComboBox.setEnabled(true);
447 
448             List<String> deckNames = new ArrayList<String>(questData.getDeckNames());
449 
450             Collections.sort(deckNames, new Comparator<String>() {
451                 public int compare(String s, String s1) {
452                     return s.compareToIgnoreCase(s1);
453                 }
454             });
455 
456             if (deckNames.contains(lastUsedDeck)) {
457                 deckNames.remove(lastUsedDeck);
458                 deckNames.add(0, lastUsedDeck);
459             }
460 
461             for (String deckName : deckNames) {
462                 deckComboBox.addItem(deckName);
463             }
464         } else {
465             deckComboBox.addItem(NO_DECKS_AVAILABLE);
466             deckComboBox.setEnabled(false);
467         }
468         deckComboBox.setMinimumSize(new Dimension(150, 0));
469 
470         questButton.setEnabled(nextQuestInWins() == 0);
471 
472         playButton.setEnabled(canGameBeLaunched());
473 
474         if (questData.getMode().equals(QuestData.FANTASY)) {
475             lifeLabel.setText(" " + questData.getLife());
476 
477             petComboBox.removeAllItems();
478 
479             Set<String> petList = questData.getPetManager().getAvailablePetNames();
480 
481             if (petList.size() > 0) {
482                 petComboBox.setEnabled(true);
483                 petCheckBox.setEnabled(true);
484                 for (String aPetList : petList) {
485                     petComboBox.addItem(aPetList);
486                 }
487             } else {
488                 petComboBox.addItem("No pets available");
489                 petComboBox.setEnabled(false);
490                 petCheckBox.setEnabled(false);
491             }
492 
493             if (!questData.getPetManager().shouldPetBeUsed()) {
494                 petCheckBox.setSelected(false);
495                 petComboBox.setEnabled(false);
496             } else {
497                 petCheckBox.setSelected(true);
498                 petComboBox.setSelectedItem(questData.getPetManager().getSelectedPet().getName());
499             }
500 
501 
502             this.plantBox.setEnabled(questData.getPetManager().getPlant().getLevel() > 0);
503             this.plantBox.setSelected(questData.getPetManager().shouldPlantBeUsed());
504 
505             QuestItemZeppelin zeppelin = (QuestItemZeppelin) questData.getInventory().getItem("Zeppelin");
506 
507             if (zeppelin.getLevel() > 0) {
508                 zeppelinPanel.removeAll();
509                 zeppelinPanel.add(zeppelinButton, BorderLayout.CENTER);
510             }
511 
512             if (!zeppelin.hasBeenUsed()) {
513                 zeppelinButton.setEnabled(true);
514             } else {
515                 zeppelinButton.setEnabled(false);
516             }
517 
518 
519         }
520 
521         if (nextQuestInWins() > 0) {
522             nextQuestLabel.setText("Next Quest in " + nextQuestInWins() + " Wins.");
523         } else {
524             nextQuestLabel.setText("Next Quest available now.");
525         }
526 
527         nextMatchLayout = new CardLayout();
528 
529         refreshNextMatchPanel();
530     }

548     /**
549      * <p>nextQuestInWins.</p>
550      *
551      * @return a int.
552      */
553     private int nextQuestInWins() {
554 
555         if (questData.getWin() < 25) {
556             return 25 - questData.getWin();
557         }
558 
559         int questsPlayed = questData.getQuestsPlayed();
560         int mul = 6;
561 
562         if (questData.getInventory().hasItem("Zeppelin")) {
563             mul = 4;
564         } else if (questData.getInventory().hasItem("Map")) {
565             mul = 5;
566         }
567 
568         int delta = (questsPlayed * mul) - questData.getWin();
569 
570         return (delta > 0) ? delta : 0;
571     }

654         //}

656         //Gui_NewGame.preferences.newGui = newGUICheckbox.isSelected();

688     /**
689      * <p>setupQuest.</p>
690      *
691      * @param humanDeck a {@link forge.deck.Deck} object.
692      */
693     private void setupQuest(Deck humanDeck) {
694         Quest_Assignment selectedQuest = ((QuestQuest) selectedOpponent).getQuestAssignment();
695 
696         Deck computerDeck = QuestBattleManager.getAIDeckNewFormat("quest" + selectedQuest.getId());
697         Constant.Runtime.ComputerDeck[0] = computerDeck;
698 
699         AllZone.setQuestAssignment(selectedQuest);
700 
701         int extraLife = 0;
702 
703         if (questData.getInventory().getItemLevel("Gear") == 2) {
704             extraLife = 3;
