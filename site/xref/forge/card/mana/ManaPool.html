177      * <p>oraclize.</p>
178      *
179      * @param manaCost a {@link java.lang.String} object.
180      * @return a {@link java.lang.String} object.
181      */
182     public static String oraclize(String manaCost) {
183         // converts RB to (R/B)
184         String[] parts = manaCost.split(" ");
185         StringBuilder res = new StringBuilder();
186         for (String s : parts) {
187             if (s.length() == 2 && colors.contains(s.charAt(1) + "")) s = s.charAt(0) + "/" + s.charAt(1);
188             if (s.length() == 3) s = "(" + s + ")";
189             if (s.equals("S")) s = "(S)";//for if/when we implement snow mana
190             if (s.equals("X")) s = "(X)";//X costs?
191             res.append(s);
192         }
193         return res.toString();
194     }

226      * <p>convertStringToMana.</p>
227      *
228      * @param manaStr a {@link java.lang.String} object.
229      * @param card a {@link forge.Card} object.
230      * @return a {@link java.util.ArrayList} object.
231      */
232     public static ArrayList<Mana> convertStringToMana(String manaStr, Card card) {
233         ArrayList<Mana> manaList = new ArrayList<Mana>();
234         manaStr = manaStr.trim();
235         String[] manaArr = manaStr.split(" ");
236 
237         String color = "";
238         int total = 0;
239         int genericTotal = 0;
240 
241         for (String c : manaArr) {
242             String longStr = Input_PayManaCostUtil.getLongColorString(c);
243             if (longStr.equals(Constant.Color.Colorless))
244                 genericTotal += Integer.parseInt(c);
245             else if (color.equals("")) {
246                 color = longStr;
247                 total = 1;
248             } else if (color.equals(longStr)) {
249                 total++;
250             } else {    // more than one color generated
251                 // add aggregate color
252                 manaList.add(new Mana(color, total, card));
253 
254                 color = longStr;
255                 total = 1;
256             }
257         }
258         if (total > 0)
259             manaList.add(new Mana(color, total, card));
260         if (genericTotal > 0)
261             manaList.add(new Mana(Constant.Color.Colorless, genericTotal, card));
262 
263         return manaList;
264     }

286     /**
287      * <p>getManaFrom.</p>
288      *
289      * @param pool a {@link java.util.ArrayList} object.
290      * @param manaStr a {@link java.lang.String} object.
291      * @return a {@link forge.card.mana.Mana} object.
292      */
293     public Mana getManaFrom(ArrayList<Mana> pool, String manaStr) {
294         String[] colors = manaStr.split("/");
295         boolean wantSnow = false;
296         for (int i = 0; i < colors.length; i++) {
297             colors[i] = Input_PayManaCostUtil.getLongColorString(colors[i]);
298             if (colors[i].equals(Constant.Color.Snow))
299                 wantSnow = true;
300         }
301 
302         Mana choice = null;
303         ArrayList<Mana> manaChoices = new ArrayList<Mana>();
304 
305         for (Mana mana : pool) {
306             if (mana.isColor(colors)) {
307                 if (choice == null)
308                     choice = mana;
309                 else if (choice.isSnow() && !mana.isSnow())
310                     choice = mana;
311             } else if (wantSnow && mana.isSnow()) {
312                 if (choice == null)
313                     choice = mana;
314                 else if (choice.isColor(Constant.Color.Colorless)) {
315                     // do nothing Snow Colorless should be used first to pay for Snow mana
316                 } else if (mana.isColor(Constant.Color.Colorless)) {
317                     // give preference to Colorless Snow mana over Colored snow mana
318                     choice = mana;
319                 } else if (floatingTotals[map.get(mana.getColor())] > floatingTotals[map.get(choice.getColor())]) {
320                     // give preference to Colored mana that there is more of to pay Snow costs
321                     choice = mana;
322                 }
323             } else if (colors[0].equals(Constant.Color.Colorless)) {    // colorless
324                 if (choice == null && mana.isColor(Constant.Color.Colorless))
325                     choice = mana;    // Colorless fits the bill nicely
326                 else if (choice == null) {
327                     manaChoices.add(mana);
328                 } else if (choice.isSnow() && !mana.isSnow()) {    // nonSnow colorless is better to spend than Snow colorless
329                     choice = mana;
330                 }
331             }
332         }
333 
334         if (choice != null)
335             return choice;
336 
337         if (colors[0].equals(Constant.Color.Colorless)) {
338             if (manaChoices.size() == 1)
339                 choice = manaChoices.get(0);
340             else if (manaChoices.size() > 1) {
341                 int[] normalMana = {0, 0, 0, 0, 0, 0};
342                 int[] snowMana = {0, 0, 0, 0, 0, 0};
343                 String[] manaStrings = {Constant.Color.White, Constant.Color.Blue, Constant.Color.Black, Constant.Color.Red, Constant.Color.Green, Constant.Color.Colorless};
344 
345                 // loop through manaChoices adding
346                 for (Mana m : manaChoices) {
347                     if (m.isSnow())
348                         snowMana[map.get(m.getColor())] += m.getAmount();
349                     else
350                         normalMana[map.get(m.getColor())] += m.getAmount();
351                 }
352 
353                 int totalMana = 0;
354                 ArrayList<String> alChoice = new ArrayList<String>();
355                 for (int i = 0; i < normalMana.length; i++) {
356                     totalMana += normalMana[i];
357                     totalMana += snowMana[i];
358                     if (normalMana[i] > 0) {
359                         alChoice.add(manaStrings[i] + "(" + normalMana[i] + ")");
360                     }
361                     if (snowMana[i] > 0) {
362                         alChoice.add("{S}" + manaStrings[i] + "(" + snowMana[i] + ")");
363                     }
364                 }
365 
366                 if (alChoice.size() == 1) {
367                     choice = manaChoices.get(0);
368                     return choice;
369                 }
370 
371                 int numColorless = Integer.parseInt(manaStr);
372                 if (numColorless >= totalMana) {
373                     choice = manaChoices.get(0);
374                     return choice;
375                 }
376 
377                 Object o;
378 
379                 if (owner.isHuman())
380                     o = GuiUtils.getChoiceOptional("Pay Mana from Mana Pool", alChoice.toArray());
381                 else o = alChoice.get(0); //owner is computer
382 
383                 if (o != null) {
384                     String ch = o.toString();
385                     boolean grabSnow = ch.startsWith("{S}");
386                     ch = ch.replace("{S}", "");
387 
388                     ch = ch.substring(0, ch.indexOf("("));
389 
390                     for (Mana m : manaChoices) {
391                         if (m.isColor(ch) && (!grabSnow || (grabSnow && m.isSnow()))) {
392                             if (choice == null)
393                                 choice = m;
394                             else if (choice.isSnow() && !m.isSnow())
395                                 choice = m;
396                         }
397                     }
398                 }
399             }
400         }
401 
402         return choice;
403     }

423         int i = 0;
424         Mana choice = null;
425         boolean flag = false;
426         while (i < pool.size()) {
427             Mana mana = pool.get(i);
428             if (flag) c = this;
429             if (c == this && mc.isNeeded(mana)) {
430                 c = mana.getSourceCard();
431                 flag = true;
432             }
433             if (mana.fromSourceCard(c)) {
434                 choice = mana;
435             }
436             i++;
437         }
438         removeManaFrom(pool, choice);
439     }

500     /**
501      * <p>formatMana.</p>
502      *
503      * @param Mana_2 a {@link java.lang.String} object.
504      * @param parsed a boolean.
505      * @return an array of {@link java.lang.String} objects.
506      */
507     public static String[] formatMana(String Mana_2, boolean parsed) {
508         String Mana = Mana_2;
509         //if (Mana.isEmpty()) return null;
510         if (Mana.trim().equals("")) return null;
511         if (!parsed)
512             Mana = oraclize(Mana);
513         try {
514             String[] Colorless = {Integer.toString(Integer.parseInt(Mana))};
515             return Colorless;
516         } catch (NumberFormatException ex) {
517         }
518 
519         ArrayList<String> res = new ArrayList<String>();
520         int Colorless = 0;
521         String clessString = "";
522         boolean parentheses = false;
523         String current = "";
524 
525         for (int i = 0; i < Mana.length(); i++) {
526             char c = Mana.charAt(i);
527             if (c == '(') {
528                 parentheses = true;
529                 continue;
530             }//Split cost handling ("(" +<W/U/B/R/G/2> + "/" + <W/U/B/R/G> + ")")
531             else if (parentheses) {
532                 if (c != ')') {
533                     current += c;
534                     continue;
535                 } else {
536                     parentheses = false;
537                     res.add(current);
538                     current = "";
539                     continue;
540                 }
541             }
542             String s = c + "";
543             if (colors.contains(s)) {
544                 res.add(s);
545                 if (clessString.trim().equals("")) continue;
546                 try {
547                     Colorless += Integer.parseInt(clessString.trim());
548                 } catch (NumberFormatException ex) {
549                     throw new RuntimeException(
550                             "Mana_Pool.getManaParts : Error, sum of noncolor mana parts is not a number - "
551                                     + clessString);
552                 }
553                 clessString = "";
554             } else clessString += s;
555         }
556         for (int i = 0; i < Colorless; i++)
557             res.add("1");
558 
559         return res.toArray(new String[0]);
560     }

564      *
565      * @param sa a {@link forge.card.spellability.SpellAbility} object.
566      * @param cost an array of {@link java.lang.String} objects.
567      * @param m a {@link forge.card.mana.ManaCost} object.
568      * @return a {@link forge.card.mana.ManaCost} object.
569      */
570     private ManaCost subtractMultiple(SpellAbility sa, String[] cost, ManaCost m) {
571         for (String s : cost) {
572             if (isEmpty())
573                 break;
574 
575             int num = 1;
576             try {
577                 num = Integer.parseInt(s);
578             } catch (NumberFormatException e) {
579                 // Not an integer, that's fine
580             }
581 
582             for (int i = 0; i < num; i++) {
583                 if (isEmpty())
584                     break;
585 
586                 m = subtractOne(sa, m, s);
587             }
588         }
589         return m;
590     }

598      * @return a {@link forge.card.mana.ManaCost} object.
599      */
600     public ManaCost subtractMana(SpellAbility sa, ManaCost m, Ability_Mana... mAbilities) {
601         ArrayList<Ability_Mana> paidAbs = sa.getPayingManaAbilities();
602 
603         if (mAbilities.length == 0) {
604             // paying from Mana Pool
605             if (m.isPaid() || isEmpty()) return m;
606 
607             String[] cost = formatMana(m.toString());
608             return subtractMultiple(sa, cost, m);
609         }
610 
611         // paying via Mana Abilities
612         for (Ability_Mana mability : mAbilities) {
613             paidAbs.add(mability);
614             String[] cost = formatMana(mability);
615             m = subtractMultiple(sa, cost, m);
616         }
617 
618         return m;
619     }

623      *
624      * @param manaStr a {@link java.lang.String} object.
625      */
626     public void subtractOne(String manaStr) {
627         // Just subtract from floating, used by removeExtrinsicKeyword
628         ManaCost manaCost = new ManaCost(manaStr);
629         if (manaStr.trim().equals("") || manaCost.isPaid()) return;
630 
631         // get a mana of this type from floating, bail if none available
632         Mana mana = getManaFrom(floatingMana, manaStr);
633         if (mana == null) return;    // no matching mana in the pool
634 
635         Mana[] manaArray = mana.toSingleArray();
636 
637         for (int i = 0; i < manaArray.length; i++) {
638             Mana m = manaArray[i];
639             if (manaCost.isNeeded(m)) {
640                 manaCost.payMana(m);
641                 findAndRemoveFrom(floatingMana, m);
642             } else
643                 break;
644         }
645     }

649      *
650      * @param sa a {@link forge.card.spellability.SpellAbility} object.
651      * @param manaCost a {@link forge.card.mana.ManaCost} object.
652      * @param manaStr a {@link java.lang.String} object.

655     public ManaCost subtractOne(SpellAbility sa, ManaCost manaCost, String manaStr) {
656         if (manaStr.trim().equals("") || manaCost.isPaid()) return manaCost;
657

711      * <p>accountFor.</p>
712      *
713      * @param sa a {@link forge.card.spellability.SpellAbility} object.
714      * @param mana an array of {@link java.lang.String} objects.
715      * @param c a {@link forge.Card} object.
716      * @return a boolean.
717      */
718     public boolean accountFor(SpellAbility sa, String[] mana, Card c) {
719         // TODO: account for unpaying mana in payMana and floatingPool
720         ArrayList<Mana> payMana = sa.getPayingMana();
721 
722         ArrayList<Mana> removePaying = new ArrayList<Mana>();
723         ArrayList<Mana> removeFloating = new ArrayList<Mana>();
724 
725         int i = 0, j = 0;
726         boolean usePay = payMana.size() > 0;
727         boolean flag = false;
728 
729         String manaStr = mana[i];
730         String color = Input_PayManaCostUtil.getLongColorString(manaStr);
731 
732         if (!usePay && floatingMana.size() == 0)
733             return false;
734 
735         while (i < mana.length) {
736 
737             Mana m = usePay ? payMana.get(j) : floatingMana.get(j);
738 
739             if (m.fromSourceCard(c) && m.getColor().equals(color)) {
740                 int amt = m.getColorlessAmount();
741                 if (amt > 0) {
742                     int difference = Integer.parseInt(manaStr) - amt;
743                     if (difference > 0)
744                         manaStr = Integer.toString(difference);
745                     else {
746                         i += amt;
747                         if (i < mana.length)
748                             manaStr = mana[i];
749                     }
750                 } else {
751                     i += m.getAmount();
752                     if (i < mana.length)
753                         manaStr = mana[i];
754                 }
755                 color = Input_PayManaCostUtil.getLongColorString(manaStr);
756                 if (usePay)
757                     removePaying.add(m);
758                 else
759                     removeFloating.add(m);
760 
761                 if (i == mana.length)    // If mana has been depleted, break from loop. All Accounted for!
762                     break;
763             }
764 
765             j++;    // increase j until we reach the end of paying, then reset and use floating.
766             if (usePay) {
767                 if (payMana.size() == j) {
768                     j = 0;
769                     usePay = false;
770                 }
771             }
772             if (!usePay && floatingMana.size() == j && !flag)
773                 return false;
774         }
775 
776 
777         for (int k = 0; k < removePaying.size(); k++) {
778             removeManaFrom(payMana, removePaying.get(k));
779         }
780         for (int k = 0; k < removeFloating.size(); k++) {
781             removeManaFrom(floatingMana, removeFloating.get(k));
782         }
783         return true;
784     }
