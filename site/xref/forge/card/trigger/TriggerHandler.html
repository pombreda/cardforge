75       * @return a {@link forge.card.trigger.Trigger} object.

78          Trigger ret = null;

81          if (mode.equals("AbilityCast")) {

94              ret = new Trigger_Blocks(mapParams, host);
95          } else if (mode.equals("Championed")) {
96              ret = new Trigger_Championed(mapParams, host);
97          } else if (mode.equals("ChangesZone")) {
98              ret = new Trigger_ChangesZone(mapParams, host);
99          } else if (mode.equals("Clashed")) {
100             ret = new Trigger_Clashed(mapParams, host);
101         } else if (mode.equals("CounterAdded")) {
102             ret = new Trigger_CounterAdded(mapParams, host);
103         } else if (mode.equals("Cycled")) {
104             ret = new Trigger_Cycled(mapParams, host);
105         } else if (mode.equals("DamageDone")) {
106             ret = new Trigger_DamageDone(mapParams, host);
107         } else if (mode.equals("Discarded")) {
108             ret = new Trigger_Discarded(mapParams, host);
109         } else if (mode.equals("Drawn")) {
110             ret = new Trigger_Drawn(mapParams, host);
111         } else if (mode.equals("LandPlayed")) {
112             ret = new Trigger_LandPlayed(mapParams, host);
113         } else if (mode.equals("LifeGained")) {
114             ret = new Trigger_LifeGained(mapParams, host);
115         } else if (mode.equals("LifeLost")) {
116             ret = new Trigger_LifeLost(mapParams, host);

121         } else if (mode.equals("Shuffled")) {

142     /**
143      * <p>parseParams.</p>
144      *
145      * @param trigParse a {@link java.lang.String} object.
146      * @return a {@link java.util.HashMap} object.
147      */
148     private static HashMap<String, String> parseParams(String trigParse) {
149         HashMap<String, String> mapParams = new HashMap<String, String>();
150 
151         if (trigParse.length() == 0)
152             throw new RuntimeException("TriggerFactory : registerTrigger -- trigParse too short");
153 
154         String params[] = trigParse.split("\\|");
155 
156         for (int i = 0; i < params.length; i++) {
157             params[i] = params[i].trim();
158         }
159 
160         for (String param : params) {
161             String[] splitParam = param.split("\\$");
162             for (int i = 0; i < splitParam.length; i++) {
163                 splitParam[i] = splitParam[i].trim();
164             }
165 
166             if (splitParam.length != 2) {
167                 StringBuilder sb = new StringBuilder();
168                 sb.append("TriggerFactory Parsing Error in registerTrigger() : Split length of ");
169                 sb.append(param).append(" is not 2.");
170                 throw new RuntimeException(sb.toString());
171             }
172 
173             mapParams.put(splitParam[0], splitParam[1]);
174         }
175 
176         return mapParams;
177     }

258     /**
259      * <p>runTrigger.</p>
260      *
261      * @param mode a {@link java.lang.String} object.
262      * @param runParams a {@link java.util.Map} object.
263      */
264     public void runTrigger(String mode, Map<String, Object> runParams) {
265         if (suppressedModes.contains(mode) || !registeredModes.contains(mode)) {
266             return;
267         }
268         //AP
269         for (Trigger regtrig : registeredTriggers) {
270             if (regtrig.getHostCard().getController().equals(AllZone.getPhase().getPlayerTurn())) {
271                 runSingleTrigger(regtrig, mode, runParams);
272             }
273         }
274         for (int i = 0; i < delayedTriggers.size(); i++) {
275             Trigger deltrig = delayedTriggers.get(i);
276             if (deltrig.getHostCard().getController().equals(AllZone.getPhase().getPlayerTurn())) {
277                 if (runSingleTrigger(deltrig, mode, runParams)) {
278                     delayedTriggers.remove(i);
279                     i--;
280                 }
281             }
282         }
283 
284         //NAP
285         for (Trigger regtrig : registeredTriggers) {
286             if (regtrig.getHostCard().getController().equals(AllZone.getPhase().getPlayerTurn().getOpponent())) {
287                 runSingleTrigger(regtrig, mode, runParams);
288             }
289         }
290         for (int i = 0; i < delayedTriggers.size(); i++) {
291             Trigger deltrig = delayedTriggers.get(i);
292             if (deltrig.getHostCard().getController().equals(AllZone.getPhase().getPlayerTurn().getOpponent())) {
293                 if (runSingleTrigger(deltrig, mode, runParams)) {
294                     delayedTriggers.remove(i);
295                     i--;
296                 }
297             }
298         }
299     }

301     //Checks if the conditions are right for a single trigger to go off, and runs it if so.
302     //Return true if the trigger went off, false otherwise.
303     /**
304      * <p>runSingleTrigger.</p>
305      *
306      * @param regtrig a {@link forge.card.trigger.Trigger} object.
307      * @param mode a {@link java.lang.String} object.
308      * @param runParams a {@link java.util.HashMap} object.
309      * @return a boolean.
310      */
311     private boolean runSingleTrigger(final Trigger regtrig, final String mode, final Map<String, Object> runParams) {
312         if (!regtrig.zonesCheck()) {
313             return false;
314         }
315         if (!regtrig.phasesCheck()) {
316             return false;
317         }
318         if (!regtrig.requirementsCheck()) {
319             return false;
320         }
321         if (regtrig.getHostCard().isFaceDown()) {
322             return false;
323         }
324         if(regtrig instanceof Trigger_Always) {
325             if(AllZone.getStack().hasStateTrigger(regtrig.ID))
326             {
327                 return false; //State triggers that are already on the stack don't trigger again.
328             }
329         }
330 
331         HashMap<String, String> trigParams = regtrig.getMapParams();
332         final Player[] decider = new Player[1];
333         //final boolean isOptional = false;
334 
335         if (mode.equals(trigParams.get("Mode"))) {
336             if (!regtrig.performTest(runParams)) {
337                 return false;
338             }
339 
340             // Any trigger should cause the phase not to skip
341             AllZone.getPhase().setSkipPhase(false);
342 
343             regtrig.setRunParams(runParams);
344 
345             //All tests passed, execute ability.
346             if (regtrig instanceof Trigger_TapsForMana) {
347                 Ability_Mana abMana = (Ability_Mana) runParams.get("Ability_Mana");
348                 if (null != abMana) abMana.setUndoable(false);
349             }
350 
351             AbilityFactory AF = new AbilityFactory();
352 
353             final SpellAbility[] sa = new SpellAbility[1];
354             Card host = AllZoneUtil.getCardState(regtrig.getHostCard());
355 
356             if (host == null)
357                 host = regtrig.getHostCard();
358 
359             // This will fix the Oblivion Ring issue, but is this the right fix?
360             for (Object o : regtrig.getHostCard().getRemembered()) {
361                 if (!host.getRemembered().contains(o)) {
362                     host.addRemembered(o);
363                 }
364             }
365 
366             sa[0] = regtrig.getOverridingAbility();
367             if (sa[0] == null) {
368                 if (!trigParams.containsKey("Execute")) {
369                     sa[0] = new Ability(regtrig.getHostCard(), "0") {
370                         @Override
371                         public void resolve() {
372                         }
373                     };
374                 } else {
375                     sa[0] = AF.getAbility(host.getSVar(trigParams.get("Execute")), host);
376                 }
377             }
378             sa[0].setTrigger(true);
379             sa[0].setSourceTrigger(regtrig.ID);
380             regtrig.setTriggeringObjects(sa[0]);
381             if (regtrig.getStoredTriggeredObjects() != null)
382                 sa[0].setAllTriggeringObjects(regtrig.getStoredTriggeredObjects());
383 
384             sa[0].setActivatingPlayer(host.getController());
385             sa[0].setStackDescription(sa[0].toString());
386 
387             boolean mand = false;
388             if (trigParams.containsKey("OptionalDecider")) {
389                 mand = false;
390                 decider[0] = AbilityFactory.getDefinedPlayers(host, trigParams.get("OptionalDecider"), sa[0]).get(0);
391             } else {
392                 mand = true;
393                 
394                 SpellAbility ability = sa[0];
395                 while(ability != null){
396 	                Target tgt = ability.getTarget();
397 	                
398 	                if (tgt != null) {
399 	                	tgt.setMandatory(true);
400 	                }
401 	                ability = ability.getSubAbility();
402                 }
403             }
404             final boolean isMandatory = mand;
405 
406             //Wrapper ability that checks the requirements again just before resolving, for intervening if clauses.
407             //Yes, it must wrap ALL SpellAbility methods in order to handle possible corner cases.
408             //(The trigger can have a hardcoded OverridingAbility which can make use of any of the methods)
409             final Ability wrapperAbility = new Ability(regtrig.getHostCard(), "0") {
410             	
411                 @Override
412                 public boolean isWrapper() {
413                     return true;
414                 }
415             	
416                 @Override
417                 public void setPaidHash(HashMap<String, CardList> hash) {
418                     sa[0].setPaidHash(hash);
419                 }
