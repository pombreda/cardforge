175     public boolean canPlay() {
176         Card source = getSourceCard();
177         if (AllZone.getStack().isSplitSecondOnStack() || source.isUnCastable()) return false;
178 
179         Player turn = AllZone.getPhase().getPlayerTurn();
180 
181         if (source.getName().equals("Serra Avenger")) {
182             if (turn.equals(source.getController()) && turn.getTurn() <= 3)
183                 return false;
184         } else if (source.getName().equals("Blizzard")) {
185             CardList lands = AllZoneUtil.getPlayerLandsInPlay(source.getController());
186             lands = lands.getType("Snow");
187             if (lands.size() == 0) return false;
188         }
189 
190         // Flash handled by super.canPlay
191         return super.canPlay();

195     @Override
196     public boolean canPlayAI() {
197 
198         Card card = getSourceCard();
199 
200         //check on legendary
201         if (card.isType("Legendary")) {
202             CardList list = AllZoneUtil.getPlayerCardsInPlay(AllZone.getComputerPlayer());
203             if (list.containsName(card.getName()))
204                 return false;
205         }
206         if (card.isPlaneswalker()) {
207             CardList list = AllZoneUtil.getPlayerCardsInPlay(AllZone.getComputerPlayer());
208             list = list.getType("Planeswalker");
209 
210             for (int i = 0; i < list.size(); i++) {
211                 String subtype = card.getType().get(card.getType().size() - 1);
212                 CardList cl = list.getType(subtype);
213 
214                 if (cl.size() > 0) {
215                     return false;
216                 }
217             }
218         }
219         if (card.isType("World")) {
220             CardList list = AllZoneUtil.getPlayerCardsInPlay(AllZone.getComputerPlayer());
221             list = list.getType("World");
222             if (list.size() > 0) return false;
223         }
224 
225         if (card.isCreature()
226                 && card.getNetDefense() <= 0
227                 && !card.hasStartOfKeyword("etbCounter")
228                 && !card.getText().contains("Modular"))
229             return false;
230 
231         if (willChampion) {
232             Object o = championGetCreature.execute();
233             if (o == null) return false;
234 
235             CardList cl = (CardList) championGetCreature.execute();
236             if ((o == null) || !(cl.size() > 0) || !AllZone.getZone(getSourceCard()).is(Constant.Zone.Hand))
237                 return false;
238         }
239 
240         if (!checkETBEffects(card, this, null))
241             return false;
242 
243         return super.canPlayAI();
244     }//canPlayAI()

247      * <p>checkETBEffects.</p>
248      *
249      * @param card a {@link forge.Card} object.

251      * @param api a {@link java.lang.String} object.
252      * @return a boolean.
253      */
254     public static boolean checkETBEffects(Card card, SpellAbility sa, String api) {
255         // Trigger play improvements
256         ArrayList<Trigger> triggers = card.getTriggers();
257         for (Trigger tr : triggers) {
258             // These triggers all care for ETB effects
259 
260             HashMap<String, String> params = tr.getMapParams();
261             if (!params.get("Mode").equals("ChangesZone"))
262                 continue;
263 
264             if (!params.get("Destination").equals("Battlefield"))
265                 continue;
266 
267             if (params.containsKey("ValidCard") && !params.get("ValidCard").contains("Self"))
268                 continue;
269 
270             if (!tr.requirementsCheck())
271                 continue;
272 
273             if (tr.getOverridingAbility() != null)    // Don't look at Overriding Abilities yet
274                 continue;
275 
276             // Maybe better considerations
277             AbilityFactory af = new AbilityFactory();
278             SpellAbility exSA = af.getAbility(card.getSVar(params.get("Execute")), card);
279 
280             if (api != null && !af.getAPI().equals(api))
281                 continue;
282 
283             exSA.setActivatingPlayer(sa.getActivatingPlayer());
284 
285             // Run non-mandatory trigger.
286             // These checks only work if the Executing SpellAbility is an Ability_Sub.
287             if (exSA instanceof Ability_Sub && !exSA.doTrigger(false)) {
288                 // AI would not run this trigger if given the chance
289 
290                 // if trigger is mandatory, return false
291                 if (params.get("OptionalDecider") == null) {
292                     return false;
293                 }
294                 // else
295                 // otherwise, return false 50% of the time?
296             }
297         }
298 
299         return true;
300     }
