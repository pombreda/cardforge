23      /**
24       * <p>shouldCycle.</p>
25       *
26       * @param c a {@link forge.Card} object.
27       * @return a int.
28       */
29      public static int shouldCycle(Card c) {
30          ArrayList<String> a = c.getKeyword();
31          for (int i = 0; i < a.size(); i++)
32              if (a.get(i).toString().startsWith("Cycling")) return i;
33  
34          return -1;
35      }

38       * <p>shouldEnchant.</p>
39       *
40       * @param c a {@link forge.Card} object.
41       * @return a int.
42       */
43      public static int shouldEnchant(Card c) {
44          ArrayList<String> a = c.getKeyword();
45          for (int i = 0; i < a.size(); i++)
46              if (a.get(i).toString().startsWith("enPump")) return i;
47  
48          return -1;
