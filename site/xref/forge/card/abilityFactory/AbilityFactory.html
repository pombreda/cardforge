157     /**
158      * <p>Getter for the field <code>mapParams</code>.</p>
159      *
160      * @param abString a {@link java.lang.String} object.
161      * @param hostCard a {@link forge.Card} object.
162      * @return a {@link java.util.HashMap} object.
163      */
164     public HashMap<String, String> getMapParams(String abString, Card hostCard) {
165         HashMap<String, String> mapParameters = new HashMap<String, String>();
166 
167         if (!(abString.length() > 0))
168             throw new RuntimeException("AbilityFactory : getAbility -- abString too short in " + hostCard.getName() + ": [" + abString + "]");
169 
170         String a[] = abString.split("\\|");
171 
172         for (int aCnt = 0; aCnt < a.length; aCnt++)
173             a[aCnt] = a[aCnt].trim();
174 
175         if (!(a.length > 0))
176             throw new RuntimeException("AbilityFactory : getAbility -- a[] too short in " + hostCard.getName());
177 
178         for (int i = 0; i < a.length; i++) {
179             String aa[] = a[i].split("\\$");
180 
181             for (int aaCnt = 0; aaCnt < aa.length; aaCnt++)
182                 aa[aaCnt] = aa[aaCnt].trim();
183 
184             if (aa.length != 2) {
185                 StringBuilder sb = new StringBuilder();
186                 sb.append("AbilityFactory Parsing Error in getAbility() : Split length of ");
187                 sb.append(a[i]).append(" in ").append(hostCard.getName()).append(" is not 2.");
188                 throw new RuntimeException(sb.toString());
189             }
190 
191             mapParameters.put(aa[0], aa[1]);
192         }
193 
194         return mapParameters;
195     }

1008     /**
1009      * <p>calculateAmount.</p>
1010      *
1011      * @param card a {@link forge.Card} object.
1012      * @param amount a {@link java.lang.String} object.
1013      * @param ability a {@link forge.card.spellability.SpellAbility} object.
1014      * @return a int.
1015      */
1016     public static int calculateAmount(Card card, String amount, SpellAbility ability) {
1017         // amount can be anything, not just 'X' as long as sVar exists
1018 
1019         // If Amount is -X, strip the minus sign before looking for an SVar of that kind
1020         int multiplier = 1;
1021         if (amount.startsWith("-")) {
1022             multiplier = -1;
1023             amount = amount.substring(1);
1024         }
1025 
1026         if (!card.getSVar(amount).equals("")) {
1027             String calcX[] = card.getSVar(amount).split("\\$");
1028             if (calcX.length == 1 || calcX[1].equals("none"))
1029                 return 0;
1030 
1031             if (calcX[0].startsWith("Count"))
1032                 return CardFactoryUtil.xCount(card, calcX[1]) * multiplier;
1033 
1034             else if (ability != null) {
1035                 //Player attribute counting
1036                 if (calcX[0].startsWith("TargetedPlayer")) {
1037                     ArrayList<Player> players = new ArrayList<Player>();
1038                     SpellAbility saTargeting = (ability.getTarget() == null) ? findParentsTargetedPlayer(ability) : ability;
1039                     if (saTargeting.getTarget() != null) {
1040                         players.addAll(saTargeting.getTarget().getTargetPlayers());
1041                     } else {
1042                         players.addAll(getDefinedPlayers(card, saTargeting.getAbilityFactory().getMapParams().get("Defined"), saTargeting));
1043                     }
1044                     return CardFactoryUtil.playerXCount(players, calcX[1], card) * multiplier;
1045                 }
1046                 if (calcX[0].startsWith("TargetedController")) {
1047                     ArrayList<Player> players = new ArrayList<Player>();
1048                     ArrayList<Card> list = getDefinedCards(card, "Targeted", ability);
1049                     ArrayList<SpellAbility> sas = getDefinedSpellAbilities(card, "Targeted", ability);
1050 
1051                     for (Card c : list) {
1052                         Player p = c.getController();
1053                         if (!players.contains(p))
1054                             players.add(p);
1055                     }
1056                     for (SpellAbility s : sas) {
1057                         Player p = s.getSourceCard().getController();
1058                         if (!players.contains(p)) players.add(p);
1059                     }
1060                     return CardFactoryUtil.playerXCount(players, calcX[1], card) * multiplier;
1061                 }
1062 
1063                 CardList list = new CardList();
1064                 if (calcX[0].startsWith("Sacrificed"))
1065                     list = findRootAbility(ability).getPaidList("Sacrificed");
1066 
1067                 else if (calcX[0].startsWith("Discarded"))
1068                     list = findRootAbility(ability).getPaidList("Discarded");
1069 
1070                 else if (calcX[0].startsWith("Exiled")) {
1071                     list = findRootAbility(ability).getPaidList("Exiled");
1072                 } else if (calcX[0].startsWith("Tapped")) {
1073                     list = findRootAbility(ability).getPaidList("Tapped");
1074                 } else if (calcX[0].startsWith("Targeted")) {
1075                     Target t = ability.getTarget();
1076                     if (null != t) {
1077                         ArrayList<Object> all = t.getTargets();
1078                         if (!all.isEmpty() && all.get(0) instanceof SpellAbility) {
1079                             SpellAbility saTargeting = findParentsTargetedSpellAbility(ability);
1080                             list = new CardList();
1081                             ArrayList<SpellAbility> sas = saTargeting.getTarget().getTargetSAs();
1082                             for (SpellAbility sa : sas) {
1083                                 list.add(sa.getSourceCard());
1084                             }
1085                         } else {
1086                             SpellAbility saTargeting = findParentsTargetedCard(ability);
1087                             list = new CardList(saTargeting.getTarget().getTargetCards().toArray());
1088                         }
1089                     } else {
1090                         SpellAbility parent = findParentsTargetedCard(ability);
1091 
1092                         ArrayList<Object> all = parent.getTarget().getTargets();
1093                         if (!all.isEmpty() && all.get(0) instanceof SpellAbility) {
1094                             list = new CardList();
1095                             ArrayList<SpellAbility> sas = parent.getTarget().getTargetSAs();
1096                             for (SpellAbility sa : sas) {
1097                                 list.add(sa.getSourceCard());
1098                             }
1099                         } else {
1100                             SpellAbility saTargeting = findParentsTargetedCard(ability);
1101                             list = new CardList(saTargeting.getTarget().getTargetCards().toArray());
1102                         }
1103                     }
1104                 } else if (calcX[0].startsWith("Triggered")) {
1105                     SpellAbility root = ability.getRootSpellAbility();
1106                     list = new CardList();
1107                     list.add((Card) root.getTriggeringObject(calcX[0].substring(9)));
1108                 } else if (calcX[0].startsWith("Remembered")) {
1109                     // Add whole Remembered list to handlePaid
1110                     list = new CardList();
1111                     for (Object o : card.getRemembered()) {
1112                         if (o instanceof Card)
1113                             list.add(AllZoneUtil.getCardState((Card) o));
1114                     }
1115                 } else if (calcX[0].startsWith("Imprinted")) {
1116                     // Add whole Imprinted list to handlePaid
1117                     list = new CardList();
1118                     for (Card c : card.getImprinted())
1119                         list.add(AllZoneUtil.getCardState(c));
1120                 } else if (calcX[0].startsWith("TriggerCount")) {
1121                     // TriggerCount is similar to a regular Count, but just pulls Integer Values from Trigger objects
1122                     String[] l = calcX[1].split("/");
1123                     String[] m = CardFactoryUtil.parseMath(l);
1124                     int count = (Integer) ability.getTriggeringObject(l[0]);
1125 
1126                     return CardFactoryUtil.doXMath(count, m, card) * multiplier;
1127                 } else
1128                     return 0;
1129 
1130                 return CardFactoryUtil.handlePaid(list, calcX[1], card) * multiplier;
1131             } else
1132                 return 0;
1133         }
1134 
1135         return Integer.parseInt(amount) * multiplier;
1136     }
