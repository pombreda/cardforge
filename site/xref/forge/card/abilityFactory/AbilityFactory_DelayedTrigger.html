178     private static String delTrigStackDescription(AbilityFactory af, SpellAbility sa) {
179         HashMap<String, String> mapParams = af.getMapParams();
180         
181         StringBuilder sb = new StringBuilder();
182         
183         if (sa instanceof Ability_Sub)
184         	sb.append(" ");
185         else
186         	sb.append(sa.getSourceCard()).append(" - ");
187             
188         
189         if(mapParams.containsKey("SpellDescription")) {
190         	sb.append(mapParams.get("SpellDescription"));
191         }
192         else if(mapParams.containsKey("TriggerDescription")) {
193             sb.append(mapParams.get("TriggerDescription"));
194         }
195         
196         Ability_Sub abSub = sa.getSubAbility();
197         if (abSub != null) {
198             sb.append(abSub.getStackDescription());
199         }
200 
201         return sb.toString();
202         
203     }

206      * <p>doResolve.</p>
207      *
208      * @param AF a {@link forge.card.abilityFactory.AbilityFactory} object.
209      * @param SA a {@link forge.card.spellability.SpellAbility} object.
210      */
211     private static void doResolve(AbilityFactory AF, SpellAbility SA) {
212         HashMap<String, String> mapParams = AF.getMapParams();
213 
214         if (mapParams.containsKey("Cost"))
215             mapParams.remove("Cost");
216 
217         if (mapParams.containsKey("SpellDescription")) {
218             mapParams.put("TriggerDescription", mapParams.get("SpellDescription"));
219             mapParams.remove("SpellDescription");
220         }
221 
222         Trigger delTrig = TriggerHandler.parseTrigger(mapParams, AF.getHostCard());
223 
224         AllZone.getTriggerHandler().registerDelayedTrigger(delTrig);
225     }
