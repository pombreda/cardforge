117     /**
118      * <p>sacrificeDescription.</p>
119      *
120      * @param af a {@link forge.card.abilityFactory.AbilityFactory} object.
121      * @param sa a {@link forge.card.spellability.SpellAbility} object.
122      * @return a {@link java.lang.String} object.
123      */
124     public static String sacrificeDescription(final AbilityFactory af, SpellAbility sa) {
125         HashMap<String, String> params = af.getMapParams();
126         StringBuilder sb = new StringBuilder();
127 
128         if (sa instanceof Ability_Sub)
129             sb.append(" ");
130         else
131             sb.append(sa.getSourceCard().getName()).append(" - ");
132 
133         String conditionDesc = params.get("ConditionDescription");
134         if (conditionDesc != null)
135             sb.append(conditionDesc).append(" ");
136 
137         Target tgt = af.getAbTgt();
138         ArrayList<Player> tgts;
139         if (tgt != null)
140             tgts = tgt.getTargetPlayers();
141         else
142             tgts = AbilityFactory.getDefinedPlayers(sa.getSourceCard(), params.get("Defined"), sa);
143 
144         String valid = params.get("SacValid");
145         if (valid == null)
146             valid = "Self";
147 
148         String num = params.get("Amount");
149         num = (num == null) ? "1" : num;
150         int amount = AbilityFactory.calculateAmount(sa.getSourceCard(), num, sa);
151 
152         if (valid.equals("Self"))
153             sb.append("Sacrifice ").append(sa.getSourceCard().toString());
154         else if (valid.equals("Card.AttachedBy")) {
155             Card toSac = sa.getSourceCard().getEnchantingCard();
156             sb.append(toSac.getController()).append(" sacrifices ").append(toSac).append(".");
157         } else {
158             for (Player p : tgts)
159                 sb.append(p.getName()).append(" ");
160 
161             String msg = params.get("SacMessage");
162             if (msg == null)
163                 msg = valid;
164 
165             sb.append("Sacrifices ").append(amount).append(" ").append(msg).append(".");
166         }
167 
168         Ability_Sub abSub = sa.getSubAbility();
169         if (abSub != null)
170             sb.append(abSub.getStackDescription());
171 
172         return sb.toString();
173     }

176      * <p>sacrificeCanPlayAI.</p>
177      *
178      * @param af a {@link forge.card.abilityFactory.AbilityFactory} object.
179      * @param sa a {@link forge.card.spellability.SpellAbility} object.
180      * @return a boolean.
181      */
182     public static boolean sacrificeCanPlayAI(final AbilityFactory af, SpellAbility sa) {
183 
184         HashMap<String, String> params = af.getMapParams();
185         boolean chance = sacrificeTgtAI(af, sa);
186 
187         // Some additional checks based on what is being sacrificed, and who is sacrificing
188         Target tgt = af.getAbTgt();
189         if (tgt != null) {
190             String valid = params.get("SacValid");
191             String num = params.get("Amount");
192             num = (num == null) ? "1" : num;
193             int amount = AbilityFactory.calculateAmount(sa.getSourceCard(), num, sa);
194 
195             CardList list = AllZoneUtil.getPlayerCardsInPlay(AllZone.getHumanPlayer());
196             list = list.getValidCards(valid.split(","), sa.getActivatingPlayer(), sa.getSourceCard());
197 
198             if (list.size() == 0)
199                 return false;
200 
201             Card source = sa.getSourceCard();
202             if (num.equals("X") && source.getSVar(num).equals("Count$xPaid")) {
203                 // Set PayX here to maximum value.
204                 int xPay = Math.min(ComputerUtil.determineLeftoverMana(sa), amount);
205                 source.setSVar("PayX", Integer.toString(xPay));
206             }
207 
208             int half = amount / 2 + amount % 2;    // Half of amount rounded up
209 
210             // If the Human has at least half rounded up of the amount to be sacrificed, cast the spell
211             if (list.size() < half)
212                 return false;
213         }
214 
215         Ability_Sub subAb = sa.getSubAbility();
216         if (subAb != null)
217             chance &= subAb.chkAI_Drawback();
218 
219         return chance;
220     }

270     /**
271      * <p>sacrificeTgtAI.</p>
272      *
273      * @param af a {@link forge.card.abilityFactory.AbilityFactory} object.
274      * @param sa a {@link forge.card.spellability.SpellAbility} object.
275      * @return a boolean.
276      */
277     public static boolean sacrificeTgtAI(AbilityFactory af, SpellAbility sa) {
278 
279         HashMap<String, String> params = af.getMapParams();
280         Card card = sa.getSourceCard();
281         Target tgt = af.getAbTgt();
282 
283         if (tgt != null) {
284             tgt.resetTargets();
285             if (AllZone.getHumanPlayer().canTarget(sa.getSourceCard()))
286                 tgt.addTarget(AllZone.getHumanPlayer());
287             else
288                 return false;
289         } else {
290             String defined = params.get("Defined");
291             if (defined == null) {
292                 // Self Sacrifice.
293             } else if (defined.equals("Each")) {
294                 // If Sacrifice hits both players:
295                 // Only cast it if Human has the full amount of valid
296                 // Only cast it if AI doesn't have the full amount of Valid
297                 // TODO: Cast if the type is favorable: my "worst" valid is worse than his "worst" valid
298                 String valid = params.get("SacValid");
299                 String num = params.containsKey("Amount") ? params.get("Amount") : "1";
300                 int amount = AbilityFactory.calculateAmount(card, num, sa);
301 
302                 Card source = sa.getSourceCard();
303                 if (num.equals("X") && source.getSVar(num).equals("Count$xPaid")) {
304                     // Set PayX here to maximum value.
305                     amount = Math.min(ComputerUtil.determineLeftoverMana(sa), amount);
306                 }
307 
308                 CardList humanList = AllZoneUtil.getPlayerCardsInPlay(AllZone.getHumanPlayer());
309                 humanList = humanList.getValidCards(valid.split(","), sa.getActivatingPlayer(), sa.getSourceCard());
310                 CardList computerList = AllZoneUtil.getPlayerCardsInPlay(AllZone.getComputerPlayer());
311                 computerList = computerList.getValidCards(valid.split(","), sa.getActivatingPlayer(), sa.getSourceCard());
312 
313                 //Since all of the cards have remAIDeck:True, I enabled 1 for 1 (or X for X) trades for special decks
314                 if (humanList.size() < amount /*|| computerList.size() >= amount */) return false;
315             }
316         }
317 
318         return true;
319     }

321     /**
322      * <p>sacrificeResolve.</p>
323      *
324      * @param af a {@link forge.card.abilityFactory.AbilityFactory} object.
325      * @param sa a {@link forge.card.spellability.SpellAbility} object.
326      */
327     public static void sacrificeResolve(final AbilityFactory af, final SpellAbility sa) {
328         HashMap<String, String> params = af.getMapParams();
329         Card card = sa.getSourceCard();
330 
331         // Expand Sacrifice keyword here depending on what we need out of it.
332         String num = params.containsKey("Amount") ? params.get("Amount") : "1";
333         int amount = AbilityFactory.calculateAmount(card, num, sa);
334 
335         Target tgt = af.getAbTgt();
336         ArrayList<Player> tgts;
337         if (tgt != null)
338             tgts = tgt.getTargetPlayers();
339         else
340             tgts = AbilityFactory.getDefinedPlayers(sa.getSourceCard(), params.get("Defined"), sa);
341 
342         String valid = params.get("SacValid");
343         if (valid == null)
344             valid = "Self";
345 
346         String msg = params.get("SacMessage");
347         if (msg == null)
348             msg = valid;
349 
350         msg = "Sacrifice a " + msg;
351 
352         if (valid.equals("Self")) {
353             if (AllZone.getZone(sa.getSourceCard()).is(Constant.Zone.Battlefield))
354                 AllZone.getGameAction().sacrifice(sa.getSourceCard());
355         }
356         //TODO - maybe this can be done smarter...
357         else if (valid.equals("Card.AttachedBy")) {
358             Card toSac = sa.getSourceCard().getEnchantingCard();
359             if (AllZone.getZone(sa.getSourceCard()).is(Constant.Zone.Battlefield) && AllZoneUtil.isCardInPlay(toSac)) {
360                 AllZone.getGameAction().sacrifice(toSac);
361             }
362         } else if (valid.equals("TriggeredCard")) {
363             Card equipee = (Card) sa.getTriggeringObject("Card");
364             if (tgts.contains(card.getController()) && AllZoneUtil.isCardInPlay(equipee)) {
365                 AllZone.getGameAction().sacrifice(equipee);
366             }
367         } else {
368             for (Player p : tgts) {
369 
370                 if (p.isComputer())
371                     sacrificeAI(p, amount, valid, sa);
372                 else
373                     sacrificeHuman(p, amount, valid, sa, msg);
374             }
375         }
376     }
