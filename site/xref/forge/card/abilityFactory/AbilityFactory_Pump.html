31       * @param newAF a {@link forge.card.abilityFactory.AbilityFactory} object.
32       */
33      public AbilityFactory_Pump(AbilityFactory newAF) {
34          AF = newAF;
35  
36          params = AF.getMapParams();
37  
38          hostCard = AF.getHostCard();
39  
40          numAttack = (params.containsKey("NumAtt")) ? params.get("NumAtt") : "0";
41          numDefense = (params.containsKey("NumDef")) ? params.get("NumDef") : "0";
42  
43          // Start with + sign now optional
44          if (numAttack.startsWith("+"))
45              numAttack = numAttack.substring(1);
46          if (numDefense.startsWith("+"))
47              numDefense = numDefense.substring(1);
48  
49          if (params.containsKey("KW")) {
50              String tmp = params.get("KW");
51              String kk[] = tmp.split(" & ");
52  
53              Keywords.clear();
54              for (int i = 0; i < kk.length; i++)
55                  Keywords.add(kk[i]);
56          } else
57              Keywords.add("none");
58      }

183     /**
184      * <p>getPumpCreatures.</p>
185      *
186      * @param defense a int.
187      * @param attack a int.
188      * @return a {@link forge.CardList} object.
189      */
190     private CardList getPumpCreatures(final int defense, final int attack) {
191 
192         final boolean kHaste = Keywords.contains("Haste");
193         final boolean evasive = (Keywords.contains("Flying") || Keywords.contains("Horsemanship") ||
194                 Keywords.contains("HIDDEN Unblockable") || Keywords.contains("Fear") || Keywords.contains("Intimidate"));
195         final boolean kSize = !Keywords.get(0).equals("none");
196         String KWpump[] = {"none"};
197         if (!Keywords.get(0).equals("none"))
198             KWpump = Keywords.toArray(new String[Keywords.size()]);
199         final String KWs[] = KWpump;
200 
201         CardList list = AllZoneUtil.getCreaturesInPlay(AllZone.getComputerPlayer());
202         list = list.filter(new CardListFilter() {
203             public boolean addCard(Card c) {
204                 if (!CardFactoryUtil.canTarget(hostCard, c))
205                     return false;
206 
207                 if (c.getNetDefense() + defense <= 0) //don't kill the creature
208                     return false;
209 
210                 //Don't add duplicate keywords
211                 boolean hKW = c.hasAnyKeyword(KWs);
212                 if (kSize && hKW) return false;
213 
214                 //give haste to creatures that could attack with it
215                 if (c.hasSickness() && kHaste && AllZone.getPhase().isPlayerTurn(AllZone.getComputerPlayer()) && CombatUtil.canAttackNextTurn(c)
216                         && AllZone.getPhase().isBefore(Constant.Phase.Combat_Declare_Attackers))
217                     return true;
218 
219                 //give evasive keywords to creatures that can attack
220                 if (evasive && AllZone.getPhase().isPlayerTurn(AllZone.getComputerPlayer()) && CombatUtil.canAttack(c)
221                         && AllZone.getPhase().isBefore(Constant.Phase.Combat_Declare_Attackers) && c.getNetCombatDamage() > 0)
222                     return true;
223 
224                 //will the creature attack (only relevant for sorcery speed)?
225                 if (CardFactoryUtil.AI_doesCreatureAttack(c) && AllZone.getPhase().isBefore(Constant.Phase.Combat_Declare_Attackers)
226                         && AllZone.getPhase().isPlayerTurn(AllZone.getComputerPlayer()))
227                     return true;
228 
229                 //is the creature blocking and unable to destroy the attacker or would be destroyed itself?
230                 if (c.isBlocking() && (CombatUtil.blockerWouldBeDestroyed(c)
231                         || !CombatUtil.attackerWouldBeDestroyed(AllZone.getCombat().getAttackerBlockedBy(c))))
232                     return true;
233 
234                 //is the creature unblocked and the spell will pump its power?
235                 if (AllZone.getPhase().isAfter(Constant.Phase.Combat_Declare_Blockers) && AllZone.getCombat().isAttacking(c)
236                         && AllZone.getCombat().isUnblocked(c) && attack > 0)
237                     return true;
238 
239                 //is the creature in blocked and the blocker would survive
240                 if (AllZone.getPhase().isAfter(Constant.Phase.Combat_Declare_Blockers) && AllZone.getCombat().isAttacking(c)
241                         && AllZone.getCombat().isBlocked(c)
242                         && CombatUtil.blockerWouldBeDestroyed(AllZone.getCombat().getBlockers(c).get(0)))
243                     return true;
244 
245                 //if the life of the computer is in danger, try to pump potential blockers before declaring blocks
246                 if (CombatUtil.lifeInDanger(AllZone.getCombat()) && AllZone.getPhase().isAfter(Constant.Phase.Combat_Declare_Attackers)
247                 		&& AllZone.getPhase().isBefore(Constant.Phase.Main2)
248                         && CombatUtil.canBlock(c, AllZone.getCombat()) && AllZone.getPhase().isPlayerTurn(AllZone.getHumanPlayer()))
249                     return true;
250 
251                 return false;
252             }
253         });
254         return list;
255     }//getPumpCreatures()

257     /**
258      * <p>getCurseCreatures.</p>
259      *
260      * @param sa a {@link forge.card.spellability.SpellAbility} object.
261      * @param defense a int.
262      * @param attack a int.
263      * @return a {@link forge.CardList} object.
264      */
265     private CardList getCurseCreatures(SpellAbility sa, final int defense, int attack) {
266         CardList list = AllZoneUtil.getCreaturesInPlay(AllZone.getHumanPlayer());
267         list = list.filter(AllZoneUtil.getCanTargetFilter(hostCard));
268 
269         if (defense < 0 && !list.isEmpty()) { // with spells that give -X/-X, compi will try to destroy a creature
270             list = list.filter(new CardListFilter() {
271                 public boolean addCard(Card c) {
272                     if (c.getNetDefense() <= -defense) return true; // can kill indestructible creatures
273                     return (c.getKillDamage() <= -defense && !c.hasKeyword("Indestructible"));
274                 }
275             }); // leaves all creatures that will be destroyed
276         } // -X/-X end
277         else if (!list.isEmpty()) {
278             String KWpump[] = {"none"};
279             if (!Keywords.get(0).equals("none"))
280                 KWpump = Keywords.toArray(new String[Keywords.size()]);
281             final String KWs[] = KWpump;
282             final boolean addsKeywords = Keywords.size() > 0;
283 
284             if (addsKeywords) {
285                 list = list.filter(new CardListFilter() {
286                     public boolean addCard(Card c) {
287                         return !c.hasAnyKeyword(KWs);    // don't add duplicate negative keywords
288                     }
289                 });
290             }
291         }
292 
293 
294         return list;
295     }//getCurseCreatures()

297     /**
298      * <p>pumpPlayAI.</p>
299      *
300      * @param sa a {@link forge.card.spellability.SpellAbility} object.
301      * @return a boolean.
302      */
303     private boolean pumpPlayAI(SpellAbility sa) {
304         // if there is no target and host card isn't in play, don't activate
305         if (AF.getAbTgt() == null && !AllZoneUtil.isCardInPlay(hostCard))
306             return false;
307 
308         // temporarily disabled until AI is improved
309         if (AF.getAbCost().getSacCost() && sa.getSourceCard().isCreature()) return false;
310         if (AF.getAbCost().getLifeCost()) {
311             if (!AF.isCurse()) return false; //Use life only to kill creatures
312             if (AllZone.getComputerPlayer().getLife() - AF.getAbCost().getLifeAmount() < 4)
313                 return false;
314         }
315         if (AF.getAbCost().getDiscardCost() && !AF.isCurse()) {
316         	return false;
317         }
318         if (AF.getAbCost().getSubCounter()) {
319             // instead of never removing counters, we will have a random possibility of failure.
320             // all the other tests still need to pass if a counter will be removed
321             Counters count = AF.getAbCost().getCounterType();
322             double chance = .66;
323             if (count.equals(Counters.P1P1)) {    // 10% chance to remove +1/+1 to pump
324                 chance = .1;
325             } else if (count.equals(Counters.CHARGE)) { // 50% chance to remove charge to pump
326                 chance = .5;
327             }
328             Random r = MyRandom.random;
329             if (r.nextFloat() > chance)
330                 return false;
331         }
332 
333         if (!ComputerUtil.canPayCost(sa))
334             return false;
335 
336         SpellAbility_Restriction restrict = sa.getRestrictions();
337 
338         // Phase Restrictions
339         if (AllZone.getStack().size() == 0 && AllZone.getPhase().isBefore(Constant.Phase.Combat_Begin)) {
340             // Instant-speed pumps should not be cast outside of combat when the stack is empty
341             if (!AF.isCurse()) {
342                 if (!AbilityFactory.isSorcerySpeed(sa))
343                     return false;
344             }
345         } else if (AllZone.getStack().size() > 0) {
346             // TODO: pump something only if the top thing on the stack will kill it via damage
347             // or if top thing on stack will pump it/enchant it and I want to kill it
348             return false;
349         }
350 
351         int activations = restrict.getNumberTurnActivations();
352         int sacActivations = restrict.getActivationNumberSacrifice();
353         //don't risk sacrificing a creature just to pump it
354         if (sacActivations != -1 && activations >= (sacActivations - 1)) {
355             return false;
356         }
357 
358         Card source = sa.getSourceCard();
359         if (source.getSVar("X").equals("Count$xPaid"))
360             source.setSVar("PayX", "");
361 
362         int defense;
363         if (numDefense.contains("X") && source.getSVar("X").equals("Count$xPaid")) {
364             // Set PayX here to maximum value.
365             int xPay = ComputerUtil.determineLeftoverMana(sa);
366             source.setSVar("PayX", Integer.toString(xPay));
367             defense = xPay;
368         } else
369             defense = getNumDefense(sa);
370 
371         int attack;
372         if (numAttack.contains("X") && source.getSVar("X").equals("Count$xPaid")) {
373             // Set PayX here to maximum value.
374             String toPay = source.getSVar("PayX");
375 
376             if (toPay.equals("")) {
377                 int xPay = ComputerUtil.determineLeftoverMana(sa);
378                 source.setSVar("PayX", Integer.toString(xPay));
379                 attack = xPay;
380             } else
381                 attack = Integer.parseInt(toPay);
382         } else
383             attack = getNumAttack(sa);
384 
385         if (AF.getAbTgt() == null || !AF.getAbTgt().doesTarget()) {
386             ArrayList<Card> cards = AbilityFactory.getDefinedCards(sa.getSourceCard(), params.get("Defined"), sa);
387 
388             if (cards.size() == 0)
389                 return false;
390 
391             // when this happens we need to expand AI to consider if its ok for everything?
392             for (Card card : cards) {
393                 // TODO: if AI doesn't control Card and Pump is a Curse, than maybe use?
394                 if ((card.getNetDefense() + defense > 0) && (!card.hasAnyKeyword(Keywords))) {
395                     if (card.hasSickness() && Keywords.contains("Haste"))
396                         return true;
397                     else if (card.hasSickness() ^ Keywords.contains("Haste"))
398                         return false;
399                     else if (hostCard.equals(card)) {
400                         Random r = MyRandom.random;
401                         if (r.nextFloat() <= Math.pow(.6667, activations))
402                             return CardFactoryUtil.AI_doesCreatureAttack(card) && !sa.getPayCosts().getTap();
403                     } else {
404                         Random r = MyRandom.random;
405                         return (r.nextFloat() <= Math.pow(.6667, activations));
406                     }
407                 }
408             }
409         } else
410             return pumpTgtAI(sa, defense, attack, false);
411 
412         return false;
413     }//pumpPlayAI()

415     /**
416      * <p>pumpTgtAI.</p>
417      *
418      * @param sa a {@link forge.card.spellability.SpellAbility} object.
419      * @param defense a int.
420      * @param attack a int.
421      * @param mandatory a boolean.
422      * @return a boolean.
423      */
424     private boolean pumpTgtAI(SpellAbility sa, int defense, int attack, boolean mandatory) {
425         if (!mandatory && AllZone.getPhase().isAfter(Constant.Phase.Combat_Declare_Blockers_InstantAbility) && !(AF.isCurse() && defense < 0))
426             return false;
427 
428         Target tgt = AF.getAbTgt();
429         tgt.resetTargets();
430         CardList list;
431         if (AF.isCurse())  // Curse means spells with negative effect
432             list = getCurseCreatures(sa, defense, attack);
433         else
434             list = getPumpCreatures(defense, attack);
435 
436         list = list.getValidCards(tgt.getValidTgts(), sa.getActivatingPlayer(), sa.getSourceCard());
437 
438         if (AllZone.getStack().size() == 0) {
439             // If the cost is tapping, don't activate before declare attack/block
440             if (sa.getPayCosts() != null && sa.getPayCosts().getTap()) {
441                 if (AllZone.getPhase().isBefore(Constant.Phase.Combat_Declare_Attackers) && AllZone.getPhase().isPlayerTurn(AllZone.getComputerPlayer()))
442                     list.remove(sa.getSourceCard());
443                 if (AllZone.getPhase().isBefore(Constant.Phase.Combat_Declare_Blockers) && AllZone.getPhase().isPlayerTurn(AllZone.getHumanPlayer()))
444                     list.remove(sa.getSourceCard());
445             }
446         }
447 
448         if (list.isEmpty())
449             return mandatory && pumpMandatoryTarget(AF, sa, mandatory);
450 
451         while (tgt.getNumTargeted() < tgt.getMaxTargets(sa.getSourceCard(), sa)) {
452             Card t = null;
453             //boolean goodt = false;
454 
455             if (list.isEmpty()) {
456                 if (tgt.getNumTargeted() < tgt.getMinTargets(sa.getSourceCard(), sa) || tgt.getNumTargeted() == 0) {
457                     if (mandatory)
458                         return pumpMandatoryTarget(AF, sa, mandatory);
459 
460                     tgt.resetTargets();
461                     return false;
462                 } else {
463                     // TODO is this good enough? for up to amounts?
464                     break;
465                 }
466             }
467 
468             /*Not needed
469                if (AF.isCurse()){
470                    t = CardFactoryUtil.AI_getBestCreature(list);
471                    goodt = true;
472                }
473                else{
474                    while(!goodt && !list.isEmpty()) {
475                        t = CardFactoryUtil.AI_getBestCreature(list);
476                        if((t.getNetDefense() + defense) > t.getDamage()) goodt = true;
477                        else list.remove(t);
478                    }
479                }*/
480 
481             t = CardFactoryUtil.AI_getBestCreature(list);
482             tgt.addTarget(t);
483             list.remove(t);
484         }
485 
486         return true;
487     }//pumpTgtAI()

489     /**
490      * <p>pumpMandatoryTarget.</p>
491      *
492      * @param af a {@link forge.card.abilityFactory.AbilityFactory} object.
493      * @param sa a {@link forge.card.spellability.SpellAbility} object.
494      * @param mandatory a boolean.
495      * @return a boolean.
496      */
497     private boolean pumpMandatoryTarget(AbilityFactory af, SpellAbility sa, boolean mandatory) {
498         CardList list = AllZoneUtil.getCardsInPlay();
499         Target tgt = sa.getTarget();
500         list = list.getValidCards(tgt.getValidTgts(), sa.getActivatingPlayer(), sa.getSourceCard());
501 
502         if (list.size() < tgt.getMinTargets(sa.getSourceCard(), sa)) {
503             tgt.resetTargets();
504             return false;
505         }
506 
507         // Remove anything that's already been targeted
508         for (Card c : tgt.getTargetCards())
509             list.remove(c);
510 
511         CardList pref;
512         CardList forced;
513         Card source = sa.getSourceCard();
514 
515         if (af.isCurse()) {
516             pref = list.getController(AllZone.getHumanPlayer());
517             forced = list.getController(AllZone.getComputerPlayer());
518         } else {
519             pref = list.getController(AllZone.getComputerPlayer());
520             forced = list.getController(AllZone.getHumanPlayer());
521         }
522 
523         while (tgt.getNumTargeted() < tgt.getMaxTargets(source, sa)) {
524             if (pref.isEmpty())
525                 break;
526 
527             Card c;
528             if (pref.getNotType("Creature").size() == 0)
529                 c = CardFactoryUtil.AI_getBestCreature(pref);
530             else
531                 c = CardFactoryUtil.AI_getMostExpensivePermanent(pref, source, true);
532 
533             pref.remove(c);
534 
535             tgt.addTarget(c);
536         }
537 
538         while (tgt.getNumTargeted() < tgt.getMinTargets(sa.getSourceCard(), sa)) {
539             if (forced.isEmpty())
540                 break;
541 
542             Card c;
543             if (forced.getNotType("Creature").size() == 0)
544                 c = CardFactoryUtil.AI_getWorstCreature(forced);
545             else
546                 c = CardFactoryUtil.AI_getCheapestPermanent(forced, source, true);
547 
548             forced.remove(c);
549 
550             tgt.addTarget(c);
551         }
552 
553         if (tgt.getNumTargeted() < tgt.getMinTargets(sa.getSourceCard(), sa)) {
554             tgt.resetTargets();
555             return false;
556         }
557 
558         return true;
559     }//pumpMandatoryTarget()

564      *
565      * @param af a {@link forge.card.abilityFactory.AbilityFactory} object.

567      * @param mandatory a boolean.
568      * @return a boolean.
569      */
570     private boolean pumpTriggerAI(AbilityFactory af, SpellAbility sa, boolean mandatory) {
571         if (!ComputerUtil.canPayCost(sa))
572             return false;
573 
574         Card source = sa.getSourceCard();
575 
576         int defense;
577         if (numDefense.contains("X") && source.getSVar("X").equals("Count$xPaid")) {
578             // Set PayX here to maximum value.
579             int xPay = ComputerUtil.determineLeftoverMana(sa);
580             source.setSVar("PayX", Integer.toString(xPay));
581             defense = xPay;
582         } else
583             defense = getNumDefense(sa);
584 
585         int attack;
586         if (numAttack.contains("X") && source.getSVar("X").equals("Count$xPaid")) {
587             // Set PayX here to maximum value.
588             String toPay = source.getSVar("PayX");
589 
590             if (toPay.equals("")) {
591                 int xPay = ComputerUtil.determineLeftoverMana(sa);
592                 source.setSVar("PayX", Integer.toString(xPay));
593                 attack = xPay;
594             } else
595                 attack = Integer.parseInt(toPay);
596         } else
597             attack = getNumAttack(sa);
598 
599         if (sa.getTarget() == null) {
600             if (mandatory)
601                 return true;
602         } else {
603             return pumpTgtAI(sa, defense, attack, mandatory);
604         }
605 
606         return true;

611      *
612      * @param sa a {@link forge.card.spellability.SpellAbility} object.
613      * @return a boolean.
614      */
615     private boolean pumpDrawbackAI(SpellAbility sa) {
616         Card source = sa.getSourceCard();
617         int defense;
618         if (numDefense.contains("X") && source.getSVar("X").equals("Count$xPaid")) {
619             defense = Integer.parseInt(source.getSVar("PayX"));
620         } else
621             defense = getNumDefense(sa);
622 
623         int attack;
624         if (numAttack.contains("X") && source.getSVar("X").equals("Count$xPaid")) {
625             attack = Integer.parseInt(source.getSVar("PayX"));
626         } else
627             attack = getNumAttack(sa);
628 
629         if (AF.getAbTgt() == null || !AF.getAbTgt().doesTarget()) {
630             if (hostCard.isCreature()) {
631                 if (!hostCard.hasKeyword("Indestructible") && hostCard.getNetDefense() + defense <= hostCard.getDamage())
632                     return false;
633                 if (hostCard.getNetDefense() + defense <= 0)
634                     return false;
635             }
636         } else
637             return pumpTgtAI(sa, defense, attack, false);
638 
639         return true;
640     }//pumpDrawbackAI()

650         // when damageStackDescription is called, just build exactly what is happening
651         StringBuilder sb = new StringBuilder();

875     private boolean pumpAllCanPlayAI(SpellAbility sa) {
876         String valid = "";
877         Random r = MyRandom.random;
878         final Card source = sa.getSourceCard();
879         params = AF.getMapParams();
880         final int defense = getNumDefense(sa);
881 
882         boolean chance = r.nextFloat() <= Math.pow(.6667, source.getAbilityUsed()); //to prevent runaway activations
883 
884         if (params.containsKey("ValidCards")) {
885             valid = params.get("ValidCards");
886         }
887 
888         CardList comp = AllZoneUtil.getPlayerCardsInPlay(AllZone.getComputerPlayer());
889         comp = comp.getValidCards(valid, hostCard.getController(), hostCard);
890         CardList human = AllZoneUtil.getPlayerCardsInPlay(AllZone.getHumanPlayer());
891         human = human.getValidCards(valid, hostCard.getController(), hostCard);
892 
893         //only count creatures that can attack
894         human = human.filter(new CardListFilter() {
895             public boolean addCard(Card c) {
896                 return CombatUtil.canAttack(c) && !AF.isCurse();
897             }
898         });
899 
900         if (AF.isCurse()) {
901             if (defense < 0) { // try to destroy creatures
902                 comp = comp.filter(new CardListFilter() {
903                     public boolean addCard(Card c) {
904                         if (c.getNetDefense() <= -defense) return true; // can kill indestructible creatures
905                         return (c.getKillDamage() <= -defense && !c.hasKeyword("Indestructible"));
906                     }
907                 }); // leaves all creatures that will be destroyed
908                 human = human.filter(new CardListFilter() {
909                     public boolean addCard(Card c) {
910                         if (c.getNetDefense() <= -defense) return true; // can kill indestructible creatures
911                         return (c.getKillDamage() <= -defense && !c.hasKeyword("Indestructible"));
912                     }
913                 }); // leaves all creatures that will be destroyed
914             } // -X/-X end
915 
916             //evaluate both lists and pass only if human creatures are more valuable
917             if (CardFactoryUtil.evaluateCreatureList(comp) + 200 >= CardFactoryUtil.evaluateCreatureList(human))
918                 return false;
919 
920             return chance;
921         }//end Curse
922 
923         //don't use non curse PumpAll after Combat_Begin until AI is improved
924         if (AllZone.getPhase().isAfter(Constant.Phase.Combat_Begin))
925             return false;
926 
927         if (comp.size() <= human.size() || comp.size() <= 1)
928             return false;
929 
930         return (r.nextFloat() < .6667) && chance;
