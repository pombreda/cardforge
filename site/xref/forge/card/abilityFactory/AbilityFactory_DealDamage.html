139     /**
140      * <p>damageStackDescription.</p>
141      *
142      * @param af a {@link forge.card.abilityFactory.AbilityFactory} object.
143      * @param sa a {@link forge.card.spellability.SpellAbility} object.
144      * @return a {@link java.lang.String} object.
145      */
146     private String damageStackDescription(AbilityFactory af, SpellAbility sa) {
147         // when damageStackDescription is called, just build exactly what is happening
148         StringBuilder sb = new StringBuilder();
149         String name = af.getHostCard().toString();
150         int dmg = getNumDamage(sa);
151 
152         ArrayList<Object> tgts;
153         if (sa.getTarget() == null)
154             tgts = AbilityFactory.getDefinedObjects(sa.getSourceCard(), af.getMapParams().get("Defined"), sa);
155         else
156             tgts = sa.getTarget().getTargets();
157 
158         if (!(sa instanceof Ability_Sub))
159             sb.append(name).append(" -");
160         sb.append(" ");
161 
162         String conditionDesc = af.getMapParams().get("ConditionDescription");
163         if (conditionDesc != null)
164             sb.append(conditionDesc).append(" ");
165 
166         ArrayList<Card> definedSources = AbilityFactory.getDefinedCards(sa.getSourceCard(), af.getMapParams().get("DamageSource"), sa);
167         Card source = definedSources.get(0);
168 
169         if (source != sa.getSourceCard())
170             sb.append(source.toString()).append(" deals");
171         else
172             sb.append("Deals");
173 
174         sb.append(" ").append(dmg).append(" damage to ");
175 
176         for (int i = 0; i < tgts.size(); i++) {
177             if (i != 0)
178                 sb.append(" ");
179 
180             Object o = tgts.get(i);
181             if (o instanceof Card || o instanceof Player)
182                 sb.append(o.toString());
183         }
184 
185         sb.append(". ");
186 
187         if (sa.getSubAbility() != null) {
188             sb.append(sa.getSubAbility().getStackDescription());
189         }
190 
191         return sb.toString();
192     }

222     /**
223      * <p>doCanPlayAI.</p>
224      *
225      * @param saMe a {@link forge.card.spellability.SpellAbility} object.
226      * @return a boolean.
227      */
228     private boolean doCanPlayAI(SpellAbility saMe) {
229         if (!ComputerUtil.canPayCost(saMe))
230             return false;
231 
232         Cost abCost = AF.getAbCost();
233         Card source = saMe.getSourceCard();
234 
235         int dmg = 0;
236         if (damage.equals("X") && source.getSVar(damage).equals("Count$xPaid")) {
237             // Set PayX here to maximum value.
238             dmg = ComputerUtil.determineLeftoverMana(saMe);
239             source.setSVar("PayX", Integer.toString(dmg));
240         } else
241             dmg = getNumDamage(saMe);
242         boolean rr = AF.isSpell();
243 
244         // temporarily disabled until better AI
245         if (abCost.getSacCost() && !abCost.getSacThis() && AllZone.getHumanPlayer().getLife() - dmg > 0) {
246             //only sacrifice something that's supposed to be sacrificed
247             String sacType = abCost.getSacType();
248             CardList typeList = AllZoneUtil.getPlayerCardsInPlay(AllZone.getComputerPlayer());
249             typeList = typeList.getValidCards(sacType.split(","), source.getController(), source);
250             if (ComputerUtil.getCardPreference(source, "SacCost", typeList) == null)
251                 return false;
252         }
253         if (AF.getAbCost().getSubCounter()) {
254             // +1/+1 counters only if damage from this ability would kill the human, otherwise ok
255             if (AllZone.getHumanPlayer().getLife() - dmg > 0 && AF.getAbCost().getCounterType().equals(Counters.P1P1))
256                 return false;
257         }
258         if (AF.getAbCost().getLifeCost()) {
259             if (AllZone.getHumanPlayer().getLife() - dmg > 0) // only if damage from this ability would kill the human
260                 return false;
261         }
262 
263         if (source.getName().equals("Stuffy Doll")) {
264             // Now stuffy sits around for blocking
265             // TODO(sol): this should also happen if Stuffy is going to die
266             if (AllZone.getPhase().is(Constant.Phase.End_Of_Turn, AllZone.getHumanPlayer()))
267                 return true;
268             else
269                 return false;
270         }
271 
272         if (AF.isAbility()) {
273             Random r = MyRandom.random; // prevent run-away activations
274             if (r.nextFloat() <= Math.pow(.6667, AF.getHostCard().getAbilityUsed()))
275                 rr = true;
276         }
277 
278         boolean bFlag = damageTargetAI(saMe, dmg);
279         if (!bFlag)
280             return false;
281 
282         if (damage.equals("X") && source.getSVar(damage).equals("Count$xPaid")) {
283             // If I can kill my target by paying less mana, do it
284             Target tgt = saMe.getTarget();
285             if (tgt != null) {
286                 int actualPay = 0;
287                 boolean noPrevention = AF.getMapParams().containsKey("NoPrevention");
288                 ArrayList<Card> cards = tgt.getTargetCards();
289                 for (Card c : cards) {
290                     int adjDamage = c.getEnoughDamageToKill(dmg, source, false, noPrevention);
291                     if (adjDamage > actualPay && adjDamage <= dmg)
292                         actualPay = adjDamage;
293                 }
294 
295                 source.setSVar("PayX", Integer.toString(actualPay));
296             }
297         }
298 
299 
300         Ability_Sub subAb = saMe.getSubAbility();
301         if (subAb != null)
302             rr &= subAb.chkAI_Drawback();
303         return rr;
304     }

306     /**
307      * <p>shouldTgtP.</p>
308      *
309      * @param sa a {@link forge.card.spellability.SpellAbility} object.
310      * @param d a int.
311      * @param noPrevention a boolean.
312      * @return a boolean.
313      */
314     private boolean shouldTgtP(SpellAbility sa, int d, final boolean noPrevention) {
315         int restDamage = d;
316         Player human = AllZone.getHumanPlayer();
317         Player comp = AllZone.getComputerPlayer();
318 
319         if (!noPrevention)
320             restDamage = human.predictDamage(restDamage, AF.getHostCard(), false);
321         else restDamage = human.staticReplaceDamage(restDamage, AF.getHostCard(), false);
322 
323         if (restDamage == 0) return false;
324 
325         if (!human.canLoseLife()) return false;
326 
327         CardList hand = AllZoneUtil.getPlayerHand(comp);
328 
329         if (AF.isSpell()) {
330             // If this is a spell, cast it instead of discarding
331             if ((AllZone.getPhase().is(Constant.Phase.End_Of_Turn) || AllZone.getPhase().is(Constant.Phase.Main2)) &&
332                     AllZone.getPhase().isPlayerTurn(comp) && (hand.size() > comp.getMaxHandSize()))
333                 return true;
334         }
335 
336         if (human.getLife() - restDamage < 5) // if damage from this spell would drop the human to less than 5 life
337             return true;
338 
339         return false;
340     }

342     /**
343      * <p>chooseTgtC.</p>
344      *
345      * @param d a int.
346      * @param noPrevention a boolean.
347      * @param pl a {@link forge.Player} object.
348      * @param mandatory a boolean.
349      * @return a {@link forge.Card} object.
350      */
351     private Card chooseTgtC(final int d, final boolean noPrevention, final Player pl, final boolean mandatory) {
352         Target tgt = AF.getAbTgt();
353         final Card source = AF.getHostCard();
354         CardList hPlay = AllZoneUtil.getPlayerCardsInPlay(pl);
355         hPlay = hPlay.getValidCards(tgt.getValidTgts(), AllZone.getComputerPlayer(), source);
356 
357         ArrayList<Object> objects = tgt.getTargets();
358         for (Object o : objects) {
359             if (o instanceof Card) {
360                 Card c = (Card) o;
361                 if (hPlay.contains(c))
362                     hPlay.remove(c);
363             }
364         }
365         hPlay = hPlay.getTargetableCards(source);
366 
367         CardList killables = hPlay.filter(new CardListFilter() {
368             public boolean addCard(Card c) {
369                 return (c.getEnoughDamageToKill(d, source, false, noPrevention) <= d)
370                         && !ComputerUtil.canRegenerate(c)
371                         && !(c.getSVar("SacMe").length() > 0);
372             }
373         });
374 
375         Card targetCard;
376         if (pl.isHuman() && killables.size() > 0) {
377             targetCard = CardFactoryUtil.AI_getBestCreature(killables);
378 
379             return targetCard;
380         }
381 
382         if (!mandatory)
383             return null;
384 
385         if (hPlay.size() > 0) {
386             if (pl.isHuman())
387                 targetCard = CardFactoryUtil.AI_getBestCreature(hPlay);
388             else
389                 targetCard = CardFactoryUtil.AI_getWorstCreature(hPlay);
390 
391             return targetCard;
392         }
393 
394         return null;
395     }

413     /**
414      * <p>damageChoosingTargets.</p>
415      *
416      * @param saMe a {@link forge.card.spellability.SpellAbility} object.
417      * @param tgt a {@link forge.card.spellability.Target} object.
418      * @param dmg a int.
419      * @param mandatory a boolean.
420      * @return a boolean.
421      */
422     private boolean damageChoosingTargets(SpellAbility saMe, Target tgt, int dmg, boolean mandatory) {
423         boolean noPrevention = AF.getMapParams().containsKey("NoPrevention");
424 
425         // target loop
426         tgt.resetTargets();
427 
428         while (tgt.getNumTargeted() < tgt.getMaxTargets(saMe.getSourceCard(), saMe)) {
429             // TODO: Consider targeting the planeswalker
430             if (tgt.canTgtCreatureAndPlayer()) {
431 
432                 if (shouldTgtP(saMe, dmg, noPrevention)) {
433                     if (tgt.addTarget(AllZone.getHumanPlayer()))
434                         continue;
435                 }
436 
437                 Card c = chooseTgtC(dmg, noPrevention, AllZone.getHumanPlayer(), mandatory);
438                 if (c != null) {
439                     tgt.addTarget(c);
440                     continue;
441                 }
442 
443                 // When giving priority to targeting Creatures for mandatory triggers
444                 // feel free to add the Human after we run out of good targets
445 
446                 // TODO: add check here if card is about to die from something on the stack
447                 // or from taking combat damage
448                 boolean freePing = mandatory || AbilityFactory.playReusable(saMe);
449 
450                 if (freePing && tgt.addTarget(AllZone.getHumanPlayer()))
451                     continue;
452             } else if (tgt.canTgtCreature()) {
453                 Card c = chooseTgtC(dmg, noPrevention, AllZone.getHumanPlayer(), mandatory);
454                 if (c != null) {
455                     tgt.addTarget(c);
456                     continue;
457                 }
458             }
459 
460             // TODO: Improve Damage, we shouldn't just target the player just because we can
461             else if (tgt.canTgtPlayer()) {
462                 if (tgt.addTarget(AllZone.getHumanPlayer()))
463                     continue;
464             }
465             // fell through all the choices, no targets left?
466             if ((tgt.getNumTargeted() < tgt.getMinTargets(saMe.getSourceCard(), saMe)
467                     || tgt.getNumTargeted() == 0)) {
468                 if (!mandatory) {
469                     tgt.resetTargets();
470                     return false;
471                 } else {
472                     // If the trigger is mandatory, gotta choose my own stuff now
473                     return damageChooseRequiredTargets(saMe, tgt, dmg, mandatory);
474                 }
475             } else {
476                 // TODO is this good enough? for up to amounts?
477                 break;
478             }
479         }
480         return true;
481     }

511      *
512      * @param saMe a {@link forge.card.spellability.SpellAbility} object.
513      * @param tgt a {@link forge.card.spellability.Target} object.
514      * @param dmg a int.
515      * @param mandatory a boolean.
516      * @return a boolean.
517      */
518     private boolean damageChooseRequiredTargets(SpellAbility saMe, Target tgt, int dmg, boolean mandatory) {
519         // this is for Triggered targets that are mandatory
520         boolean noPrevention = AF.getMapParams().containsKey("NoPrevention");
521 
522         while (tgt.getNumTargeted() < tgt.getMinTargets(saMe.getSourceCard(), saMe)) {
523             // TODO: Consider targeting the planeswalker
524             if (tgt.canTgtCreature()) {
525                 Card c = chooseTgtC(dmg, noPrevention, AllZone.getComputerPlayer(), mandatory);
526                 if (c != null) {
527                     tgt.addTarget(c);
528                     continue;
529                 }
530             }
531 
532             if (tgt.canTgtPlayer()) {
533                 if (tgt.addTarget(AllZone.getComputerPlayer()))
534                     continue;
535             }
536 
537             // if we get here then there isn't enough targets, this is the only time we can return false
538             return false;
539         }
540         return true;
541     }

543     /**
544      * <p>damageDoTriggerAI.</p>
545      *
546      * @param af a {@link forge.card.abilityFactory.AbilityFactory} object.
547      * @param sa a {@link forge.card.spellability.SpellAbility} object.
548      * @param mandatory a boolean.
549      * @return a boolean.
550      */
551     private boolean damageDoTriggerAI(AbilityFactory af, SpellAbility sa, boolean mandatory) {
552         if (!ComputerUtil.canPayCost(sa) && !mandatory)
553             return false;
554 
555         Card source = sa.getSourceCard();
556         int dmg;
557         if (damage.equals("X") && source.getSVar(damage).equals("Count$xPaid")) {
558             // Set PayX here to maximum value.
559             dmg = ComputerUtil.determineLeftoverMana(sa);
560             source.setSVar("PayX", Integer.toString(dmg));
561         } else
562             dmg = getNumDamage(sa);
563 
564         Target tgt = sa.getTarget();
565         if (tgt == null) {
566             // If it's not mandatory check a few things
567             if (!mandatory && !damageChooseNontargeted(sa, dmg)) {
568                 return false;
569             }
570         } else {
571             if (!damageChoosingTargets(sa, tgt, dmg, mandatory) && !mandatory)
572                 return false;
573 
574             if (damage.equals("X") && source.getSVar(damage).equals("Count$xPaid")) {
575                 // If I can kill my target by paying less mana, do it
576                 int actualPay = 0;
577                 boolean noPrevention = AF.getMapParams().containsKey("NoPrevention");
578                 ArrayList<Card> cards = tgt.getTargetCards();
579                 for (Card c : cards) {
580                     int adjDamage = c.getEnoughDamageToKill(dmg, source, false, noPrevention);
581                     if (adjDamage > actualPay)
582                         actualPay = adjDamage;
583                 }
584 
585                 source.setSVar("PayX", Integer.toString(actualPay));
586             }
587         }
588 
589 
590         if (sa.getSubAbility() != null)
591             return sa.getSubAbility().doTrigger(mandatory);
592 
593         return true;
594     }

597     /**
598      * <p>doResolve.</p>
599      *
600      * @param saMe a {@link forge.card.spellability.SpellAbility} object.
601      */
602     private void doResolve(SpellAbility saMe) {
603         HashMap<String, String> params = AF.getMapParams();
604 
605         int dmg = getNumDamage(saMe);
606 
607         boolean noPrevention = params.containsKey("NoPrevention");
608 
609         ArrayList<Object> tgts;
610         if (saMe.getTarget() == null)
611             tgts = AbilityFactory.getDefinedObjects(saMe.getSourceCard(), params.get("Defined"), saMe);
612         else
613             tgts = saMe.getTarget().getTargets();
614 
615         boolean targeted = (AF.getAbTgt() != null);
616 
617         ArrayList<Card> definedSources = AbilityFactory.getDefinedCards(saMe.getSourceCard(), params.get("DamageSource"), saMe);
618         Card source = definedSources.get(0);
619 
620         for (Object o : tgts) {
621             if (o instanceof Card) {
622                 Card c = (Card) o;
623                 if (AllZoneUtil.isCardInPlay(c) && (!targeted || CardFactoryUtil.canTarget(AF.getHostCard(), c))) {
624                     if (noPrevention)
625                         c.addDamageWithoutPrevention(dmg, source);
626                     else
627                         c.addDamage(dmg, source);
628                 }
629 
630             } else if (o instanceof Player) {
631                 Player p = (Player) o;
632                 if (!targeted || p.canTarget(AF.getHostCard())) {
633                     if (noPrevention)
634                         p.addDamageWithoutPrevention(dmg, source);
635                     else
636                         p.addDamage(dmg, source);
637                 }
638             }
639         }
640     }
