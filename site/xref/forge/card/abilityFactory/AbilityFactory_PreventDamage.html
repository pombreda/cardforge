127      * @param sa a {@link forge.card.spellability.SpellAbility} object.

129      */
130     private static String preventDamageStackDescription(AbilityFactory af, SpellAbility sa) {
131         final HashMap<String, String> params = af.getMapParams();
132         StringBuilder sb = new StringBuilder();
133         Card host = af.getHostCard();
134 
135         ArrayList<Object> tgts;
136         if (sa.getTarget() == null)
137             tgts = AbilityFactory.getDefinedObjects(sa.getSourceCard(), params.get("Defined"), sa);
138         else
139             tgts = sa.getTarget().getTargets();
140 
141         if (sa instanceof Ability_Sub)
142             sb.append(" ");
143         else
144             sb.append(host).append(" - ");
145 
146         sb.append("Prevent the next ");
147         sb.append(params.get("Amount"));
148         sb.append(" that would be dealt to ");
149         for (int i = 0; i < tgts.size(); i++) {
150             if (i != 0)
151                 sb.append(" ");
152 
153             Object o = tgts.get(i);
154             if (o instanceof Card) {
155                 Card tgtC = (Card) o;
156                 if (tgtC.isFaceDown()) sb.append("Morph");
157                 else sb.append(tgtC);
158             } else sb.append(o.toString());
159         }
160         sb.append(" this turn.");
161 
162         Ability_Sub abSub = sa.getSubAbility();
163         if (abSub != null) {
164             sb.append(abSub.getStackDescription());
165         }
166 
167         return sb.toString();
168     }

171      * <p>preventDamageCanPlayAI.</p>
172      *
173      * @param af a {@link forge.card.abilityFactory.AbilityFactory} object.
174      * @param sa a {@link forge.card.spellability.SpellAbility} object.
175      * @return a boolean.
176      */
177     private static boolean preventDamageCanPlayAI(final AbilityFactory af, final SpellAbility sa) {
178         final HashMap<String, String> params = af.getMapParams();
179         final Card hostCard = af.getHostCard();
180         boolean chance = false;
181 
182         // temporarily disabled until better AI
183         if (af.getAbCost().getSacCost()) return false;
184         if (af.getAbCost().getSubCounter())
185             if (af.getAbCost().getCounterType().equals(Counters.P1P1))
186                 return false;
187         if (af.getAbCost().getLifeCost()) return false;
188 
189         if (!ComputerUtil.canPayCost(sa))
190             return false;
191 
192         Target tgt = af.getAbTgt();
193         if (tgt == null) {
194             // As far as I can tell these Defined Cards will only have one of them
195             ArrayList<Object> objects = AbilityFactory.getDefinedObjects(sa.getSourceCard(), params.get("Defined"), sa);
196 
197             if (AllZone.getStack().size() > 0) {
198                 // check stack for something that will kill this
199             } else {
200                 if (AllZone.getPhase().is(Constant.Phase.Combat_Declare_Blockers_InstantAbility)) {
201                     boolean flag = false;
202                     for (Object o : objects) {
203                         if (o instanceof Card) {
204                             Card c = (Card) o;
205                             flag |= CombatUtil.combatantWouldBeDestroyed(c);
206                         } else if (o instanceof Player) {
207                             Player p = (Player) o;
208                             flag |= (p.isComputer() && CombatUtil.lifeInDanger(AllZone.getCombat()));
209                         }
210                     }
211 
212                     chance = flag;
213                 } else {    // if nothing on the stack, and it's not declare blockers. no need to regen
214                     return false;
215                 }
216             }
217         } else if (AllZone.getStack().size() == 0 && AllZone.getPhase().is(Constant.Phase.Combat_Declare_Blockers_InstantAbility)) {
218             tgt.resetTargets();
219 
220             if (tgt.canTgtPlayer() && CombatUtil.wouldLoseLife(AllZone.getCombat())
221                     && (CombatUtil.lifeInDanger(AllZone.getCombat()) || sa.isAbility())) {
222                 tgt.addTarget(AllZone.getComputerPlayer());
223                 chance = true;
224             } else {
225                 // filter AIs battlefield by what I can target
226                 CardList targetables = AllZoneUtil.getPlayerCardsInPlay(AllZone.getComputerPlayer());
227                 targetables = targetables.getValidCards(tgt.getValidTgts(), AllZone.getComputerPlayer(), hostCard);
228 
229                 if (targetables.size() == 0)
230                     return false;
231                 CardList combatants = targetables.getType("Creature");
232                 CardListUtil.sortByEvaluateCreature(combatants);
233 
234                 for (Card c : combatants) {
235                     if (CombatUtil.combatantWouldBeDestroyed(c)) {
236                         tgt.addTarget(c);
237                         chance = true;
238                         break;
239                     }
240                 }
241             }
242         }
243 
244         Ability_Sub subAb = sa.getSubAbility();
245         if (subAb != null)
246             chance &= subAb.chkAI_Drawback();
247 
248         return chance;
249     }

254      * @param af a {@link forge.card.abilityFactory.AbilityFactory} object.

256      * @param mandatory a boolean.
257      * @return a boolean.

259     private static boolean doPreventDamageTriggerAI(final AbilityFactory af, final SpellAbility sa, boolean mandatory) {
260         boolean chance = false;
261 
262         if (!ComputerUtil.canPayCost(sa))
263             return false;
264 
265         Target tgt = sa.getTarget();
266         if (tgt == null) {
267             // If there's no target on the trigger, just say yes.
268             chance = true;
269         } else {
270             chance = preventDamageMandatoryTarget(af, sa, mandatory);
271         }
272 
273         Ability_Sub subAb = sa.getSubAbility();
274         if (subAb != null)
275             chance &= subAb.doTrigger(mandatory);
276 
277         return chance;
278     }

281      * <p>preventDamageMandatoryTarget.</p>
282      *
283      * @param af a {@link forge.card.abilityFactory.AbilityFactory} object.
284      * @param sa a {@link forge.card.spellability.SpellAbility} object.
285      * @param mandatory a boolean.
286      * @return a boolean.
287      */
288     private static boolean preventDamageMandatoryTarget(AbilityFactory af, SpellAbility sa, boolean mandatory) {
289         final Card hostCard = af.getHostCard();
290         Target tgt = sa.getTarget();
291         tgt.resetTargets();
292         // filter AIs battlefield by what I can target
293         CardList targetables = AllZoneUtil.getCardsInPlay();
294         targetables = targetables.getValidCards(tgt.getValidTgts(), AllZone.getComputerPlayer(), hostCard);
295         CardList compTargetables = targetables.getController(AllZone.getComputerPlayer());
296 
297         if (targetables.size() == 0)
298             return false;
299 
300         if (!mandatory && compTargetables.size() == 0)
301             return false;
302 
303         if (compTargetables.size() > 0) {
304             CardList combatants = compTargetables.getType("Creature");
305             CardListUtil.sortByEvaluateCreature(combatants);
306             if (AllZone.getPhase().is(Constant.Phase.Combat_Declare_Blockers_InstantAbility)) {
307                 for (Card c : combatants) {
308                     if (CombatUtil.combatantWouldBeDestroyed(c)) {
309                         tgt.addTarget(c);
310                         return true;
311                     }
312                 }
313             }
314 
315             // TODO see if something on the stack is about to kill something i can target
316 
317             tgt.addTarget(combatants.get(0));
318             return true;
319         }
320 
321         tgt.addTarget(CardFactoryUtil.AI_getCheapestPermanent(targetables, hostCard, true));
322         return true;
323     }

331     private static void preventDamageResolve(final AbilityFactory af, final SpellAbility sa) {
332         final HashMap<String, String> params = af.getMapParams();
333         int numDam = AbilityFactory.calculateAmount(af.getHostCard(), params.get("Amount"), sa);
334 
335         ArrayList<Object> tgts;
336         if (sa.getTarget() == null)
337             tgts = AbilityFactory.getDefinedObjects(sa.getSourceCard(), params.get("Defined"), sa);
338         else
339             tgts = sa.getTarget().getTargets();
340 
341         boolean targeted = (af.getAbTgt() != null);
342 
343         for (Object o : tgts) {
344             if (o instanceof Card) {
345                 Card c = (Card) o;
346                 if (AllZoneUtil.isCardInPlay(c) && (!targeted || CardFactoryUtil.canTarget(af.getHostCard(), c))) {
347                     c.addPreventNextDamage(numDam);
348                 }
349 
350             } else if (o instanceof Player) {
351                 Player p = (Player) o;
352                 if (!targeted || p.canTarget(af.getHostCard())) {
353                     p.addPreventNextDamage(numDam);
354                 }
355             }
356         }
