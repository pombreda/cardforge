97  		StringBuilder sb = new StringBuilder();
98  
99  		if (!(sa instanceof Ability_Sub))
100 			sb.append(sa.getSourceCard().getName()).append(" - ");
101 		else
102 			sb.append(" ");
103 		
104 		String conditionDesc = af.getMapParams().get("ConditionDescription");
105 		if (conditionDesc != null)
106 			sb.append(conditionDesc).append(" ");
107 
108 		sb.append(" Attach to ");
109 		
110 		ArrayList<Object> targets;
111 
112 		// Should never allow more than one Attachment per card
113 		Target tgt = af.getAbTgt();
114 		if (tgt != null)
115 			targets = tgt.getTargets();
116 		else
117 			targets = AbilityFactory.getDefinedObjects(sa.getSourceCard(), af.getMapParams().get("Defined"), sa);
118 		
119 		for(Object o : targets)
120 			sb.append(o).append(" ");
121 
122 		Ability_Sub abSub = sa.getSubAbility();
123 		if (abSub != null)
124 			sb.append(abSub.getStackDescription());
125 
126 		return sb.toString();

145 		Target tgt = sa.getTarget();
146 		Card attachSource = sa.getSourceCard();
147 		// TODO AttachSource is currently set for the Source of the Spell, but at some point can support attaching a different card
148 		
149 		CardList list = AllZoneUtil.getCardsInZone(tgt.getZone());
150 		list = list.getValidCards(tgt.getValidTgts(), sa.getActivatingPlayer(), sa.getSourceCard());
151 		
152 		if (list.size() == 0)
153 			return null;
154 		
155 		Card c = null;
156 		String logic = params.get("AILogic");
157 		
158 		if ("GainControl".equals(logic))
159 			c = attachAIControlPreference(sa, list, mandatory, attachSource);
160 		
161 		// TODO: Pump and ChangeType definitely don't have enough AI yet, Curse might but better wait on converting
162 		
163 		else if ("Curse".equals(logic))
164 			c = attachAICursePreference(sa, list, mandatory, attachSource);
165 		else if ("Pump".equals(logic))
166 			c = attachAIPumpPreference(sa, list, mandatory, attachSource);
167 		else if ("ChangeType".equals(logic))	// Evil Presence, Spreading Seas
168 			c = attachAIChangeTypePreference(sa, list, mandatory, attachSource);
169 		// Does KeepTapped need it's own list? Might be able to lump with Curse
170 		
171 		if (c == null && mandatory){
172 			list.shuffle();
173 			c = list.get(0);
174 		}
175 		
176 		return c;

344 		Random r = MyRandom.random;
345 		HashMap<String,String> params = af.getMapParams();
346 		Cost abCost = sa.getPayCosts();
347 		final Card source = sa.getSourceCard();
348 
349 		if (abCost != null){
350 			// No Aura spells have Additional Costs 
351 			
352 			// AI currently disabled for these costs
353 			if (abCost.getSacCost()){
354 
355 			}
356 			if (abCost.getLifeCost())	 return false;
357 			if (abCost.getDiscardCost()) return false;
358 
359 			if (abCost.getSubCounter()){
360 				//non +1/+1 counters should be used
361 				if (abCost.getCounterType().equals(Counters.P1P1)){
362 					// A card has a 25% chance per counter to be able to pass through here
363 					// 4+ counters will always pass. 0 counters will never
364 				int currentNum = source.getCounters(abCost.getCounterType());
365 				double percent = .25 * (currentNum / abCost.getCounterNum());
366 				if (percent <= r.nextFloat())
367 					return false;
368 				}
369 			}
370 		}
371 
372 		// prevent run-away activations - first time will always return true
373 		boolean chance = r.nextFloat() <= Math.pow(.6667, source.getAbilityUsed());
374 		
375 		// Attach spells always have a target
376 		Target tgt = sa.getTarget();
377 		if (tgt != null){
378 			tgt.resetTargets();
379 			if (!attachPreference(af, sa, params, tgt, false))
380 				return false;
381 		}
382 		
383 		if (abCost.getMana().contains("X") && source.getSVar("X").equals("Count$xPaid")){
384 			// Set PayX here to maximum value. (Endless Scream and Venarian Gold)
385 			int xPay = ComputerUtil.determineLeftoverMana(sa);
386 			
387 			if (xPay == 0)
388 				return false;
389 			
390 			source.setSVar("PayX", Integer.toString(xPay));
391 		}
392 		
393 		chance &= r.nextFloat() <= .6667;
394 		return chance;
395 	}

459 		if (card.isEnchanting()){
460 			// If this Card is already Enchanting something
461 			// Need to unenchant it, then clear out the commands
462 			Card oldEnchanted = card.getEnchantingCard();
463 			card.removeEnchanting(oldEnchanted);
464 			card.clearEnchantCommand();
465 			card.clearUnEnchantCommand();
466 			card.clearTriggers();	// not sure if cleartriggers is needed?
467 		}
468 		
469 		if (gainControl){
470 			// Handle GainControl Auras
471 			final Player[] pl = new Player[1];
472 			pl[0] = tgt.getController();
473 			
474 	        Command onEnchant = new Command() {
475 				private static final long serialVersionUID = -2519887209491512000L;
476 	
477 				public void execute() {
478 	                if(card.isEnchanting()) {
479 	                    Card crd = card.getEnchanting().get(0);
480 	                    pl[0] = crd.getController();
481 	                    
482 	                    AllZone.getGameAction().changeController(new CardList(crd), pl[0], card.getController());
483 	                }
484 	            }//execute()
485 	        };//Command
486

488 				private static final long serialVersionUID = 3426441132121179288L;
489

491 	                if(card.isEnchanting()) {
492 	                    Card crd = card.getEnchanting().get(0);
493 	                    if(AllZoneUtil.isCardInPlay(crd)) {
494 	                        AllZone.getGameAction().changeController(new CardList(crd), crd.getController(), pl[0]);
495 	                    }
496 	                }
497 	                
498 	            }//execute()
499 	        };//Command
500 	        
501 	        Command onLeavesPlay = new Command() {
502 				private static final long serialVersionUID = -639204333673364477L;
503 	
504 				public void execute() {
505 	                if(card.isEnchanting()) {
506 	                    Card crd = card.getEnchanting().get(0);
507 	                    card.unEnchantCard(crd);
508 	                }
509 	            }
510 	        };//Command
511 			
512 	        // Add Enchant Commands
513 	        card.addEnchantCommand(onEnchant);
514 	        card.addUnEnchantCommand(onUnEnchant);
515 	        card.addLeavesPlayCommand(onLeavesPlay);
516 		}
