14  //GainControl specific params:
15  //	LoseControl - the lose control conditions (as a comma separated list)
16  //			-Untap - source card becomes untapped
17  //			-LoseControl - you lose control of source card
18  //			-LeavesPlay - source card leaves the battlefield
19  //			-PowerGT - (not implemented yet for Old Man of the Sea)
20  //	AddKWs	- Keywords to add to the controlled card (as a "&"-separated list; like Haste, Sacrifice CARDNAME at EOT, any standard keyword)
21  //  OppChoice - set to True if opponent chooses creature (for Preacher) - not implemented yet
22  //	Untap	- set to True if target card should untap when control is taken
23  //	DestroyTgt - actions upon which the tgt should be destroyed.  same list as LoseControl
24  //	NoRegen - set if destroyed creature can't be regenerated.  used only with DestroyTgt

66          }

68              destroyOn = new ArrayList<String>(Arrays.asList(params.get("DestroyTgt").split(",")));
69          }
70          if (params.containsKey("NoRegen")) {
71              bNoRegen = true;
72          }

180      * <p>gainControlStackDescription.</p>
181      *
182      * @param sa a {@link forge.card.spellability.SpellAbility} object.
183      * @return a {@link java.lang.String} object.
184      */
185     private String gainControlStackDescription(SpellAbility sa) {
186         StringBuilder sb = new StringBuilder();
187 
188         if (!(sa instanceof Ability_Sub))
189             sb.append(sa.getSourceCard()).append(" - ");
190         else
191             sb.append(" ");
192 
193         ArrayList<Card> tgtCards;
194 
195         Target tgt = af.getAbTgt();
196         if (tgt != null)
197             tgtCards = tgt.getTargetCards();
198         else {
199             tgtCards = AbilityFactory.getDefinedCards(hostCard, params.get("Defined"), sa);
200         }
201 
202         ArrayList<Player> newController = AbilityFactory.getDefinedPlayers(sa.getSourceCard(), params.get("NewController"), sa);
203         if (newController.size() == 0) newController.add(sa.getActivatingPlayer());
204 
205         sb.append(newController).append(" gains control of ");
206 
207         for (Card c : tgtCards) {
208             sb.append(" ");
209             if (c.isFaceDown()) sb.append("Morph");
210             else sb.append(c);
211         }
212         sb.append(".");
213 
214         Ability_Sub abSub = sa.getSubAbility();
215         if (abSub != null) {
216             sb.append(abSub.getStackDescription());
217         }
218 
219         return sb.toString();
220     }

222     /**
223      * <p>gainControlTgtAI.</p>
224      *
225      * @param sa a {@link forge.card.spellability.SpellAbility} object.
226      * @return a boolean.
227      */
228     private boolean gainControlTgtAI(SpellAbility sa) {
229         boolean hasCreature = false;
230         boolean hasArtifact = false;
231         boolean hasEnchantment = false;
232         boolean hasLand = false;
233 
234         Target tgt = af.getAbTgt();
235 
236         //if Defined, then don't worry about targeting
237         if (tgt == null) {
238             return true;
239         }
240 
241         CardList list = AllZoneUtil.getPlayerCardsInPlay(AllZone.getHumanPlayer());
242         list = list.getValidCards(tgt.getValidTgts(), hostCard.getController(), hostCard);
243         //AI won't try to grab cards that are filtered out of AI decks on purpose
244         list = list.filter(new CardListFilter() {
245             public boolean addCard(Card c) {
246                 Hashtable<String, String> vars = c.getSVars();
247                 return !vars.containsKey("RemAIDeck") && CardFactoryUtil.canTarget(hostCard, c);
248             }
249         });
250 
251         if (list.isEmpty())
252             return false;
253 
254         // Don't steal something if I can't Attack without, or prevent it from blocking at least
255         if (lose != null && lose.contains("EOT") && AllZone.getPhase().isAfter(Constant.Phase.Combat_Declare_Blockers))
256             return false;
257 
258         while (tgt.getNumTargeted() < tgt.getMaxTargets(sa.getSourceCard(), sa)) {
259             Card t = null;
260             for (Card c : list) {
261                 if (c.isCreature()) hasCreature = true;
262                 if (c.isArtifact()) hasArtifact = true;
263                 if (c.isLand()) hasLand = true;
264                 if (c.isEnchantment()) hasEnchantment = true;
265             }
266 
267             if (list.isEmpty()) {
268                 if (tgt.getNumTargeted() < tgt.getMinTargets(sa.getSourceCard(), sa) || tgt.getNumTargeted() == 0) {
269                     tgt.resetTargets();
270                     return false;
271                 } else {
272                     // TODO is this good enough? for up to amounts?
273                     break;
274                 }
275             }
276 
277             if (hasCreature) t = CardFactoryUtil.AI_getBestCreature(list);
278             else if (hasArtifact) t = CardFactoryUtil.AI_getBestArtifact(list);
279             else if (hasLand) t = CardFactoryUtil.AI_getBestLand(list);
280             else if (hasEnchantment) t = CardFactoryUtil.AI_getBestEnchantment(list, sa.getSourceCard(), true);
281             else t = CardFactoryUtil.AI_getMostExpensivePermanent(list, sa.getSourceCard(), true);
282 
283             tgt.addTarget(t);
284             list.remove(t);
285 
286             hasCreature = false;
287             hasArtifact = false;
288             hasLand = false;
289             hasEnchantment = false;
290         }
291 
292         return true;
293 
294     }

296     /**
297      * <p>gainControlResolve.</p>
298      *
299      * @param sa a {@link forge.card.spellability.SpellAbility} object.
300      */
301     private void gainControlResolve(SpellAbility sa) {
302         ArrayList<Card> tgtCards;
303 
304         Target tgt = af.getAbTgt();
305         if (tgt != null)
306             tgtCards = tgt.getTargetCards();
307         else {
308             tgtCards = AbilityFactory.getDefinedCards(hostCard, params.get("Defined"), sa);
309         }
310         //tgtCards.add(hostCard);
311 
312         ArrayList<Player> newController = AbilityFactory.getDefinedPlayers(sa.getSourceCard(), params.get("NewController"), sa);
313         if (newController.size() == 0) newController.add(sa.getActivatingPlayer());
314 
315         int size = tgtCards.size();
316         for (int j = 0; j < size; j++) {
317             final Card tgtC = tgtCards.get(j);
318             final Player originalController = tgtC.getController();
319 
320             movedCards[j] = tgtC;
321             hostCard.addGainControlTarget(tgtC);
322 
323             if (AllZoneUtil.isCardInPlay(tgtC) && CardFactoryUtil.canTarget(hostCard, tgtC)) {
324 
325                 AllZone.getGameAction().changeController(new CardList(tgtC), tgtC.getController(), newController.get(0));
326 
327                 if (bUntap) tgtC.untap();
328 
329                 if (null != kws) {
330                     for (String kw : kws) {
331                         tgtC.addExtrinsicKeyword(kw);
332                     }
333                 }
334             }
335 
336             //end copied
337 
338             if (lose != null) {
339                 if (lose.contains("LeavesPlay")) {
340                     hostCard.addLeavesPlayCommand(getLoseControlCommand(j, originalController));
341                 }
342                 if (lose.contains("Untap")) {
343                     hostCard.addUntapCommand(getLoseControlCommand(j, originalController));
344                 }
345                 if (lose.contains("LoseControl")) {
346                     hostCard.addChangeControllerCommand(getLoseControlCommand(j, originalController));
347                 }
348                 if (lose.contains("EOT")) {
349                     AllZone.getEndOfTurn().addAt(getLoseControlCommand(j, originalController));
350                 }
351             }
352 
353             if (destroyOn != null) {
354                 if (destroyOn.contains("LeavesPlay")) {
355                     hostCard.addLeavesPlayCommand(getDestroyCommand(j));
356                 }
357                 if (destroyOn.contains("Untap")) {
358                     hostCard.addUntapCommand(getDestroyCommand(j));
359                 }
360                 if (destroyOn.contains("LoseControl")) {
361                     hostCard.addChangeControllerCommand(getDestroyCommand(j));
362                 }
363             }
364 
365             //for Old Man of the Sea - 0 is hardcoded since it only allows 1 target
366             hostCard.clearGainControlReleaseCommands();
367             hostCard.addGainControlReleaseCommand(getLoseControlCommand(0, originalController));
368 
369         }//end foreach target
370     }

414     private Command getDestroyCommand(final int i) {
415         final Command destroy = new Command() {
416             private static final long serialVersionUID = 878543373519872418L;
417 
418             public void execute() {
419                 final Card c = movedCards[i];
420                 Ability ability = new Ability(hostCard, "0") {
421                     public void resolve() {
422 
423                         if (bNoRegen) {
424                             AllZone.getGameAction().destroyNoRegeneration(c);
425                         } else {
426                             AllZone.getGameAction().destroy(c);
427                         }
428                     }
429                 };
430                 StringBuilder sb = new StringBuilder();
431                 sb.append(hostCard).append(" - destroy ").append(c.getName()).append(".");
432                 if (bNoRegen) sb.append("  It can't be regenerated.");
433                 ability.setStackDescription(sb.toString());
434 
435                 AllZone.getStack().addSimultaneousStackEntry(ability);
436             }
437 
438         };
439         return destroy;
440     }

452 
453             public void execute() {
454                 Card c = movedCards[i];
455                 //ArrayList<Card> c = hostCard.getGainControlTargets();
456                 if (null == c) return;
457 
458                 if (AllZoneUtil.isCardInPlay(c)) {
459                     AllZone.getGameAction().changeController(new CardList(c), c.getController(), originalController);
460 
461                     if (bTapOnLose) c.tap();
462 
463                     if (null != kws) {
464                         for (String kw : kws) {
465                             c.removeExtrinsicKeyword(kw);
466                         }
467                     }
468                 }//if
469                 hostCard.clearGainControlTargets();
470                 hostCard.clearGainControlReleaseCommands();
471                 movedCards[i] = null;
472             }//execute()
473         };
474 
475         return loseControl;
