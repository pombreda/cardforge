128     private static boolean hasProtectionFrom(Card card, String color) {
129     	ArrayList<String> onlyColors = new ArrayList<String>(Arrays.asList(Constant.Color.onlyColors));
130     	
131     	//make sure we have a valid color
132     	if(!onlyColors.contains(color)) return false;
133     	
134     	String protection = "Protection from " + color;
135     	if(card.hasKeyword(protection)) return true;
136     	else return false;
137     }

139     private static boolean hasProtectionFromAny(Card card, ArrayList<String> colors) {
140     	boolean protect = false;
141     	for(String color : colors) {
142     		protect |= hasProtectionFrom(card, color);
143     	}
144     	return protect;
145     }

147     private static boolean hasProtectionFromAll(Card card, ArrayList<String> colors) {
148     	boolean protect = true;
149     	if(colors.size() < 1) return false;
150     	
151     	for(String color : colors) {
152     		protect &= hasProtectionFrom(card, color);
153     	}
154     	return protect;

163     private static CardList getProtectCreatures(AbilityFactory af, SpellAbility sa) {
164     	final Card hostCard = af.getHostCard();
165     	final ArrayList<String> gains = getProtectionList(hostCard, af.getMapParams());
166 
167         CardList list = AllZoneUtil.getCreaturesInPlay(AllZone.getComputerPlayer());
168         list = list.filter(new CardListFilter() {
169             public boolean addCard(Card c) {
170                 if(!CardFactoryUtil.canTarget(hostCard, c))
171                     return false;
172 
173                 //Don't add duplicate protections
174                 if(hasProtectionFromAll(c, gains)) return false;
175 
176                 //will the creature attack (only relevant for sorcery speed)?
177                 if(CardFactoryUtil.AI_doesCreatureAttack(c) && AllZone.getPhase().isBefore(Constant.Phase.Combat_Declare_Attackers)
178                         && AllZone.getPhase().isPlayerTurn(AllZone.getComputerPlayer()))
179                     return true;
180 
181                 //is the creature blocking and unable to destroy the attacker or would be destroyed itself?
182                 if(c.isBlocking() && (CombatUtil.blockerWouldBeDestroyed(c)
183                         || CombatUtil.attackerWouldBeDestroyed(AllZone.getCombat().getAttackerBlockedBy(c))))
184                     return true;
185 
186                 //is the creature in blocked and the blocker would survive
187                 if(AllZone.getPhase().isAfter(Constant.Phase.Combat_Declare_Blockers) && AllZone.getCombat().isAttacking(c)
188                         && AllZone.getCombat().isBlocked(c)
189                         && CombatUtil.blockerWouldBeDestroyed(AllZone.getCombat().getBlockers(c).get(0)))
190                     return true;
191 
192                 return false;
193             }
194         });
195         return list;
196     }//getProtectCreatures()

208         // if there is no target and host card isn't in play, don't activate

210             return false;
211 
212         // temporarily disabled until AI is improved
213         if(af.getAbCost().getSacCost() && sa.getSourceCard().isCreature()) return false;
214         if(af.getAbCost().getLifeCost()) {
215             if(!af.isCurse()) return false; //Use life only to kill creatures
216             if(AllZone.getComputerPlayer().getLife() - af.getAbCost().getLifeAmount() < 4)
217                 return false;
218         }
219         if(af.getAbCost().getSubCounter()) {
220             // instead of never removing counters, we will have a random possibility of failure.
221             // all the other tests still need to pass if a counter will be removed
222             Counters count = af.getAbCost().getCounterType();
223             double chance = .66;
224             if(count.equals(Counters.P1P1)) {    // 10% chance to remove +1/+1 to protect
225                 chance = .1;
226             } 
227             else if(count.equals(Counters.CHARGE)) { // 50% chance to remove charge to protect
228                 chance = .5;
229             }
230             Random r = MyRandom.random;
231             if(r.nextFloat() > chance)
232                 return false;
233         }
234 
235         if(!ComputerUtil.canPayCost(sa))
236             return false;
237 
238         // Phase Restrictions
239         if(AllZone.getStack().size() == 0 && AllZone.getPhase().isBefore(Constant.Phase.Combat_FirstStrikeDamage)) {
240         	// Instant-speed protections should not be cast outside of combat when the stack is empty
241         	if(!AbilityFactory.isSorcerySpeed(sa))
242         		return false;
243         }
244         else if(AllZone.getStack().size() > 0) {
245         	// TODO: protection something only if the top thing on the stack will kill it via damage or destroy
246         	return false;
247         }
248 
249         if(af.getAbTgt() == null || !af.getAbTgt().doesTarget()) {
250         	ArrayList<Card> cards = AbilityFactory.getDefinedCards(sa.getSourceCard(), params.get("Defined"), sa);
251 
252         	if(cards.size() == 0)
253         		return false;
254 
255         	/*
256             // when this happens we need to expand AI to consider if its ok for everything?
257             for (Card card : cards) {
258                 // TODO: if AI doesn't control Card and Pump is a Curse, than maybe use?
259 
260             }*/
261         }

347     	
348         CardList list = AllZoneUtil.getCardsInPlay();
349         Target tgt = sa.getTarget();
350         list = list.getValidCards(tgt.getValidTgts(), sa.getActivatingPlayer(), sa.getSourceCard());
351 
352         if(list.size() < tgt.getMinTargets(sa.getSourceCard(), sa)) {
353             tgt.resetTargets();
354             return false;
355         }
356 
357         // Remove anything that's already been targeted
358         for(Card c : tgt.getTargetCards())
359             list.remove(c);
360 
361         CardList pref = list.getController(AllZone.getComputerPlayer());
362         pref = pref.filter(new CardListFilter() {
363         	public boolean addCard(Card c) {
364         		return !hasProtectionFromAll(c, getProtectionList(host, params));
365         	}
366         });
367         CardList pref2 = list.getController(AllZone.getComputerPlayer());
368         pref = pref.filter(new CardListFilter() {
369         	public boolean addCard(Card c) {
370         		return !hasProtectionFromAny(c, getProtectionList(host, params));
371         	}
372         });
373         CardList forced = list.getController(AllZone.getHumanPlayer());
374         Card source = sa.getSourceCard();
375 
376         while(tgt.getNumTargeted() < tgt.getMaxTargets(source, sa)) {
377             if(pref.isEmpty())
378                 break;
379 
380             Card c;
381             if(pref.getNotType("Creature").size() == 0)
382                 c = CardFactoryUtil.AI_getBestCreature(pref);
383             else
384                 c = CardFactoryUtil.AI_getMostExpensivePermanent(pref, source, true);
385 
386             pref.remove(c);
387 
388             tgt.addTarget(c);
389         }
390         
391         while(tgt.getNumTargeted() < tgt.getMaxTargets(source, sa)) {
392             if(pref2.isEmpty())
393                 break;
394 
395             Card c;
396             if(pref2.getNotType("Creature").size() == 0)
397                 c = CardFactoryUtil.AI_getBestCreature(pref2);
398             else
399                 c = CardFactoryUtil.AI_getMostExpensivePermanent(pref2, source, true);
400 
401             pref2.remove(c);
402 
403             tgt.addTarget(c);
404         }
405 
406         while(tgt.getNumTargeted() < tgt.getMinTargets(source, sa)) {
407             if(forced.isEmpty())
408                 break;
409 
410             Card c;
411             if(forced.getNotType("Creature").size() == 0)
412                 c = CardFactoryUtil.AI_getWorstCreature(forced);
413             else
414                 c = CardFactoryUtil.AI_getCheapestPermanent(forced, source, true);
415 
416             forced.remove(c);
417 
418             tgt.addTarget(c);
419         }
420 
421         if(tgt.getNumTargeted() < tgt.getMinTargets(sa.getSourceCard(), sa)) {
422             tgt.resetTargets();
423             return false;
424         }
425 
426         return true;

580             tgtCards = AbilityFactory.getDefinedCards(host, params.get("Defined"), sa);

589                 continue;
590 
591             // if this is a target, make sure we can still target now
592             if(tgt != null && !CardFactoryUtil.canTarget(host, tgtC))
593             	continue;
594 
595             for(String gain : gains) {
596             	tgtC.addExtrinsicKeyword("Protection from "+gain);
597             }
598 
599             if(!params.containsKey("Permanent")) {
600             	// If not Permanent, remove protection at EOT
601             	final Command untilEOT = new Command() {
602             		private static final long serialVersionUID = 7682700789217703789L;
603 
604             		public void execute() {
605             			if(AllZoneUtil.isCardInPlay(tgtC)) {
606             				for (String gain : gains) {
607             					tgtC.removeExtrinsicKeyword("Protection from "+gain);
608             				}
609             			}
610             		}
611             	};
612                 if(params.containsKey("UntilEndOfCombat")) AllZone.getEndOfCombat().addUntil(untilEOT);
613                 else AllZone.getEndOfTurn().addUntil(untilEOT);
614             }
615         }
616     }//protectResolve()

618     private static ArrayList<String> getProtectionList(Card host, HashMap<String,String> params) {
619     	final ArrayList<String> gains = new ArrayList<String>();
620     	
621     	String gainStr = params.get("Gains"); 
622     	if(gainStr.equals("Choice")) {
623     		String choices = params.get("Choices");
624 
625     		// Replace AnyColor with the 5 colors
626     		if (choices.contains("AnyColor")){
627     			gains.addAll(Arrays.asList(Constant.Color.onlyColors));
628     			choices = choices.replaceAll("AnyColor,?", "");
629     		}
630     		// Add any remaining choices
631     		if (choices.length() > 0)
632     			gains.addAll(Arrays.asList(choices.split(",")));
633     	}
634     	else {
635     		gains.addAll(Arrays.asList(gainStr.split(",")));
636     	}
637         return gains;
638     }
