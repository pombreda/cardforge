224     /**
225      * <p>setDefendingDamage.</p>
226      */
227     public void setDefendingDamage() {
228         defendingDamageMap.clear();
229         CardList att = new CardList(getAttackers());
230         // sum unblocked attackers' power
231         for (int i = 0; i < att.size(); i++) {
232             if (!isBlocked(att.get(i))
233                     || (getBlockers(att.get(i)).size() == 0 && att.get(i).hasKeyword("Trample"))) {
234 
235                 int damageDealt = att.get(i).getNetCombatDamage();
236 
237                 if (damageDealt > 0) {
238                     //if the creature has first strike do not do damage in the normal combat phase
239                     if (!att.get(i).hasFirstStrike() || att.get(i).hasDoubleStrike())
240                         addDefendingDamage(damageDealt, att.get(i));
241                 }
242             } // ! isBlocked...
243         }// for
244     }

253         boolean needsFirstStrike = false;
254         defendingDamageMap.clear();
255         CardList att = new CardList(getAttackers());
256         // sum unblocked attackers' power
257         for (int i = 0; i < att.size(); i++) {
258             if (!isBlocked(att.get(i))) {
259 
260                 int damageDealt = att.get(i).getNetCombatDamage();
261 
262                 if (damageDealt > 0) {
263                     // if the creature has first strike or double strike do damage in the first strike combat phase
264                     if (att.get(i).hasFirstStrike() || att.get(i).hasDoubleStrike()) {
265                         addDefendingDamage(damageDealt, att.get(i));
266                         needsFirstStrike = true;
267                     }
268                 }
269             }
270         } // for
271 
272         return needsFirstStrike;

277      * <p>addDefendingDamage.</p>
278      *
279      * @param n a int.
280      * @param source a {@link forge.Card} object.
281      */
282     public void addDefendingDamage(int n, Card source) {
283         String slot = getDefenderByAttacker(source).toString();
284         Object o = defenders.get(Integer.parseInt(slot));
285 
286         if (o instanceof Card) {
287             Card pw = (Card) o;
288             pw.addAssignedDamage(n, source);
289 
290             return;
291         }
292 
293         if (!defendingDamageMap.containsKey(source))
294             defendingDamageMap.put(source, n);
295         else {
296             defendingDamageMap.put(source, defendingDamageMap.get(source) + n);
297         }
298     }

478      * <p>removeFromCombat.</p>
479      *
480      * @param c a {@link forge.Card} object.
481      */
482     public void removeFromCombat(Card c) {
483         // is card an attacker?
484         CardList att = new CardList(getAttackers());
485         if (att.contains(c)) {
486             map.remove(c);
487             attackerToDefender.remove(c);
488         } else// card is a blocker
489         {
490             for (Card a : att)
491                 if (getBlockers(a).contains(c)) {
492                     getList(a).remove(c);
493                     // TODO if Declare Blockers and Declare Blockers (Abilities) merge this logic needs to be tweaked
494                     if (getBlockers(a).size() == 0 && AllZone.getPhase().is(Constant.Phase.Combat_Declare_Blockers))
495                         blocked.remove(a);
496                 }
497         }
498         // update combat
499         CombatUtil.showCombat();
500     }// removeFromCombat()

539     /**
540      * <p>setAssignedFirstStrikeDamage.</p>
541      *
542      * @return a boolean.
543      */
544     public boolean setAssignedFirstStrikeDamage() {
545 
546         boolean needFirstStrike = setDefendingFirstStrikeDamage();
547 
548         CardList block;
549         CardList attacking = new CardList(getAttackers());
550 
551         for (int i = 0; i < attacking.size(); i++) {
552 
553             Card attacker = attacking.get(i);
554             block = getBlockers(attacker);
555 
556             int damageDealt = attacker.getNetCombatDamage();
557 
558             // attacker always gets all blockers' attack
559 
560             for (Card b : block) {
561                 if (b.hasFirstStrike() || b.hasDoubleStrike()) {
562                     needFirstStrike = true;
563                     int attack = b.getNetCombatDamage();
564                     attacker.addAssignedDamage(attack, b);
565                 }
566             }
567 
568             if (block.size() == 0) {
569                 // this damage is assigned to a player by setDefendingFirstStrikeDamage()
570             } else if (attacker.hasFirstStrike() || attacker.hasDoubleStrike()) {
571                 needFirstStrike = true;
572                 if (getAttackingPlayer().isHuman()) {// human attacks
573                     if (attacker.hasKeyword("Trample") || block.size() > 1)
574                         AllZone.getDisplay().assignDamage(attacker, block, damageDealt);
575                     else block.get(0).addAssignedDamage(damageDealt, attacking.get(i));
576                 } else {// computer attacks
577                     distributeAIDamage(attacker, block, damageDealt);
578                 }
579             }// if(hasFirstStrike || doubleStrike)
580         }// for
581         return needFirstStrike;
582     }// setAssignedFirstStrikeDamage()

628     /**
629      * <p>distributeAIDamage.</p>
630      *
631      * @param attacker a {@link forge.Card} object.
632      * @param block a {@link forge.CardList} object.
633      * @param damage a int.
634      */
635     private void distributeAIDamage(Card attacker, CardList block, int damage) {
636         Card c = attacker;
637 
638         if (block.size() == 1) {
639 
640             Card blocker = block.get(0);
641 
642             // trample
643             if (attacker.hasKeyword("Trample")) {
644 
645                 int damageNeeded = 0;
646 
647                 //TODO: if the human can be killed distribute only the minimum of damage to the blocker
648 
649                 damageNeeded = blocker.getEnoughDamageToKill(damage, attacker, true);
650 
651                 if (damageNeeded > damage)
652                     damageNeeded = Math.min(blocker.getLethalDamage(), damage);
653                 else
654                     damageNeeded = Math.max(blocker.getLethalDamage(), damageNeeded);
655 
656                 int trample = damage - damageNeeded;
657 
658                 if (0 < trample)     // If Extra trample damage, assign to defending player/planeswalker
659                     this.addDefendingDamage(trample, attacker);
660 
661                 blocker.addAssignedDamage(damageNeeded, attacker);
662             } else blocker.addAssignedDamage(damage, attacker);
663         }// 1 blocker
664         else {
665             boolean killsAllBlockers = true;//Does the attacker deal lethal damage to all blockers
666             for (Card b : block) {
667                 int enoughDamageToKill = b.getEnoughDamageToKill(damage, attacker, true);
668                 if (enoughDamageToKill <= damage) {
669                     damage -= enoughDamageToKill;
670                     CardList cl = new CardList();
671                     cl.add(attacker);
672 
673                     b.addAssignedDamage(enoughDamageToKill, c);
674                 } else killsAllBlockers = false;
675             }// for
676 
677             // if attacker has no trample, and there's damage left, assign the rest
678             // to a random blocker
679             if (damage > 0
680                     && !(c.hasKeyword("Trample")
681                     && killsAllBlockers == true)) {
682                 int index = CardUtil.getRandomIndex(block);
683                 block.get(index).addAssignedDamage(damage, c);
684                 damage = 0;
685             } else if (c.hasKeyword("Trample")
686                     && killsAllBlockers == true) {
687                 this.addDefendingDamage(damage, c);
688             }
689         }
690     }// setAssignedDamage()

692     /**
693      * <p>dealAssignedDamage.</p>
694      */
695     public static void dealAssignedDamage() {
696         // This function handles both Regular and First Strike combat assignment
697         Player player = AllZone.getCombat().getDefendingPlayer();
698 
699         boolean bFirstStrike = AllZone.getPhase().is(Constant.Phase.Combat_FirstStrikeDamage);
700 
701         HashMap<Card, Integer> defMap = AllZone.getCombat().getDefendingDamageMap();
702 
703         for (Entry<Card, Integer> entry : defMap.entrySet()) {
704             player.addCombatDamage(entry.getValue(), entry.getKey());
705         }
706 
707         CardList unblocked = new CardList(bFirstStrike ? AllZone.getCombat().getUnblockedAttackers() :
708                 AllZone.getCombat().getUnblockedFirstStrikeAttackers());
709 
710         for (int j = 0; j < unblocked.size(); j++) {
711             if (bFirstStrike)
712                 CombatUtil.checkUnblockedAttackers(unblocked.get(j));
713             else {
714                 if (!unblocked.getCard(j).hasFirstStrike() && !unblocked.getCard(j).hasDoubleStrike())
715                     CombatUtil.checkUnblockedAttackers(unblocked.get(j));
716             }
717         }
718 
719         // this can be much better below here...
720 
721         CardList combatants = new CardList();
722         combatants.addAll(AllZone.getCombat().getAttackers());
723         combatants.addAll(AllZone.getCombat().getAllBlockers());
724         combatants.addAll(AllZone.getCombat().getDefendingPlaneswalkers());
725 
726         Card c;
727         for (int i = 0; i < combatants.size(); i++) {
728             c = combatants.get(i);
729 
730             // if no assigned damage to resolve, move to next
731             if (c.getTotalAssignedDamage() == 0)
732                 continue;
733 
734             Map<Card, Integer> assignedDamageMap = c.getAssignedDamageMap();
735             HashMap<Card, Integer> damageMap = new HashMap<Card, Integer>();
736 
737             for (Entry<Card, Integer> entry : assignedDamageMap.entrySet()) {
738                 Card crd = entry.getKey();
739                 damageMap.put(crd, entry.getValue());
740             }
741             c.addCombatDamage(damageMap);
742 
743             damageMap.clear();
744             c.clearAssignedDamage();
745         }
746 
747         //This was deeper before, but that resulted in the stack entry acting like before.
748 
749     }
