129         area.setEditable(false);
130         area.setLineWrap(true);

132 
133         //Button is not modified, String gets the automatic listener to hide the dialog 
134         Object[] options = {
135                 new JButton(new SaveAction(area)), getLocalized(BUTTON_CLOSE), new JButton(new ExitAction())};
136         JOptionPane pane = new JOptionPane(new JScrollPane(area), ERROR_MESSAGE, DEFAULT_OPTION, null, options,
137                 options[1]);
138         JDialog dlg = pane.createDialog(null, getLocalized(TITLE));
139         dlg.setResizable(true);
140         dlg.setVisible(true);
141         dlg.dispose();

144     /**
145      * Prints the error message for the specified exception to the print writer
146      *
147      * @param pw a {@link java.io.PrintWriter} object.
148      * @param ex a {@link java.lang.Throwable} object.
149      * @param message a {@link java.lang.String} object.
150      */
151     private static void printError(PrintWriter pw, Throwable ex, String message) {
152         if (message != null) System.err.println(message);
153         ex.printStackTrace();
154 
155         pw.printf(getLocalized(MESSAGE), getProperty(FORUM), getProperty(MAIL),
156                 message != null ? message : ex.getMessage(), getProperty(VERSION),
157                 System.getProperty(nameOS), System.getProperty(versionOS), System.getProperty(architectureOS),
158                 System.getProperty(versionJava), System.getProperty(vendorJava));
159         ex.printStackTrace(pw);
160     }

162     /**
163      * Prints the error message to the print writer, showing all running threads' stack traces.
164      *
165      * @param pw a {@link java.io.PrintWriter} object.
166      * @param message a {@link java.lang.String} object.
167      */
168     private static void printError(PrintWriter pw, String message) {
169         System.err.println(message);
170 
171         pw.printf(getLocalized(MESSAGE), getProperty(FORUM), getProperty(MAIL), message, getProperty(VERSION),
172                 System.getProperty(nameOS), System.getProperty(versionOS), System.getProperty(architectureOS),
173                 System.getProperty(versionJava), System.getProperty(vendorJava));
174         Map<Thread, StackTraceElement[]> traces = Thread.getAllStackTraces();
175         for (Entry<Thread, StackTraceElement[]> e : traces.entrySet()) {
176             pw.println();
177             pw.printf("%s (%s):%n", e.getKey().getName(), e.getKey().getId());
178             for (StackTraceElement el : e.getValue()) {
179                 pw.println(el);
180             }
181         }

184     private static class SaveAction extends AbstractAction {
185 
186         private static final long serialVersionUID = 9146834661273525959L;
187 
188         private static JFileChooser c;
189 
190         private JTextArea area;
191 
192         public SaveAction(JTextArea area) {
193             super(getLocalized(BUTTON_SAVE));
194             putValue(ACCELERATOR_KEY, KeyStroke.getKeyStroke(VK_S, CTRL_DOWN_MASK));
195             this.area = area;
196         }
197 
198         public void actionPerformed(ActionEvent e) {
199             if (c == null) c = new JFileChooser();
200 
201             File f;
202             for (int i = 0; ; i++) {
203                 String name = String.format("%TF-%02d.txt", System.currentTimeMillis(), i);
204                 f = new File(name);
205                 if (!f.exists()) break;
206             }
207             c.setSelectedFile(f);
208             c.showSaveDialog(null);
209             f = c.getSelectedFile();
210 
211             try {
212                 BufferedWriter bw = new BufferedWriter(new FileWriter(f));
213                 bw.write(area.getText());
214                 bw.close();
215             } catch (IOException ex) {
216                 showError(ex, getLocalized(ERRORS.SAVE_MESSAGE));
217             }
218         }
219     }
