24              return true;

35       * <p>handleUntap.</p>
36       */
37      public static void handleUntap() {
38          Player turn = AllZone.getPhase().getPlayerTurn();
39  
40          AllZone.getPhase().turnReset();
41  
42          AllZone.getCombat().reset();
43          AllZone.getCombat().setAttackingPlayer(turn);
44          AllZone.getCombat().setDefendingPlayer(turn.getOpponent());
45  
46          // For tokens a player starts the game with they don't recover from Sum. Sickness on first turn
47          if (turn.getTurn() > 0) {
48              CardList list = AllZoneUtil.getPlayerCardsInPlay(turn);
49              for (Card c : list)
50                  c.setSickness(false);
51          }
52          turn.incrementTurn();
53  
54          AllZone.getGameAction().resetActivationsPerTurn();
55  
56          CardList lands = AllZoneUtil.getPlayerLandsInPlay(turn);
57          lands = lands.filter(AllZoneUtil.untapped);
58          turn.setNumPowerSurgeLands(lands.size());
59  
60          // anything before this point happens regardless of whether the Untap phase is skipped
61  
62          if (skipUntap(turn)) {
63              AllZone.getPhase().setNeedToNextPhase(true);
64              return;
65          }
66  
67          // Phasing would happen here
68  
69          doUntap();
70  
71          //otherwise land seems to stay tapped when it is really untapped
72          AllZone.getHumanBattlefield().updateObservers();
73  
74          AllZone.getPhase().setNeedToNextPhase(true);
75      }

81          Player player = AllZone.getPhase().getPlayerTurn();

84          for (Card c : list) {

88          }
89  
90          list = list.filter(new CardListFilter() {
91              public boolean addCard(Card c) {
92                  if (!canUntap(c)) return false;
93                  if (canOnlyUntapOneLand() && c.isLand()) return false;
94                  if ((AllZoneUtil.isCardInPlay("Damping Field") || AllZoneUtil.isCardInPlay("Imi Statue"))
95                          && c.isArtifact()) return false;
96                  if ((AllZoneUtil.isCardInPlay("Smoke") || AllZoneUtil.isCardInPlay("Stoic Angel")
97                          || AllZoneUtil.isCardInPlay("Intruder Alarm")) && c.isCreature()) return false;
98                  return true;
99              }
100         });
101 
102         for (Card c : list) {
103             if (c.hasKeyword("You may choose not to untap CARDNAME during your untap step.")) {
104                 if (c.isTapped()) {
105                     if (c.getController().isHuman()) {
106                         String prompt = "Untap " + c.getName() + "?";
107                         boolean defaultNo = false;
108                         if (c.getGainControlTargets().size() > 0) {
109                             ArrayList<Card> targets = c.getGainControlTargets();
110                             prompt += "\r\n" + c + " is controlling: ";
111                             for (Card target : targets) {
112                                 prompt += target;
113                                 if(AllZoneUtil.isCardInPlay(target)) defaultNo |= true;
114                             }
115                         }
116                         if (GameActionUtil.showYesNoDialog(c, prompt, defaultNo)) {
117                             c.untap();
118                         }
119                     } else {  //computer
120                         //if it is controlling something by staying tapped, leave it tapped
121                         //if not, untap it
122                     	if (c.getGainControlTargets().size() > 0) {
123                             ArrayList<Card> targets = c.getGainControlTargets();
124                             boolean untap = true;

147         for (Card oppCard : opp)

150         //end opponent untapping during your untap phase
151 
152         if (canOnlyUntapOneLand()) {
153             if (AllZone.getPhase().getPlayerTurn().isComputer()) {
154                 //search for lands the computer has and only untap 1
155                 CardList landList = AllZoneUtil.getPlayerLandsInPlay(AllZone.getComputerPlayer());

262     /**
263      * <p>canUntap.</p>
264      *
265      * @param c a {@link forge.Card} object.
266      * @return a boolean.
267      */
268     public static boolean canUntap(Card c) {
269 
270         if (c.hasKeyword("CARDNAME doesn't untap during your untap step.")
271                 || c.hasKeyword("This card doesn't untap during your next untap step.")) return false;
272 
273         CardList allp = AllZoneUtil.getCardsInPlay();
274         for (Card ca : allp) {
275             if (ca.hasStartOfKeyword("Permanents don't untap during their controllers' untap steps")) {
276                 int KeywordPosition = ca.getKeywordPosition("Permanents don't untap during their controllers' untap steps");
277                 String parse = ca.getKeyword().get(KeywordPosition).toString();
278                 String k[] = parse.split(":");
279                 final String restrictions[] = k[1].split(",");
280                 final Card card = ca;
281                 if (c.isValidCard(restrictions, card.getController(), card)) return false;
282             }
283         } // end of Permanents don't untap during their controllers' untap steps
284 
285         return true;
286     }

312     public static void handleUpkeep() {
313     	Player turn = AllZone.getPhase().getPlayerTurn();
314         if (skipUpkeep()) {
315             // Slowtrips all say "on the next turn's upkeep" if there is no upkeep next turn, the trigger will never occur.
316             turn.clearSlowtripList();
317             turn.getOpponent().clearSlowtripList();
318             AllZone.getPhase().setNeedToNextPhase(true);
319             return;
320         }
321 
322         AllZone.getUpkeep().executeUntil(turn);
323         GameActionUtil.executeUpkeepEffects();
324     }

361      *
362      * @param player a {@link forge.Player} object.
363      * @return a boolean.
364      */
365     private static boolean skipDraw(Player player) {
366         // starting player skips his draw
367         if (AllZone.getPhase().getTurn() == 1) {
368             return true;
369         }
370 
371         CardList list = AllZoneUtil.getPlayerCardsInPlay(player);
372 
373         if (list.containsName("Necropotence") || list.containsName("Yawgmoth's Bargain") || list.containsName("Recycle") ||
374                 list.containsName("Dragon Appeasement") || list.containsName("Null Profusion") || list.containsName("Colfenor's Plans") ||
375                 list.containsName("Psychic Possession") || list.containsName("Solitary Confinement") ||
376                 list.containsName("Symbiotic Deployment"))
377             return true;
378 
379         return false;

427             // Make sure exalted effects get applied only once per combat
428 
429         }
430 
431         for (Card c : list)

433         AllZone.getStack().unfreezeStack();

437      * <p>handleDeclareBlockers.</p>
438      */
439     public static void handleDeclareBlockers() {
440         verifyCombat();
441 
442         AllZone.getStack().freezeStack();
443 
444         AllZone.getCombat().setUnblocked();
445 
446         CardList list = new CardList();
447         list.addAll(AllZone.getCombat().getAllBlockers());
448 
449         list = list.filter(new CardListFilter() {
450             public boolean addCard(Card c) {
451                 return !c.getCreatureBlockedThisCombat();
452             }
453         });
454 
455         CardList attList = new CardList();
456         attList.addAll(AllZone.getCombat().getAttackers());
457 
458         CombatUtil.checkDeclareBlockers(list);
459 
460         for (Card a : attList) {
461             CardList blockList = AllZone.getCombat().getBlockers(a);
462             for (Card b : blockList)
463                 CombatUtil.checkBlockedAttackers(a, b);
464         }
465 
466         AllZone.getStack().unfreezeStack();
467         CombatUtil.showCombat();
468     }
