125             private static final long serialVersionUID = 8120331222693706164L;
126

128             protected Card[] getCards() {
129                 return CardFactoryUtil.getGraveyardActivationCards(AllZone.getHumanPlayer()).toArray();
130             }
131 
132             @Override
133             protected void doAction(Card c) {
134                 if (!c.isLand()) {
135                     SpellAbility[] sa = c.getSpellAbility();
136                     sa[1].setActivatingPlayer(AllZone.getHumanPlayer());
137                     if (sa[1].canPlay() && !c.isUnCastable()) AllZone.getGameAction().playSpellAbility(sa[1]);
138                 } else    // PlayLand checks if the land can be played
139                     AllZone.getHumanPlayer().playLand(c);
140             }

143         COMPUTER_REMOVED_ACTION = new ZoneAction(AllZone.getComputerExile(), COMPUTER_REMOVED);

352     /** {@inheritDoc} */
353     public void showMessage(String s) {
354         messageArea.setText(s);
355 
356         Border border = null;
357         int thickness = 3;
358 
359         if (AllZone.getStack().size() > 0 && AllZone.getStack().peekInstance().getActivatingPlayer().isComputer())
360             border = BorderFactory.createLineBorder(new Color(0, 255, 255), thickness);
361         else if (s.contains("Main"))
362             border = BorderFactory.createLineBorder(new Color(30, 0, 255), thickness);
363         else if (s.contains("To Block"))
364             border = BorderFactory.createLineBorder(new Color(13, 179, 0), thickness);
365         else if (s.contains("Play Instants and Abilities") || s.contains("Combat") || s.contains("Damage"))
366             border = BorderFactory.createLineBorder(new Color(255, 174, 0), thickness);
367         else if (s.contains("Declare Attackers"))
368             border = BorderFactory.createLineBorder(new Color(255, 0, 0), thickness);
369         else if (s.contains("Upkeep") || s.contains("Draw") || s.contains("End of Turn"))
370             border = BorderFactory.createLineBorder(new Color(200, 0, 170), thickness);
371         else
372             border = new EmptyBorder(1, 1, 1, 1);
373 
374         messageArea.setBorder(border);
375     }

401 
402             @Override
403             public void mousePressed(MouseEvent e) {
404                 inputControl.selectPlayer(AllZone.getHumanPlayer());
405             }
406         });
407 
408         //self play (land) ---- Mouse
409         playerLandPanel.addMouseListener(new MouseAdapter() {
410 
411             @Override
412             public void mousePressed(MouseEvent e) {
413                 Object o = playerLandPanel.getComponentAt(e.getPoint());
414                 if (o instanceof CardPanel) {
415                     CardPanel cardPanel = (CardPanel) o;
416 
417                     if (cardPanel.getCard().isUntapped()) {
418                         MP3Player mp3 = new MP3Player("tap.mp3");
419                         mp3.play();
420                     }
421 
422                     if (cardPanel.getCard().isTapped()
423                             && (inputControl.input instanceof Input_PayManaCost || inputControl.input instanceof Input_PayManaCost_Ability)) {
424 
425                         while (cardPanel.connectedCard != null) {
426                             cardPanel = cardPanel.connectedCard;
427 
428                             if (cardPanel.getCard().isUntapped()) {
429 
430                                 break;
431                             }
432                         }
433                     }
434 
435                     inputControl.selectCard(cardPanel.getCard(), AllZone.getHumanBattlefield());
436 
437                 }
438             }
439         });
440         //self play (no land) ---- Mouse
441         playerCreaturePanel.addMouseListener(new MouseAdapter() {
442 
443             @Override
444             public void mousePressed(MouseEvent e) {
445                 Object o = playerCreaturePanel.getComponentAt(e.getPoint());
446                 if (o instanceof CardPanel) {
447                     CardPanel cardPanel = (CardPanel) o;
448 
449                     CardList att = new CardList(AllZone.getCombat().getAttackers());
450                     //CardList block = AllZone.getCombat().getAllBlockers();
451 
452                     if ((cardPanel.getCard().isTapped()
453                             || cardPanel.getCard().hasSickness()
454                             || ((cardPanel.getCard().hasKeyword("Vigilance")) && att.contains(cardPanel.getCard())))
455                             && (inputControl.input instanceof Input_Attack)) {
456                         while (cardPanel.connectedCard != null) {
457                             cardPanel = cardPanel.connectedCard;
458                             if (cardPanel.getCard().isUntapped() && !cardPanel.getCard().hasSickness()) {
459                                 break;
460                             }
461                         }
462                     }
463                     //right click:
464                     if (e.isMetaDown()) {
465                         if (att.contains(cardPanel.getCard())
466                                 && (inputControl.input instanceof Input_Attack)
467                                 && !cardPanel.getCard().hasKeyword("CARDNAME attacks each turn if able.")) {
468                             cardPanel.getCard().untap();
469                             AllZone.getCombat().removeFromCombat(cardPanel.getCard());
470                         } else if (inputControl.input instanceof Input_Block) {
471                             Card crd = cardPanel.getCard();
472                             if (crd.getController().isHuman())
473                                 AllZone.getCombat().removeFromCombat(crd);
474                             ((Input_Block) inputControl.input).removeFromAllBlocking(crd);
475                         }
476                     } else inputControl.selectCard(cardPanel.getCard(), AllZone.getHumanBattlefield());
477                 }
478             }
479         });
480         //self hand ---- Mouse
481         playerHandPanel.addMouseListener(new MouseAdapter() {
482 
483             @Override
484             public void mousePressed(MouseEvent e) {
485                 Object o = playerHandPanel.getComponentAt(e.getPoint());
486                 if (o instanceof CardPanel) {
487                     CardContainer cardPanel = (CardContainer) o;
488                     inputControl.selectCard(cardPanel.getCard(), AllZone.getHumanHand());
489                     okButton.requestFocusInWindow();
490                 }
491             }
492         });
493 
494         //*****************************************************************
495         //computer
496 
497         //computer play (land) ---- Mouse
498         oppLandPanel.addMouseListener(new MouseAdapter() {
499 
500             @Override
501             public void mousePressed(MouseEvent e) {
502                 Object o = oppLandPanel.getComponentAt(e.getPoint());
503                 if (o instanceof CardPanel) {
504                     CardContainer cardPanel = (CardContainer) o;
505                     inputControl.selectCard(cardPanel.getCard(), AllZone.getComputerBattlefield());
506                 }
507             }
508         });

1153             else if (phase.equals(Constant.Phase.Combat_Begin))
1154                 return cbAIBeginCombat.isSelected();
1155             else if (phase.equals(Constant.Phase.Combat_End))
1156                 return cbAIEndCombat.isSelected();
1157         } else {
1158             if (phase.equals(Constant.Phase.End_Of_Turn))
1159                 return cbHumanEndOfTurn.isSelected();
1160             else if (phase.equals(Constant.Phase.Upkeep))
1161                 return cbHumanUpkeep.isSelected();
1162             else if (phase.equals(Constant.Phase.Draw))
1163                 return cbHumanDraw.isSelected();
1164             else if (phase.equals(Constant.Phase.Combat_Begin))
1165                 return cbHumanBeginCombat.isSelected();
1166             else if (phase.equals(Constant.Phase.Combat_End))
1167                 return cbHumanEndCombat.isSelected();
