584     private Comparator<Card> bestCreature = new Comparator<Card>() {
585         public int compare(Card a, Card b) {
586             int cmcA = a.getCMC();
587             if (cmcA == 0)
588                 cmcA = 1;
589             cmcA *= 10;
590 
591             int cmcB = b.getCMC();
592             if (cmcB == 0)
593                 cmcB = 1;
594             cmcB *= 10;
595 
596             int attA = a.getBaseAttack() * 10;
597             int attB = b.getBaseAttack() * 10;
598 
599             int defA = a.getBaseDefense() * 10;
600             int defB = b.getBaseDefense() * 10;
601 
602             int keyA = a.getKeyword().size() * 10;
603             int keyB = b.getKeyword().size() * 10;
604 
605             int abA = a.getSpellAbility().length * 10;
606             int abB = b.getSpellAbility().length * 10;
607 
608             int trgA = a.getTriggers().size() * 10;
609             int trgB = b.getTriggers().size() * 10;
610
