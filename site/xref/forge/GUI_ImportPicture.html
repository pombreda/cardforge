156         if (jButtonSource == null) {
157             jButtonSource = new JButton();
158             jButtonSource.setBounds(new Rectangle(329, 45, 47, 17));
159             jButtonSource.setText("...");
160             jButtonSource.addMouseListener(new java.awt.event.MouseAdapter() {
161                 @Override
162                 public void mouseClicked(java.awt.event.MouseEvent e) {
163                     JFileChooser chooser;
164                     String choosertitle;
165                     choosertitle = "Select source directory.";
166                     chooser = new JFileChooser();
167                     chooser.setCurrentDirectory(new java.io.File("."));
168                     chooser.setDialogTitle(choosertitle);
169                     chooser.setFileSelectionMode(JFileChooser.DIRECTORIES_ONLY);
170 
171                     chooser.setAcceptAllFileFilterUsed(false);
172                     oldText = jLabelSource.getText();
173                     jLabelSource.setText("Please wait...");
174                     if (chooser.showOpenDialog(frame) == JFileChooser.APPROVE_OPTION) {
175                         FileFinder ff = new FileFinder();
176                         try {
177                             listFiles = ff.findFiles(chooser.getSelectedFile().toString(), ".+\\.jpg");
178                         } catch (Exception e2) {
179                             e2.printStackTrace();
180                         }
181                         jLabelSource.setText(chooser.getSelectedFile().toString());
182                         if (jCheckBox.isSelected()) {
183                             filesForCopy = ff.getFilesNumber();
184                             jLabelTotalFiles.setText("Total files for copying: " + filesForCopy);
185                             jLabelNeedSpace.setText("HDD Need Space: " + ff.getDirectorySize() / 1024 / 1024
186                                     + " MB");
187                             jProgressBar.setValue(0);
188                             if ((freeSpaceM > (ff.getDirectorySize() / 1024 / 1024)) && (filesForCopy > 0)) {
189                                 jButtonStart.setEnabled(true);
190                             }
191 
192                         } else {
193                             String fName;
194                             int start;
195                             long filesToCopySize;
196                             filesForCopy = 0;
197                             filesToCopySize = 0;
198                             fileCopyList = new ArrayList<File>();
199 
200                             for (int i = 0; i < listFiles.size(); i++) {
201 
202                                 fName = listFiles.get(i).getName();
203                                 start = fName.indexOf("full");
204                                 fName = fName.substring(0, start - 1)
205                                         + fName.substring(start + 4, fName.length() - 4);
206                                 fName = GuiDisplayUtil.cleanString(fName) + ".jpg";
207                                 File file = new File(ForgeProps.getFile(IMAGE_BASE), fName);
208                                 if (!file.exists()) {
209                                     filesForCopy = filesForCopy + 1;
210                                     filesToCopySize = filesToCopySize + listFiles.get(i).length();
211                                     fileCopyList.add(listFiles.get(i));
212                                 }
213                             }
214                             jLabelTotalFiles.setText("Total files for copying: " + filesForCopy);
215                             jLabelNeedSpace.setText("HDD Need Space: " + filesToCopySize / 1024 / 1024 + " MB");
216                             jProgressBar.setValue(0);
217                             if ((freeSpaceM > (filesToCopySize / 1024 / 1024)) && (filesForCopy > 0)) {
218                                 jButtonStart.setEnabled(true);
219                             }
220                         }
221 
222                     } else {
223                         if (oldText.equals("")) {
224                             jLabelSource.setText("");
225                         } else {
226                             jLabelSource.setText(oldText);
227                         }
228 
229                     }
230                 }
231             });
232         }
