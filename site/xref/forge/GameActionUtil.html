176             void DoCascade(Card c) {
177                 final Player controller = c.getController();
178                 final Card cascCard = c;
179 
180                 final Ability ability = new Ability(c, "0") {
181                     @Override
182                     public void resolve() {
183                         CardList topOfLibrary = AllZoneUtil.getPlayerCardsInLibrary(controller);
184                         CardList revealed = new CardList();
185 
186                         if (topOfLibrary.size() == 0) return;
187 
188                         Card cascadedCard = null;
189                         Card crd;
190                         int count = 0;
191                         while (cascadedCard == null) {
192                             crd = topOfLibrary.get(count++);
193                             revealed.add(crd);
194                             if ((!crd.isLand() && CardUtil.getConvertedManaCost(crd.getManaCost()) < CardUtil.getConvertedManaCost(cascCard.getManaCost())))
195                                 cascadedCard = crd;
196 
197                             if (count == topOfLibrary.size()) break;
198 
199                         }//while
200                         GuiUtils.getChoiceOptional("Revealed cards:", revealed.toArray());
201 
202                         if (cascadedCard != null && !cascadedCard.isUnCastable()) {
203 
204                             if (cascadedCard.getController().isHuman()) {
205                                 StringBuilder title = new StringBuilder();
206                                 title.append(cascCard.getName()).append(" - Cascade Ability");
207                                 StringBuilder question = new StringBuilder();
208                                 question.append("Cast ").append(cascadedCard.getName()).append(" without paying its mana cost?");
209 
210                                 int answer = JOptionPane.showConfirmDialog(null, question.toString(), title.toString(), JOptionPane.YES_NO_OPTION);
211 
212                                 if (answer == JOptionPane.YES_OPTION) {
213                                     AllZone.getGameAction().playCardNoCost(cascadedCard);
214                                     revealed.remove(cascadedCard);
215                                 }
216                             } else //computer
217                             {
218                                 ArrayList<SpellAbility> choices = cascadedCard.getBasicSpells();
219 
220                                 for (SpellAbility sa : choices) {
221                                     if (sa.canPlayAI()) {
222                                         ComputerUtil.playStackFree(sa);
223                                         revealed.remove(cascadedCard);
224                                         break;
225                                     }
226                                 }
227                             }
228                         }
229                         revealed.shuffle();
230                         for (Card bottom : revealed) {
231                             AllZone.getGameAction().moveToBottomOfLibrary(bottom);
232                         }
233                     }
234                 };
235                 StringBuilder sb = new StringBuilder();
236                 sb.append(c).append(" - Cascade.");
237                 ability.setStackDescription(sb.toString());
238 
239                 AllZone.getStack().addSimultaneousStackEntry(ability);
240 
241             }
242         };
243         Cascade.execute();
244     }

329                                             }
330                                         }
