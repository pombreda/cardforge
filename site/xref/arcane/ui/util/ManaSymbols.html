27      static public void loadImages() {
28          String[] symbols = new String[]{"0", "1", "10", "11", "12", "15", "16", "2", "20", "2W", "2U", "2R", "2G", "2B", "3",
29                  "4", "5", "6", "7", "8", "9", "B", "BG", "BR", "G", "GU", "GW", "R", "RG", "RW", "S", "T", "U", "UB",
30                  "UR", "W", "WB", "WU", "PW", "PU", "PB", "PR", "PG", "X", "Y", "Z", "slash", "attack", "defend", "summonsick"};
31          for (String symbol : symbols)
32              manaImages.put(symbol, UI.getImageIcon("res/images/symbols-13/" + symbol + ".png").getImage());
33      }

35      /**
36       * <p>draw.</p>
37       *
38       * @param g a {@link java.awt.Graphics} object.
39       * @param manaCost a {@link java.lang.String} object.
40       * @param x a int.
41       * @param y a int.
42       */
43      static public void draw(Graphics g, String manaCost, int x, int y) {
44          if (manaCost.length() == 0) return;
45          manaCost = UI.getDisplayManaCost(manaCost);
46          StringTokenizer tok = new StringTokenizer(manaCost, " ");
47          while (tok.hasMoreTokens()) {
48              String symbol = tok.nextToken();
49              Image image = manaImages.get(symbol);
50              if (image == null) {
51                  Log.info("Symbol not recognized \"" + symbol + "\" in mana cost: " + manaCost);
52                  continue;
53              }
54              g.drawImage(image, x, y, null);
55              x += symbol.length() > 2 ? 10 : 14; // slash.png is only 10 pixels wide.
56          }
57      }

101     /**

103      *
104      * @param value a {@link java.lang.String} object.
105      * @param small a boolean.
106      * @return a {@link java.lang.String} object.
107      */
108     static public synchronized String replaceSymbolsWithHTML(String value, boolean small) {
109         if (small) {
110             value = value.replace("{C}", "<img src='file:res/images/symbols-11/C.png' width=13 height=11>");
111             return replaceSymbolsPattern.matcher(value).replaceAll("<img src='file:res/images/symbols-11/$1$2.png' width=11 height=11>");
112         } else {
113             value = value.replace("{slash}", "<img src='file:res/images/symbols-13/slash.png' width=10 height=13>");
114             value = value.replace("{C}", "<img src='file:res/images/symbols-13/C.png' width=16 height=13>");
115             return replaceSymbolsPattern.matcher(value).replaceAll("<img src='file:res/images/symbols-13/$1$2.png' width=13 height=13>");
116         }
