127     private void setImage(Image srcImage, Image srcImageBlurred) {
128         synchronized (imagePanel) {
129             imagePanel.setImage(srcImage, srcImageBlurred);
130             repaint();
131             for (CardPanel cardPanel : imageLoadListeners) {
132                 cardPanel.setImage(srcImage, srcImageBlurred);

135             imageLoadListeners.clear();

260         if (isSelected) {

322         imagePanel.setSize(cardWidth - borderSize * 2, cardHeight - borderSize * 2);

324         int fontHeight = Math.round(cardHeight * (27f / 680));
325         boolean showText = !imagePanel.hasImage() || (!isAnimationPanel && fontHeight < 12);
326         titleText.setVisible(showText);

328 
329         int titleX = Math.round(cardWidth * (20f / 480));
330         int titleY = Math.round(cardHeight * (9f / 680));
331         titleText.setBounds(cardXOffset + titleX, cardYOffset + titleY, cardWidth - titleX, cardHeight);
332 
333         Dimension ptSize = ptText.getPreferredSize();

337         ptText.setLocation(cardXOffset + ptX - TEXT_GLOW_SIZE / 2, cardYOffset + ptY - TEXT_GLOW_SIZE / 2);

339         if (isAnimationPanel || cardWidth < 200)
340             imagePanel.setScalingType(ScalingType.nearestNeighbor);

354     /**
355      * <p>setCardBounds.</p>
356      *
357      * @param x      a int.
358      * @param y      a int.
359      * @param width  a int.
360      * @param height a int.
361      */
362     public void setCardBounds(int x, int y, int width, int height) {
363         cardWidth = width;
364         cardHeight = height;
365         int rotCenterX = Math.round(width / 2f);
366         int rotCenterY = height - rotCenterX;
367         int rotCenterToTopCorner = Math.round(width * CardPanel.rotCenterToTopCorner);
368         int rotCenterToBottomCorner = Math.round(width * CardPanel.rotCenterToBottomCorner);
369         int xOffset = rotCenterX - rotCenterToBottomCorner;
370         int yOffset = rotCenterY - rotCenterToTopCorner;
371         cardXOffset = -xOffset;
372         cardYOffset = -yOffset;
373         width = -xOffset + rotCenterX + rotCenterToTopCorner;
374         height = -yOffset + rotCenterY + rotCenterToBottomCorner;
375         setBounds(x + xOffset, y + yOffset, width, height);
376     }
