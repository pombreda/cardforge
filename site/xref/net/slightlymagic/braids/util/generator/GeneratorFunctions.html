20  	/**
21  	 * Estimate the number of items in this generator by traversing all of its
22  	 * elements.
23  	 * 
24  	 * Note this only works on a generator that can be reinstantiated once it
25  	 * has been traversed.  This is only an estimate, because a generator's size 
26  	 * may vary been traversals. This is especially true if the generator 
27  	 * relies on external resources, such as a file system.
28  	 * 
29  	 * If you call this on an infinite generator, this method will never 
30  	 * return.
31  	 * 
32  	 * @return  the estimated number of items provided by this generator
33  	 */
34  	public static <T> long estimateSize(Generator<T> gen) {
35  		long result = 0;
36  		for (@SuppressWarnings("unused") T ignored : YieldUtils.toIterable(gen)) 
37  		{
38  			result++;
39  		}
40  		
41  		return result;
42  	}
