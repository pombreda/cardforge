27  	 * Estimate the number of items in this generator by traversing all of its
28  	 * elements.
29  	 * 
30  	 * Note this only works on a generator that can be reinstantiated once it
31  	 * has been traversed.  This is only an estimate, because a generator's size 
32  	 * may vary been traversals. This is especially true if the generator 
33  	 * relies on external resources, such as a file system.
34  	 * 
35  	 * If you call this on an infinite generator, this method will never 
36  	 * return.
37  	 * 
38  	 * @return  the estimated number of items provided by this generator
39  	 */
40  	public static <T> long estimateSize(Generator<T> gen) {
41  		long result = 0;
42  		for (@SuppressWarnings("unused") T ignored : YieldUtils.toIterable(gen)) 
43  		{
44  			result++;
45  		}
46  		
47  		return result;
48  	}

50  	/**
51  	 * Highly efficient means of filtering a long or infinite sequence.
52  	 * 
53  	 * @param <T> any type
54  	 * 
55  	 * @param predicate a Lambda (function) whose apply method takes an object 
56  	 * of type <T> and returns a Boolean. If it returns false or null, the 
57  	 * item from the inputGenerator is not yielded by this Generator;
58  	 * if predicate.apply returns true, then this Generator <i>does</i>
59  	 * yield the value. 
60  	 * 
61  	 * @param inputGenerator the sequence upon which we operate
62  	 * 
63  	 * @return a generator which produces a subset <= the inputGenerator
64  	 */
65  	public static <T> Generator<T> filterGenerator(
66  			final Lambda1<Boolean,T> predicate, final Generator<T> inputGenerator) 
67  	{
68  		Generator<T> result = new Generator<T>() {
69  
70  			@Override
71  			public void generate(final Yieldable<T> outputYield) {
72  
73  				Yieldable<T> inputYield = new Yieldable<T>() {
74  					Boolean pResult;
75  					
76  					@Override
77  					public void yield(T input) {
78  						pResult = predicate.apply(input); 
79  						if (pResult != null && pResult) {
80  							outputYield.yield(input);
81  						}
82  					}
83  				};
84  				
85  				inputGenerator.generate(inputYield);
86  			}
87  			
88  		};
89  		
90  		return result;
91  	}

93  	/**
94  	 * Highly efficient means of applying a transform to a long or infinite 
95  	 * sequence.
96  	 * 
97  	 * @param <T> any type
98  	 * 
99  	 * @param transform a Lambda (function) whose apply method takes an object 
100 	 * of type <T> and returns an object of the same type. This transforms
101 	 * the values from the inputGenerator into this Generator.
102 	 * 
103 	 * @param inputGenerator the sequence upon which we operate
104 	 * 
105 	 * @return a generator that yields transform.apply's return value for
106 	 * each item in the inputGenerator
107 	 */
108 	public static <T> Generator<T> transformGenerator(
109 			final Lambda1<T,T> transform, final Generator<T> inputGenerator) 
110 	{
111 		Generator<T> result = new Generator<T>() {
112 
113 			@Override
114 			public void generate(final Yieldable<T> outputYield) {
115 
116 				Yieldable<T> inputYield = new Yieldable<T>() {
117 					@Override
118 					public void yield(T input) {
119 						outputYield.yield(transform.apply(input));
120 					}
121 				};
122 				
123 				inputGenerator.generate(inputYield);
124 			}
125 			
126 		};
127 		
128 		return result;
129 	}

131 	/**
132 	 * Forces a generator to be completely evaluated into a temporary data
133 	 * structure, then returns the generator over that same structure.
134 	 * 
135 	 * This effectively returns the same Generator, but it is a faster one.
136 	 * This trades away heap space for reduced CPU intensity.  This is
137 	 * particuarly helpful if you know that a Generator is going to be
138 	 * totally evaluated more than once in the near future.
139 	 * 
140 	 * @param <T> inferred automatically
141 	 * 
142 	 * @param unevaluated  a Generator of T instances
143 	 * 
144 	 * @return  the equivalent Generator, except that the result's generate
145 	 * method can be invoked multiple times for fast results.
146 	 */
147 	public static <T> Generator<T> solidify(Generator<T> unevaluated) {
148         ArrayList<T> solidTmp = YieldUtils.toArrayList(unevaluated);
149         solidTmp.trimToSize();
150         return YieldUtils.toGenerator(solidTmp);
151 	}

153 	/**
154 	 * Select an item at random from a Generator; this causes the entire
155 	 * Generator to be evaluated once, but only once.
156 	 * 
157 	 * @param generator
158 	 *            the generator from which to select a random item
159 	 * 
160 	 * @return an item chosen at random from the generator; this may be null, if
161 	 *         the generator contains null items.
162 	 * 
163 	 * @throws NoSuchElementException
164 	 *             if the generator has no contents
165 	 */
166 	public static <T> T selectRandom(Generator<T> generator) 
167 		throws NoSuchElementException
168 	{
169 		/*
170 		 * This algorithm requires some explanation. Each time we encounter a
171 		 * new item from the generator, we determine via random chance if the
172 		 * item is the one we select. At the end of each iteration, we have a
173 		 * candidate, and we have a count of the number of items encountered so
174 		 * far. Each iteration has a 1/n chance of replacing the candidate with
175 		 * the current item, where n is the number of items encountered so far.
176 		 * This allows us to randomly select an item from the generated contents
177 		 * with an equal distribution; and we don't have to count the number of
178 		 * items first!
179 		 */
180 
181 		int n = 0;
182 		T candidate = null;
183 		
184 		for (T item : YieldUtils.toIterable(generator)) {
185 			n++;
186 			int rand = (int) (Math.random() * n);
187 			// At this point, 0 <= rand < n.
188 			rand++;  // Now, 1 <= rand <= n.
189 			
190 			if (rand == 1) {
191 				// We rolled a 1 on an n-sided die. We have a new candidate!
192 				// Note that on the first iteration, this always happens,
193 				// because n = 1.
194 				candidate = item;
195 			}
196 		}
197 
198 		if (n == 0) {
199 			// There were no items in the generator!
200 			throw new NoSuchElementException("generator is empty");
201 		}
202 
203 		return candidate;
204 	}
