2   package net.slightlymagic.braids.util.generator;
3   
4   import java.io.File;
5   
6   import com.google.code.jyield.Generator;
7   import com.google.code.jyield.Yieldable;
8   
9   /**
10   * This is a generator over all of the non-directories residing in a given
11   * starting directory and all subdirectories of it that do NOT start with a 
12   * dot; this prevents the code from descending into .svn directories.
13   * 
14   * For documentation on Java-Yield and its generators, see
15   * {@link http://code.google.com/p/java-yield/}
16   */
17  public class FindNonDirectoriesSkipDotDirectoriesGenerator implements Generator<File> {
18  	private File startDir;
19  
20  	/**
21  	 * Create a generator at a given starting directory.
22  	 * 
23  	 * One can invoke this generator more than once by calling its generate
24  	 * method.
25  	 * 
26  	 * @param startDir  the directory to start in; we ignore this directory's
27  	 * name, so if it starts with a dot, we treat it as if it didn't.
28  	 */
29  	public FindNonDirectoriesSkipDotDirectoriesGenerator(File startDir) {
30  		this.startDir = startDir;
31  	}
32  
33  	/**
34  	 * Standard generate method.
35  	 * 
36  	 * <p>Yields results to the given Yieldable.  Convert Generator instances to
37  	 * Iterables with YieldUtils.toIterable.</p>
38  	 * 
39  	 * See {@link https://java-yield.googlecode.com/hg/docs/com/google/code/jyield/YieldUtils.html#toIterable(com.google.code.jyield.Generator)}	
40  	 */
41  	public void generate(Yieldable<File> yy) {
42  		String[] list = startDir.list();
43  		
44  		for (String filename : list) {
45  			File entry = new File(startDir, filename);
46  			
47  			if (entry.isDirectory()) {
48  				if (!filename.startsWith(".")) {
49  					FindNonDirectoriesSkipDotDirectoriesGenerator child = new FindNonDirectoriesSkipDotDirectoriesGenerator(entry);
50  					child.generate(yy);
51  					child = null;
52  				}
53  				// else do nothing, because it's a dot directory
54  			}
55  			else {
56  				// Use this instead of a return statement.
57  				yy.yield(entry);
58  			}
59  		}
60  	}
61  }
