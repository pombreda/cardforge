199     /**
200      * {@inheritDoc}
201      *
202      * Calls the UI delegate's paint method, if the UI delegate
203      * is non-<code>null</code>.  We pass the delegate a copy of the
204      * <code>Graphics</code> object to protect the rest of the
205      * paint code from irrevocable changes
206      * (for example, <code>Graphics.translate</code>).
207      * <p/>
208      * If you override this in a subclass you should not make permanent
209      * changes to the passed in <code>Graphics</code>. For example, you
210      * should not alter the clip <code>Rectangle</code> or modify the
211      * transform. If you need to do these operations you may find it
212      * easier to create a new <code>Graphics</code> from the passed in
213      * <code>Graphics</code> and manipulate it. Further, if you do not
214      * invoker super's implementation you must honor the opaque property,
215      * that is
216      * if this component is opaque, you must completely fill in the background
217      * in a non-opaque color. If you do not honor the opaque property you
218      * will likely see visual artifacts.
219      * <p/>
220      * The passed in <code>Graphics</code> object might
221      * have a transform other than the identify transform
222      * installed on it.  In this case, you might get
223      * unexpected results if you cumulatively apply
224      * another transform.
225      */
226     @Override
227     protected void paintComponent(Graphics g) {
228         if (backgroundPainter != null) {
229             Graphics2D g2 = (Graphics2D) g.create();
230 
231             try {
232                 Insets ins = this.getInsets();
233                 g2.translate(ins.left, ins.top);
234                 backgroundPainter.paint(g2, this, this.getWidth() - ins.left
235                         - ins.right, this.getHeight() - ins.top - ins.bottom);
236             } finally {
237                 g2.dispose();
238             }
239         } else {
240             super.paintComponent(g);
241         }
242     }

274     /**
275      * Uses the DividerPainter (if any) to paint each Divider that
276      * overlaps the clip Rectangle.  This is done after the call to
277      * <code>super.paintChildren()</code> so that Dividers can be
278      * rendered "on top of" the children.
279      * <p/>
280      * {@inheritDoc}
281      */
282     @Override
283     protected void paintChildren(Graphics g) {
284         super.paintChildren(g);
285         DividerPainter dp = getDividerPainter();
286         Rectangle clipR = g.getClipBounds();
287         if ((dp != null) && (clipR != null)) {
288             MultiSplitLayout msl = getMultiSplitLayout();
289             if (msl.hasModel()) {
290                 for (Divider divider : msl.dividersThatOverlap(clipR)) {
291                     Rectangle bounds = divider.getBounds();
292                     Graphics cg = g.create(bounds.x, bounds.y, bounds.width, bounds.height);
293                     try {
294                         dp.paint((Graphics2D) cg, divider, bounds.width, bounds.height);
295                     } finally {
296                         cg.dispose();
297                     }
298                 }
299             }
300         }
301     }

303     private boolean dragUnderway = false;
304     private MultiSplitLayout.Divider dragDivider = null;
305     private Rectangle initialDividerBounds = null;
306     private boolean oldFloatingDividers = true;
307     private int dragOffsetX = 0;
308     private int dragOffsetY = 0;
309     private int dragMin = -1;
310     private int dragMax = -1;

312     /**
313      * <p>startDrag.</p>
314      *
315      * @param mx a int.
316      * @param my a int.
317      */
318     private void startDrag(int mx, int my) {
319         requestFocusInWindow();
320         MultiSplitLayout msl = getMultiSplitLayout();
321         MultiSplitLayout.Divider divider = msl.dividerAt(mx, my);
322         if (divider != null) {
323             MultiSplitLayout.Node prevNode = divider.previousSibling();
324             MultiSplitLayout.Node nextNode = divider.nextSibling();
325             if ((prevNode == null) || (nextNode == null)) {
326                 dragUnderway = false;
327             } else {
328                 initialDividerBounds = divider.getBounds();
329                 dragOffsetX = mx - initialDividerBounds.x;
330                 dragOffsetY = my - initialDividerBounds.y;
331                 dragDivider = divider;
332 
333                 Rectangle prevNodeBounds = prevNode.getBounds();
334                 Rectangle nextNodeBounds = nextNode.getBounds();
335                 if (dragDivider.isVertical()) {
336                     dragMin = prevNodeBounds.x;
337                     dragMax = nextNodeBounds.x + nextNodeBounds.width;
338                     dragMax -= dragDivider.getBounds().width;
339                     if (msl.getLayoutMode() == MultiSplitLayout.USER_MIN_SIZE_LAYOUT)
340                         dragMax -= msl.getUserMinSize();
341                 } else {
342                     dragMin = prevNodeBounds.y;
343                     dragMax = nextNodeBounds.y + nextNodeBounds.height;
344                     dragMax -= dragDivider.getBounds().height;
345                     if (msl.getLayoutMode() == MultiSplitLayout.USER_MIN_SIZE_LAYOUT)
346                         dragMax -= msl.getUserMinSize();
347                 }
348 
349                 if (msl.getLayoutMode() == MultiSplitLayout.USER_MIN_SIZE_LAYOUT) {
350                     dragMin = dragMin + msl.getUserMinSize();
351                 } else {
352                     if (dragDivider.isVertical()) {
353                         dragMin = Math.max(dragMin, dragMin + getMinNodeSize(msl, prevNode).width);
354                         dragMax = Math.min(dragMax, dragMax - getMinNodeSize(msl, nextNode).width);
355 
356                         Dimension maxDim = getMaxNodeSize(msl, prevNode);
357                         if (maxDim != null)
358                             dragMax = Math.min(dragMax, prevNodeBounds.x + maxDim.width);
359                     } else {
360                         dragMin = Math.max(dragMin, dragMin + getMinNodeSize(msl, prevNode).height);
361                         dragMax = Math.min(dragMax, dragMax - getMinNodeSize(msl, nextNode).height);
362 
363                         Dimension maxDim = getMaxNodeSize(msl, prevNode);
364                         if (maxDim != null)
365                             dragMax = Math.min(dragMax, prevNodeBounds.y + maxDim.height);
366                     }
367                 }
368 
369                 oldFloatingDividers = getMultiSplitLayout().getFloatingDividers();
370                 getMultiSplitLayout().setFloatingDividers(false);
371                 dragUnderway = true;
372             }
373         } else {
374             dragUnderway = false;
375         }
376     }

415     /**
416      * <p>repaintDragLimits.</p>
417      */
418     private void repaintDragLimits() {
419         Rectangle damageR = dragDivider.getBounds();
420         if (dragDivider.isVertical()) {
421             damageR.x = dragMin;
422             damageR.width = dragMax - dragMin;
423         } else {
424             damageR.y = dragMin;
425             damageR.height = dragMax - dragMin;
426         }
427         repaint(damageR);
428     }

430     /**
431      * <p>updateDrag.</p>
432      *
433      * @param mx a int.
434      * @param my a int.
435      */
436     private void updateDrag(int mx, int my) {
437         if (!dragUnderway) {
438             return;
439         }
440         Rectangle oldBounds = dragDivider.getBounds();
441         Rectangle bounds = new Rectangle(oldBounds);
442         if (dragDivider.isVertical()) {
443             bounds.x = mx - dragOffsetX;
444             bounds.x = Math.max(bounds.x, dragMin);
445             bounds.x = Math.min(bounds.x, dragMax);
446         } else {
447             bounds.y = my - dragOffsetY;
448             bounds.y = Math.max(bounds.y, dragMin);
449             bounds.y = Math.min(bounds.y, dragMax);
450         }
451         dragDivider.setBounds(bounds);
452         if (isContinuousLayout()) {
453             revalidate();
454             repaintDragLimits();
455         } else {
456             repaint(oldBounds.union(bounds));
457         }
458     }

463     private void clearDragState() {
464         dragDivider = null;

472     /**
473      * <p>finishDrag.</p>
474      *
475      * @param x a int.
476      * @param y a int.
477      */
478     private void finishDrag(int x, int y) {
479         if (dragUnderway) {
480             clearDragState();
481             if (!isContinuousLayout()) {
482                 revalidate();
483                 repaint();
484             }
485         }
486         setCursor(Cursor.getPredefinedCursor(Cursor.DEFAULT_CURSOR));
487     }

489     /**
490      * <p>cancelDrag.</p>
491      */
492     private void cancelDrag() {
493         if (dragUnderway) {
494             dragDivider.setBounds(initialDividerBounds);
495             getMultiSplitLayout().setFloatingDividers(oldFloatingDividers);
496             setCursor(Cursor.getPredefinedCursor(Cursor.DEFAULT_CURSOR));
497             repaint();
498             revalidate();
499             clearDragState();
500         }
501     }

503     /**
504      * <p>updateCursor.</p>
505      *
506      * @param x a int.
507      * @param y a int.
508      * @param show a boolean.
509      */
510     private void updateCursor(int x, int y, boolean show) {
511         if (dragUnderway) {
512             return;
513         }
514         int cursorID = Cursor.DEFAULT_CURSOR;
515         if (show) {
516             MultiSplitLayout.Divider divider = getMultiSplitLayout().dividerAt(x, y);
517             if (divider != null) {
518                 cursorID = (divider.isVertical()) ?
519                         Cursor.E_RESIZE_CURSOR :
520                         Cursor.N_RESIZE_CURSOR;
521             }
522         }
523         setCursor(Cursor.getPredefinedCursor(cursorID));
524     }
