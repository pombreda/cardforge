21  /**
22   * The class TreeProperties. This class allows for a tree-like structure of properties-files. This class lays some
23   * restrictions on the keys used in properties-files:
24   * <ul>
25   * <li>The use of "--" is forbidden</li>
26   * <li>The suffixes "--properties" and "--transparent-properties" are reserved for specifying additional
27   * properties-files in the tree (relative paths are relative to the properties-file where they are referenced)</li>
28   * <li>Other suffixes are used by {@link PropertyType}s. PropertyTypes are registered or unregistered using
29   * {@link #addType(PropertyType)} or {@link #removeType(PropertyType)}.</li>
30   * </ul>
31   * Take a look at these files:
32   * <p/>
33   * <pre>
34   * #root.properties
35   * title=directions
36   * icons--properties=img/icons.properties
37   *
38   * #img/icons.properties
39   * left--file=left.jpg
40   * right--file=right.jpg
41   * up--file=up.jpg
42   * down--file=down.jpg
43   * #note that the path does not contain &quot;img/&quot;
44   * size--transparent-properties=size.properties
45   *
46   * #img/size.properties
47   * width=20
48   * height=20
49   * </pre>
50   * <p/>
51   * These properties are retrieved with
52   * <p/>
53   * <pre>
54   * getProperty(&quot;title&quot;) //directions
55   * getFile(&quot;icons/left&quot;) //img/left.jpg
56   * getFile(&quot;icons/right&quot;) //img/right.jpg
57   * getFile(&quot;icons/up&quot;) //img/up.jpg
58   * getFile(&quot;icons/down&quot;) //img/down.jpg
59   * getProperty(&quot;icons/width&quot;) //20
60   * getProperty(&quot;icons/height&quot;) //20
61   * </pre>
62   * <p/>
63   * As you can see, a properties file included with "--transparent-properties" hides its existence from the user. A
64   * file included with "--properties" is not hidden. The child properties are accessible as if their keys were
65   * prepended with the parent key, separated by a slash.
66   * <p/>
67   * Note that --file, --properties and --transparent-properties entries will be relative to the included file, even
68   * if the properties file is transparent.
69   * <p/>
70   * Also, the TreeProperties can be retrieved:
71   * <p/>
72   * <pre>
73   * getChildProperties(&quot;icons&quot;)
74   * getTransparentProperties(&quot;icons/size&quot;)
75   * </pre>
76   * <p/>
77   * <p/>
78   * TODO add edit & save support
79   *
80   * @author Clemens Koza
81   * @version V0.0 19.08.2009
82   * @see Properties
83   */

151     /**
152      * The constructor is forgiving in the way that Exceptions are not directly forwarded. The only fatal exception
153      * is if the parameter is null or not found. Instead, the rest of the properties are processed, so that the
154      * erroneous properties are the only ones not present in this TreeProperties. Afterwards, the exceptions can be
155      * accessed.
156      *
157      * @param f a {@link java.io.File} object.
158      * @throws java.io.IOException if any.
159      */
160     public TreeProperties(File f) throws IOException {
161         if (f == null) throw new FileNotFoundException("null");
162         this.path = f.getParentFile();
163         instanceTypes = new HashMap<Class<?>, PropertyType<?>>(types);
164         instanceSuffixes = new HashMap<String, PropertyType<?>>(suffixes);
165         Properties p = new Properties();
166 
167 //        BufferedReader r = new BufferedReader(new FileReader(f));
168 //        p.load(r);
169 //        r.close();
170         BufferedInputStream is = new BufferedInputStream(new FileInputStream(f));
171         p.load(is);
172         is.close();
173 
174         Set<Entry<Object, Object>> entries = p.entrySet();
175         properties = new HashMap<String, Object>();
176         List<Exception> exceptions = new ArrayList<Exception>();
177         this.exceptions = unmodifiableList(exceptions);
178         for (Entry<Object, Object> entry : entries) {
179             String key = (String) entry.getKey();
180             String value = (String) entry.getValue();
181             try {
182                 String[] parts = key.split("--", 2);
183                 Object result;
184                 if (parts.length == 1) {
185                     //Regular
186                     result = value;
187                 } else {
188                     //suffix
189                     if (parts[1].equals(transparent) || parts[1].equals(child)) {
190                         TreeProperties child = new TreeProperties(path, FileType.getPath(value));
191                         exceptions.addAll(child.exceptions);
192                         result = child;
193                     } else {
194                         PropertyType<?> t = instanceSuffixes.get(parts[1]);
195                         if (t == null) throw new IllegalArgumentException("No content type: " + parts[1]);
196                         result = t.toObject(this, value);
197                     }
198                 }
199                 properties.put(key, result);
200             } catch (Exception ex) {
201                 exceptions.add(new Exception(format("File '%s', Property '%s':%n    %s", f, key, ex.getMessage()),
202                         ex.getCause()));
203             }
204         }
205     }

299     @SuppressWarnings("unchecked")
300     public <T> T getProperty(String key, Class<T> cls) {
301         String suffix;
302         if (cls == String.class) suffix = "";
303         else {
304             PropertyType<?> t = instanceTypes.get(cls);
305             suffix = "--" + t.getSuffix();
306         }
307         return (T) getProperty(key, suffix, true);

310     /**
311      * <p>getProperty.</p>
312      *
313      * @param key a {@link java.lang.String} object.
314      * @param suffix a {@link java.lang.String} object.
315      * @param top a boolean.
316      * @return a {@link java.lang.Object} object.
317      */
318     private Object getProperty(String key, String suffix, boolean top) {
319         checkQueryKey(key);
320         //first, try the key in the current file, as if there were no slash
321         //No subpath - either directly in the properties...
322         Object result;
323         if ((result = properties.get(key + suffix)) != null) return result;
324 
325         //...or in a transparent properties
326 
327         //look for all --transparent-properties
328         for (Entry<String, Object> entry : properties.entrySet()) {
329             if (entry.getKey().endsWith("--" + transparent)) {
330                 TreeProperties p = (TreeProperties) entry.getValue();
331                 if ((result = p.getProperty(key, suffix, false)) != null) {
332                     return result;
333                 }
334             }
335         }
336 
337         //if there is no immediate containment, try the children
338         //try every combination
339         //for a/b/c, there could be a child "a/b--properties" that contains "c"
340         //or "a--properties" with "b/c"
341         int index = -1;
342         while ((index = key.indexOf('/', index + 1)) != -1) {
343             String first = key.substring(0, index), second = key.substring(index + 1);
344 
345             TreeProperties p = (TreeProperties) getProperty(first, "--" + child, false);
346             if (p == null) continue;
347             if ((result = p.getProperty(second, suffix, false)) != null) {
348                 return result;
349             }
350         }
351         if (top) {
352             Exception ex = new Exception("TreeProperties returns null for " + key + suffix);
353 //        ex.printStackTrace();
354             System.err.println(ex);
355         }
356         return null;
357     }

381     private final class TreePropertiesIterator implements Iterator<PropertyElement> {
382         private final String prefix;
383         private Iterator<Entry<String, Object>> entries;
384         private TreePropertiesIterator child;
385         private PropertyElement next;
386 
387         private TreePropertiesIterator(String prefix) {
388             entries = properties.entrySet().iterator();
389             this.prefix = prefix;
390         }
391 
392         //After this call, the next element is determined, or the child iterator has next
393 
394         public boolean hasNext() {
395             if (next != null) return true;
396             else if (child != null && child.hasNext()) return true;
397             else if (entries.hasNext()) {
398                 Entry<String, Object> entry = entries.next();
399                 final String[] parts = entry.getKey().split("--");
400                 final Class<?> cls;
401                 final Object value = entry.getValue();
402 
403                 if (parts.length == 1) {
404                     cls = String.class;
405                 } else if (parts[1].equals(transparent)) {
406                     child = ((TreeProperties) entry.getValue()).iterator(prefix);
407                     //recursive, for the case that the child iterator is empty
408                     return hasNext();
409                 } else if (parts[1].equals(TreeProperties.child)) {
410                     child = ((TreeProperties) entry.getValue()).iterator(prefix + parts[0] + "/");
411                     //recursive, for the case that the child iterator is empty
412                     return hasNext();
413                 } else {
414                     //class is determined by the content type
415                     PropertyType<?> t = instanceSuffixes.get(parts[1]);
416                     cls = t.getType();
417                 }
418                 next = new PropertyElement() {
419 
420                     public String getKey() {
421                         return prefix + parts[0];
422                     }
423 
424 
425                     public Class<?> getType() {
426                         return cls;
427                     }
428 
429 
430                     public Object getValue() {
431                         return value;
432                     }
433 
434 
435                     public void setValue(String value) {
436                     }
437                 };
438                 return true;
439             } else return false;
440         }
441 
442 
443         public PropertyElement next() {
444             if (!hasNext()) throw new NoSuchElementException();
445             else if (next != null) {
446                 PropertyElement next = this.next;
447                 this.next = null;
448                 return next;
449             } else return child.next();
450         }

454             throw new UnsupportedOperationException();
455         }
